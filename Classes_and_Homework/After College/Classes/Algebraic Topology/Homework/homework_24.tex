\section{Revised Problems}%
\begin{problem}
  Prove that $\widetilde{H}_q(X) = H_q(X)$ for $q\neq 0$ and $H_0(X) = \widetilde{H}_0(X)\oplus \Z$ for any space $X$. 
\end{problem}
\begin{solution}
  For any $q\neq -1$, we have that $\widetilde{C}_q(X) = C_q(X)$, so since the boundary maps $\widetilde{\partial_q}$ in reduced homology are identical to the boundary maps $\partial_q$ in singular homology for any $q > 1$, so the boundaries $B_q$ and cycles $Z_q$ are identical for any $q\geq 1$, meaning $\widetilde{H}_q(X) = H_q(X)$ for all $q\neq 0$.

  Concerning $\widetilde{H}_0(X)$, we start by observing that $\ve\colon C_0\left( X \right)\rightarrow \Z$ is a surjective map with kernel $\ker\left( \ve \right)$. Since $C_0(X)$ is the free $\Z$-module on the singular $0$-simplices, and $\Z$ is a free $\Z$-module, it follows from the universal property of free modules that $\ve$ admits a splitting, so that $C_0(X)\cong \img\left( \ve \right)\oplus \ker\left( \ve \right)$. Since $\img\left( \ve \right) = \Z$, and $Z_0(X) = C_0(X)$ as $\partial_0$ is the zero map, we have
  \begin{align*}
    H_0(X) &= Z_0(X)/B_0(X)\\
           &= C_0(X)/B_0(X)\\
           &= \left( \ker\left( \ve \right) \oplus \img\left( \ve \right) \right)/\left( B_0(X) \right)\\
           &= \ker\left( \ve \right)/B_0(X)\oplus \img\left( \ve \right)\\
           &= \widetilde{C}_0(X)/B_0(X)\oplus \Z\\
           &= \widetilde{Z}_0(X)/B_0(X)\oplus \Z\\
           &= \widetilde{H}_0(X)\oplus \Z.
  \end{align*}
\end{solution}

\begin{problem}[Problem 1]\hfill
  \begin{enumerate}[(a)]
    \item Prove that $ \widetilde{H}_q\left( X,A \right) = H_q\left( X,A \right) $ for all $q\geq 0$.
    \item Let $x_0\in X$. Then, $\widetilde{H}_q(X)\cong H_q\left( X,\set{x_0} \right)$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Since $C_q\left( X \right)\cong \widetilde{C}_q\left( X \right)$ and $C_q\left( A \right)\cong \widetilde{C}_q\left( A \right)$ for any $q > 0$, we have that the short exact sequence of chain complexes
      \begin{align*}
        0\rightarrow \widetilde{C}_{\ast}\left( A \right)\rightarrow \widetilde{C}_{\ast}\left( X \right)\rightarrow \widetilde{C}_{\ast}\left( X,A \right)\rightarrow 0
      \end{align*}
      induces the same long exact sequence of homology groups as
      \begin{align*}
        0 \rightarrow C_{\ast}\left( A \right)\rightarrow C_{\ast}\left( X \right)\rightarrow C_{\ast}\left( X,A \right)\rightarrow 0
      \end{align*}
      whenever $q > 0$.

      As for the case of $q = 0$, we use the fact that the augmentation homomorphism $\ve$ respects both $C_0\left( X \right)$ and $C_0\left( A \right)$, meaning that in degree $0$, our short exact sequence of chain complexes reads as
      \begin{align*}
        0 \rightarrow \widetilde{C}_0\left( A \right)\rightarrow \widetilde{C}_0\left( X \right)\rightarrow C_0\left( X,A \right)\rightarrow 0.
      \end{align*}
      I am not exactly sure how to progress from this point.
    \item Using the long exact sequence, we have
      \begin{align*}
        \cdots\rightarrow H_q\left( \set{x_0} \right) \rightarrow H_q\left( X \right) \rightarrow H_q\left( X,\set{x_0} \right)\rightarrow H_{q-1}\left( \set{x_0} \right)\rightarrow\cdots
      \end{align*}
      meaning that we have $H_q\left( X \right)\cong H_q\left( X,\set{x_0} \right)$, so since for any $q > 1$, $ \widetilde{H}_q\left( X \right)\cong H_q(X) $, we obtain our desired result in this case. Concentrating on $q = 1$ and $q = 0$, we get
      \begin{align*}
        0 \rightarrow H_1\left( X \right)\xrightarrow{j_{\ast}} H_1\left( X,\set{x_0} \right)\xrightarrow{\partial} \Z\xrightarrow{i_{\ast}} H_0\left( X \right)\xrightarrow{j_{\ast}} H_0\left( X,\set{x_0} \right)\rightarrow 0.
      \end{align*}
      By exactness, we have that $j_{\ast}$ at level $0$ is injective, giving
      \begin{align*}
        H_{0}\left( X,\set{x_0} \right) &\cong H_0\left( X \right)/\ker\left( j_{\ast} \right)\\
                                        &= H_0\left( X \right)/\img\left( i_{\ast} \right).
      \end{align*}
      Similarly, $j_{\ast}$ at level $1$ is injective, meaning that
      \begin{align*}
        H_1\left( X \right) &= \img\left( j_{\ast} \right)\\
                            &\cong \ker\left( \partial \right).
      \end{align*}
      I am not exactly sure how to progress from this point.
  \end{enumerate}
\end{solution}
