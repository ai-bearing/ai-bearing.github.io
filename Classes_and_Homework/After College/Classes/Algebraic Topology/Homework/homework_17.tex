\begin{problem}[Problem 1]
  Give two examples of nontrivial connected covering spaces of the Klein bottle, one that is normal and one that is not.
\end{problem}
\begin{solution}
  We observe that the torus can be viewed as a $2$-sheeted cover of the Klein bottle as seen below. Since the number of sheets in the cover is equal to the index of $p_{\ast}\left( \pi_1\left( \widetilde{X},\widetilde{x_0} \right) \right)\subseteq \pi_1\left( X,x_0 \right)$, we have that this is a normal cover.
\end{solution}
\begin{problem}[Problem 3]
  Let $G$ be a group, $N\trianglelefteq G$ a normal subgroup. Prove that there exists a normal covering space $\widetilde{X}\rightarrow X$ with $\pi_1(X) = G$, $\pi_1\left( \widetilde{X} \right) = N$, and deck transformations $ G\left( \widetilde{X} \right) = G/N $.
\end{problem}
\begin{solution}
  Write $G$ in terms of generators and relations,
  \begin{align*}
    G &= \left\langle \set{g_i}_{i\in I} | \set{r_j}_{j\in J} \right\rangle.
  \end{align*}
  This allows us to construct a CW complex $X$ by taking $1$-cells $e^1_i$ for each $i\in I$, and attaching maps $\varphi_j\colon e^2_j\rightarrow X^1$ via the relations $\left[ \varphi_j \right] = e$.

  From the Galois correspondence, we know that there is some covering space $ p\colon \widetilde{X}\rightarrow X $ corresponding to $N$ which is necessarily a normal covering by the definition of a normal covering space. We then have that $\pi_1\left( X \right) = G$, $\pi_1\left( \widetilde{X} \right) = N$, and
  \begin{align*}
    G\left( \widetilde{X} \right) &= \pi_1(X)/\pi_1\left( \widetilde{X} \right)\\
                                  &= G/N.
  \end{align*}
\end{solution}
\begin{problem}[Problem 4]
  Prove that the only covering space $ \mathbb{RP}^2\rightarrow X $ is the identity map on $ \mathbb{RP}^2 $.
\end{problem}
\begin{solution}
  We know that if $p\colon \widetilde{X}\rightarrow X$ is a $k$-sheeted of $2$-manifolds, then we have
  \begin{align*}
    \chi\left( \widetilde{X} \right) &= k\chi\left( X \right).
  \end{align*}
  Yet, since $ \chi\left( \mathbb{RP}^2 \right) = 1 $, it follows that if $\widetilde{X} = \mathbb{RP}^2$, then this covering space on $X$ can only have one sheet, so that the only covering space $p\colon \mathbb{RP}^2 \rightarrow X$ is the identity map on $ \mathbb{RP}^2 $.
\end{solution}
