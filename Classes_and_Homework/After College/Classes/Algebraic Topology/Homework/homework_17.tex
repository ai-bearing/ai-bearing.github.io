\begin{problem}[Problem 1]
  Give two examples of nontrivial connected covering spaces of the Klein bottle, one that is normal and one that is not.
\end{problem}
\begin{solution}
  We observe that the torus can be viewed as a $2$-sheeted cover of the Klein bottle as seen below. Since the number of sheets in the cover is equal to the index of $p_{\ast}\left( \pi_1\left( \widetilde{X},\widetilde{x_0} \right) \right)\subseteq \pi_1\left( X,x_0 \right)$, we have that this is a normal cover.
\end{solution}
\begin{problem}[Problem 2]
  Let $p\colon \widetilde{X}\rightarrow X$ be a covering space with $X$ path-connected, locally path-connected, and semilocally simply connected, but $\widetilde{X}$ is not necessarily path-connected. Prove that the path-components of $ \widetilde{X} $ are in bijection with the orbits of the action of $\pi_1\left( X,x_0 \right)$ on the fiber $p^{-1}\left( x_0 \right)$.
\end{problem}
\begin{solution}
  The action of $\pi_1\left( X,x_0 \right)$ on the fiber $p^{-1}\left( x_0 \right)$ gives a partition of $p^{-1}\left( x_0 \right)$ into orbits. Suppose $O_1$ is one of these orbits, and let $y_1$ be a representative for $O_1$, with $Y_1\subseteq \widetilde{X}$ its corresponding path-component. Restricting $p$ to $Y_1$, we have a covering map $p|_{Y_1}\colon Y_1\rightarrow X$, whence $\pi_1\left( X,x_0 \right)$ acts transitively on $p^{-1}\left( x_0 \right)|_{Y_1}$ since $X$ is path-connected, locally path-connected, and semilocally simply connected, meaning that there is an injection
  \begin{align*}
    \set{\text{orbits of action of $\pi_1\left( X,x_0 \right)$}} \rightarrow \set{\text{path-components of }\widetilde{X}}.
  \end{align*}
  Finally, since every path-component contains an element of the fiber $ p^{-1}\left( x_0 \right) $, we have an injection
  \begin{align*}
    \set{\text{path-components of }\widetilde{X}}\rightarrow \set{\text{orbits of action of $\pi_1\left( X,x_0 \right)$}}
  \end{align*}
  determined by picking representatives. Thus, by Cantor--Schröder--Bernstein, we have a bijection between orbits of the action of $\pi_1\left( X,x_0 \right)$ on $p^{-1}\left( x_0 \right)$ and path-components of $ \widetilde{X} $.
\end{solution}
\begin{problem}[Problem 3]
  Let $G$ be a group, $N\trianglelefteq G$ a normal subgroup. Prove that there exists a normal covering space $\widetilde{X}\rightarrow X$ with $\pi_1(X) = G$, $\pi_1\left( \widetilde{X} \right) = N$, and deck transformations $ G\left( \widetilde{X} \right) = G/N $.
\end{problem}
\begin{solution}
  Write $G$ in terms of generators and relations,
  \begin{align*}
    G &= \left\langle \set{g_i}_{i\in I} | \set{r_j}_{j\in J} \right\rangle.
  \end{align*}
  This allows us to construct a CW complex $X$ by taking $1$-cells $e^1_i$ for each $i\in I$, and attaching maps $\varphi_j\colon e^2_j\rightarrow X^1$ via the relations $\left[ \varphi_j \right] = e$.

  From the Galois correspondence, we know that there is some covering space $ p\colon \widetilde{X}\rightarrow X $ corresponding to $N$ which is necessarily a normal covering by the definition of a normal covering space. We then have that $\pi_1\left( X \right) = G$, $\pi_1\left( \widetilde{X} \right) = N$, and
  \begin{align*}
    G\left( \widetilde{X} \right) &= \pi_1(X)/\pi_1\left( \widetilde{X} \right)\\
                                  &= G/N.
  \end{align*}
\end{solution}
