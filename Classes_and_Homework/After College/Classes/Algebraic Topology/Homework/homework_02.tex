\begin{problem}[Problem 1]
  Prove that a space $X$ is contractible if and only if the identity map is null-homotopic.
\end{problem}
\begin{solution}
  Suppose $X$ is contractible, so there are functions $f\colon X\rightarrow \ast$ and $g\colon \ast\rightarrow X$ such that $f\circ g = \id_{\ast}$ and $g\circ f \simeq \id_X$. Since the domain of $g$ is one-point, it follows that $g\circ f = c$ for some constant map $c$. Since homotopy is an equivalence relation, we have $\id_{X}\simeq c$, so the identity map on $X$ is null-homotopic.

  Now, suppose the identity map on $X$ is null-homotopic. That is, there is a constant map $c$ such that $c\simeq \id_X$. We let $w$ be the point for which $c(x) = w$ for all $x\in X$. We may define the map $g\colon \ast\rightarrow X$ by taking $\ast\mapsto w$, so if $f\colon X\rightarrow \ast$ is the unique map that takes $x\in X$ to $\ast$, then $f\circ g = \id_{\ast}$ and $g\circ f = c\simeq \id_X$, meaning that $X$ has the homotopy type of $\ast$.
\end{solution}
\begin{problem}[Problem 2]
  Prove that a retract of a contractible space is contractible.
\end{problem}
\begin{solution}
  Let $A\subseteq X$ be a retract, with $r\colon X\rightarrow X$ the relevant retraction. That is, $r|_{A} = \id_A$, and $r(X) = A$.

  Now, since $X$ is contractible, there are maps $f\colon X\rightarrow \ast$ and $g\colon \ast\rightarrow X$ such that $f\circ g = \id_{\ast}$ and $g\circ f \simeq \id_X$. We may now define
  \begin{align*}
    \bar{f} \colon A\rightarrow \ast\\
    \bar{g}\colon \ast\rightarrow A
  \end{align*}
  by taking $\bar{f} = f|_{A}$ and $\bar{g} = r\circ g$. We observe that $\bar{f}\circ \bar{g} = \id_{\ast}$, and that
  \begin{align*}
    \bar{g}\circ \bar{f} &= r\circ \left( g\circ f|_{A} \right)\\
                         &= \left( r\circ g\circ f|_{A} \right)|_{A}\\
                         &\simeq \left( r\circ \id_{X} \right)|_{A}\\
                         &= r|_{A}\circ \id_{X}|_{A}\\
                         &= \id_{A},
  \end{align*}
  so $A$ is contractible.
\end{solution}
