\begin{problem}[Problem 1]
  Assume that $\left( \rho,V \right)$ is a unitary representation of a group $G$. Prove that if $W\subseteq V$ is a subrepresentation, then so is $W^{\perp}$.
\end{problem}
\begin{solution}
  We must show that if $g\in G$ and $v\in W^{\perp}$, then $\rho(g)v\in W^{\perp}$. For this, observe that if $w\in W$ and $v\in W^{\perp}$, then $ \iprod{v}{w} = 0 $ by symmetry of the definitions of $W$ and $W^{\perp}$. Therefore, we have
  \begin{align*}
    \iprod{\rho(g)v}{w} &= \iprod{v}{\rho(g)^{\ast}w}\\
                        &= \iprod{v}{\rho\left( g^{-1} \right)w}\\
                        &= 0,
  \end{align*}
  where the second equality follows from the fact that $\rho(g)$ is unitary, and the last equality from the fact that $W$ is a subrepresentation of $V$.
\end{solution}
\begin{problem}[Problem 2]\hfill
  \begin{enumerate}[(a)]
    \item Prove that the following are equivalent for a given matrix $A$: 
      \begin{itemize}
        \item $A$ is unitary;
        \item the rows of $A$ form an orthonormal basis of $\C^n$ with respect to the Hermitian inner product;
        \item the columns of $A$ form an orthonormal basis of $\C^n$ with respect to the Hermitian inner product.
      \end{itemize}
    \item Assume $\left( \rho,V \right)$ is a representation of a group $G$, and let $\beta$ be an orthonormal basis for $V$. Prove that $\left( \rho,V \right)$ is unitary if and only if $\left[ \rho(g) \right]_{\beta}$ is unitary for each $g\in G$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Assume $A$ is unitary. Then, the $j$th column of $A$ is extracted via $c_j = Ae_j$. Therefore, since $A$ is unitary, we have
      \begin{align*}
        \iprod{Ae_j}{Ae_k} &= \iprod{A^{\ast}Ae_j}{e_k}\\
                           &= \iprod{e_j}{e_k}\\
                           &= \delta_{jk},
      \end{align*}
      so that the columns of $A$ form an orthonormal basis of $\C^n$. Now, if the columns of $A$ form an orthonormal basis, then we observe that
      \begin{align*}
        \left( A^{\ast}A \right)_{i,j} &= \left( \sum_{k=1}^{n}a_{ik} \overline{a_{kj}} \right)_{i,j}\\
                                       &= \left( \iprod{Ae_i}{Ae_j} \right)_{i,j}\\
                                       &= \left( \delta_{ij} \right)_{i,j},
      \end{align*}
      so that $A^{\ast}A$ is identity. Since $A$ is square, this also gives that $AA^{\ast}$ is identity.

      Now, if $A$ is unitary, then we know that both $A^{\ast}A$ is identity and $AA^{\ast}$ is identity. Expressing the latter as $\left( A^{\ast} \right)^{\ast}A^{\ast}$, we find that $A^{\ast}$ is also unitary, meaning that the columns of $A^{\ast}$ form an orthonormal basis. Yet, the columns of $A^{\ast}$ are the complex conjugates of the rows of $A$, so the rows of $A$ form an orthonormal basis.

      If the rows of $A$ form an orthonormal basis, then the complex conjugates of the columns of $A^{\ast}$ form an orthonormal basis (as these are the rows of $A$), so $A^{\ast}$ is unitary, meaning $A$ is unitary by the discussion from earlier.
    \item Let $\beta = \set{v_1,\dots,v_n}$ be this orthonormal basis. If $\left( \rho,V \right)$ is a unitary representation, then we know that for all $1\leq i,j\leq n$ and $g\in G$,
      \begin{align*}
        \delta_{ij} &= \iprod{v_i}{v_j}\\
                    &= \iprod{\rho(g)v_i}{\rho(g)v_j},
      \end{align*}
      so that $\set{\rho(g)v_1,\dots,\rho(g)v_n}$ forms an orthonormal basis for $V$. In particular, since $\rho(g)v_i$ forms column $i$ of the matrix $\left[ \rho(g) \right]_{\beta}$, it follows that the columns of $\left[ \rho(g) \right]_{\beta}$ form an orthonormal basis for $V$, so $\left[ \rho(g) \right]_{\beta}$ is a unitary matrix for each $g\in G$.

      Now, suppose $\left[ \rho(g) \right]_{\beta}$ is a unitary matrix for each $g\in G$. Then, if $\set{e_1,\dots,e_n}$ denotes the standard basis for $\C^n$ and $\varphi\colon V\rightarrow \C^n$ is an inner product-preserving isomorphism, we have
      \begin{align*}
        \delta_{ij} &= \iprod{e_i}{e_j}_{\C^n}\\
                    &= \iprod{\left[ \rho(g) \right]_{\beta}e_i}{\left[ \rho(g) \right]_{\beta}e_j}_{\C^n}\\
                    &= \iprod{\left[ \rho(g)v_i \right]_{\beta}}{\left[ \rho(g)v_j \right]_{\beta}}_{\C^n}\\
                    &= \iprod{\rho(g)v_i}{\rho(g)v_j}_{V}\\
                    &= \iprod{v_i}{v_j}_{V},
      \end{align*}
      meaning that $\rho(g)$ is a unitary map on $V$, so $\rho$ is a unitary representation on $V$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 3]
  The goal of this problem is to prove the general case of Maschke's Theorem: if $G$ is a finite group and $F$ is any field with $\operatorname{char}\left( F \right)\nmid \left\vert G \right\vert$, then any representation of $G$ over $F$ is completely reducible.

  We only need to show that every representation of $G$ over $F$ has the complement property. That is, if $\left( \rho,V \right)$ is a representation of $G$ over $F$, and $W$ is a $G$-invariant subspace, then there exists a $G$-invariant subspace $U$ such that $V = W\oplus U$.

  Consider the following maps. Choose any subspace $Z$ of $V$ such that $V = W\oplus Z$, and let $P\colon V\rightarrow W$ be the projection onto $W$ along $Z$. Define $Q\colon V\rightarrow V$ by
  \begin{align*}
    Qv &= \frac{1}{\left\vert G \right\vert} \sum_{g\in G} \left( \rho\left( g \right)^{-1}P\rho(g) \right) v.
  \end{align*}
  \begin{enumerate}[(i)]
    \item Show that $Qw = w$ for all $W\in W$ and $\img(Q) = W$. Deduce that $Q^2 = Q$.
    \item Show that $\ker\left( Q \right)$ is $G$-invariant.
  \end{enumerate}
  Deduce from (i) that $V = W\oplus \ker\left( Q \right)$, and thus $U = \ker\left( Q \right)$ is the desired complement.
\end{problem}
\begin{solution}
  We have, for any fixed $g\in G$ and any $w\in W$, we have $\rho(g)w\in W$, $P\rho(g)w = \rho(g)w$, so that $\rho\left( g \right)^{-1}\rho(g)w = w$. Averaging gives that $Qw = w$ for all $w\in W$, where the sum in the averaging formula cannot be equal to zero since $\operatorname{char}\left( F \right)\nmid \left\vert G \right\vert$.

  Now, for any $z\in Z$, we have that $\rho(g)z = w_1 + z_1$ for some $w_1\in W$ and $z_1\in Z$, so that $Qz= w_1\in W$. Since $Qw = w$ for all $w\in W$ and $Qz\in W$ for all $z\in Z$, it follows that $\img\left( Q \right) = W$. For any $v\in V$, we have some $w\in W$ such that $Qv = w$, meaning that $Q^2 v = Qw = w$, so since $v$ is arbitrary, we have $Q^2 = Q$. In particular, this means that $Q$ is a projection onto $\img\left( Q \right) = W$. Since the exact sequence
  \begin{align*}
    0 \rightarrow \ker\left( Q \right)\rightarrow V \rightarrow W \rightarrow 0
  \end{align*}
  splits, as all vector spaces are projective modules, it follows that $V = W \oplus \ker\left( Q \right)$.

  Now, suppose $v\in \ker\left( Q \right)$. Our goal is to show that for any $h\in G$, $\rho(h)v\in \ker\left( Q \right)$. Computing, we have
  \begin{align*}
    Q\rho(h)v &= \left( \frac{1}{\left\vert G \right\vert}\sum_{g\in G} \rho\left( g^{-1} \right)P\rho(g) \right)\rho(h)v\\
              &= \rho\left( h \right)\rho\left( h \right)^{-1}\left( \frac{1}{\left\vert G \right\vert}\sum_{g\in G}\rho\left( g^{-1} \right)P\rho(gh) \right)v\\
              &= \rho\left( h \right)\left( \frac{1}{\left\vert G \right\vert}\sum_{g\in G} \rho\left( \left( gh \right)^{-1} \right)P\rho\left( gh \right) \right)v\\
              &= \rho\left( h \right) \left( \frac{1}{\left\vert G \right\vert}\rho\left( g^{-1} \right)P\rho(g) \right)v\\
              &= 0.
  \end{align*}
  Thus, we find that $U = \ker\left( Q \right)$ is the desired complement to $W$.
\end{solution}
\begin{problem}[Problem 4]\hfill
  \begin{enumerate}[(a)]
    \item Let $M$ and $Q$ be modules over some ring $R$. Suppose $M$ is completely reducible and $Q$ is a quotient of $M$. Prove that $M$ has a submodule isomorphic to $Q$.
    \item Use (a) and Maschke's theorem to prove that if $G$ is a finite group and $F$ is a field with $\operatorname{char}\left( F \right)\nmid \left\vert G \right\vert$, then for any irreducible representation $V$ of $G$, there is a subrepresentation of the regular representation $F[G]$ equivalent to $V$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Let $N$ be a submodule of $M$ such that $M/N\cong Q$. Since $M$ is completely reducible, it has the complement property, so there is some module $S$ such that $M = N\oplus S$. Since $N$ is a direct summand of $M$, it follows that $Q = M/N\cong S$.
    \item Representing the action of $G$ on $F[G]$ by regular permutations gives a $F[G]$-module structure on $F[G]$, meaning it is free of rank $1$.

      Now, our goal is to show that if $V$ is an irreducible representation of $G$, then there is a surjection from $F[G]$ onto $V$, giving that $V$ is a quotient of $F[G]$ so by Maschke's theorem, the representation is completely reducible, and so by (a) there is a submodule of $F[G]$ that is isomorphic to $V$ as an $F[G]$-module, meaning that there is a subrepresentation of $G$ that is isomorphic to $V$.

      Endowing $V$ with the structure of an $F[G]$-module, we have that $V$ does not have any $F[G]$-submodules. If $v\in V$ is any nonzero vector, it then follows that $F[G]\cdot v = V$, so the homomorphism $\varphi\colon F[G]\rightarrow V$ given by $\sum_{g\in G}x_g \delta_g\mapsto \sum_{g\in G}x_g\rho(g)v$ is surjective, meaning $F[G]/\varphi\cong V$ as $F[G]$-modules. Since $V$ is a quotient of $F[G]$, we get our desired result.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 5]
  In lecture 12, we proved that if $G$ is an abelian group, then any finite-dimensional irreducible representation of $G$ over an algebraically closed field is one-dimensional.
  \begin{enumerate}[(a)]
    \item Let $n > 2$ be an integer. Construct a finite-dimensional irreducible representation of $\Z/n\Z$ over $\R$ that is not one-dimensional.
    \item Prove that any irreducible representation of $\Z/2\Z$ over any field is one-dimensional.
    \item Describe (with proof) all finite abelian groups $G$ such that any irreducible representation of $G$ over any field is one-dimensional.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Using the rotation matrices
      \begin{align*}
        a &\mapsto \begin{pmatrix} \cos\left( 2\pi/n \right) & -\sin\left( 2\pi/n \right) \\ \sin\left( 2\pi/n \right) & \cos\left( 2\pi/n \right)\end{pmatrix},
      \end{align*}
      where we have
      \begin{align*}
        \Z/n\Z &= \left\langle a | a^{n} = 1 \right\rangle
      \end{align*}
      is written multiplicatively. Note that since this is a rotation matrix, any one-dimensional subspace of $\R^2$ will be subject to a $2\pi/n$ rotation when acted upon by the generator of $\Z/n\Z$, meaning that no one-dimensional subspace is invariant, as a one-dimensional subspace is only invariant if it is subject to an order $2$ rotation, but the generator of $\Z/n\Z$ has order greater than $2$ by assumption.
    \item Assume $\rho\colon \Z/2\Z\rightarrow \GL\left( V \right)$ is an irreducible finite-dimensional representation, where we write $\Z/2\Z = \set{e,a}$. Since $V$ is irreducible as an $F[G]$-module, we have by Schur's Lemma that $\rho(a) - \lambda I$ is either zero or invertible for any $\lambda\in I$. Now, note that $A = \rho(a)$ has minimal polynomial given by $\mu_A(x) = x^2 - 1 = \left( x-1 \right)\left( x+1 \right)$. Since this minimal polynomial splits, it follows that we have some $\lambda\in F$ with $\rho(a) = \lambda I$. Yet, since this representation is irreducible, we have that $\rho(a)$ has no proper invariant subspaces if and only if $V$ is one-dimensional.
    \item We claim that all the finite abelian groups of the form
      \begin{align*}
        G &= \bigoplus_{i=1}^{n}\Z/2\Z
      \end{align*}
      exclusively admit one-dimensional irreducible representations. For this, we use a similar proof strategy as in the case of part (b). If $\rho\colon G\rightarrow \GL(V)$ is any representation, then we know that $V$ is irreducible as an $F[G]$-module, meaning that $\operatorname{end}_{F[G]}(V)$ is a division ring. Letting $g\in G$ be any element, we have $\rho(g)-\lambda I\in \operatorname{end}_{F[G]}(V)$, meaning it is either zero or invertible. Thus, it suffices to show that there is some $\lambda\in F$ $\rho(g)-\lambda I$ is not invertible. Since all elements of the form $\rho(g)$ are of order $2$, it follows that they all admit minimal polynomial $\mu(x) = x^2 - 1$, which splits over $F$. In particular, this means that $\rho(g) = \lambda I$ for some $\lambda\in F$ with $\lambda^2 = 1$, and thus we can only have this is an irreducible representation if there are no nonzero proper subspaces of $V$, meaning that $V$ is one-dimensional.
  \end{enumerate}
\end{solution}

\begin{problem}[Problem 7]
  Let $\left( \alpha,V \right)$ and $\left( \beta,W \right)$ be representations of the same group over the same field. The tensor product of these representations is the representation $\left( \rho,V\otimes W \right)$ with $\rho(g) = \alpha(g)\otimes \beta(g)$ for each $g\in G$.
  \begin{enumerate}[(a)]
    \item Prove that $\left( \rho,V\otimes W \right)$ is indeed a representation. That is, $\rho\colon G\rightarrow \GL\left( V\otimes W \right)$ is a homomorphism.
    \item Prove that if $\left( \alpha,V \right)$ is \textit{not} irreducible, and $W\neq 0$, then $\left( \rho,V\otimes W \right)$ is not irreducible either.
    \item Now prove that if $\left( \alpha,V \right)$ is irreducible and $\operatorname{dim}\left( W \right) = 1$, then $\left( \rho,V\otimes W \right)$ is irreducible.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item We will show that if $T_1\otimes S_1$ and $T_2\otimes S_2$ are elements of $\GL\left( V\otimes W \right)$, then $\left( T_1\otimes S_1 \right)\left( T_2\otimes S_2 \right) = T_1T_2\otimes S_1S_2$.

      For this, it suffices to show equality on elementary tensors, as elements of $V\otimes W$ are finite sums of elementary tensors. If $v\otimes w\in V\otimes W$, then
      \begin{align*}
        \left( \left( T_1\otimes S_1 \right)\left( T_2\otimes S_2 \right) \right)\left( v\otimes w \right) &= \left( T_1\otimes S_1 \right) \left( T_2v\otimes S_2w \right)\\
                                                                                                           &= T_1\left( T_2v \right)\otimes S_1\left( S_2 w \right)\\
                                                                                                           &= \left( T_1T_2 \right)v\otimes \left( S_1S_2 \right)w\\
                                                                                                           &= \left( T_1T_2\otimes S_1S_2 \right)\left( v\otimes w \right).
      \end{align*}
      Therefore, we have that for any $g,h\in G$, we have
      \begin{align*}
        \rho\left( gh \right) &= \alpha\left( gh \right)\otimes \beta\left( gh \right)\\
                              &= \alpha(g)\alpha(h)\otimes \beta(g)\beta(h)\\
                              &= \left( \alpha(g)\otimes \beta(g) \right)\left( \alpha(h)\otimes \beta(h) \right)\\
                              &= \rho(g)\rho(h),
      \end{align*}
      meaning $\rho$ is a group homomorphism.
    \item We will show that if $V'\leq V$ is any subspace, then $V'\otimes W\leq V\otimes W$ is a subspace. For any simple tensors $v_1\otimes w_1,v_2\otimes w_2\in V'\otimes W$, we observe that by definition, their sum is contained in $V'\otimes W$ as $V'\otimes W$ consists of sums of simple tensors $v\otimes w$ with $v\in V'$ and $w\in W$. Similarly, for any $\lambda\in F$, we have $\lambda\left( v_1\otimes w_1 \right) = v_1\otimes \lambda w_1\in V'\otimes W$. Thus, $V'\otimes W$ is a subspace of $V\otimes W$.

      In particular, if $\left( \alpha,V \right)$ not irreducible, then there is some subspace $V'\leq V$ with $V'$ invariant under $\alpha$. For any simple tensor $v\otimes w\in V'\otimes W$, we have
      \begin{align*}
        \rho\left( g \right)\left( v\otimes w \right) &= \alpha(g)v \otimes \beta(g)w\\
                                                      &\in V'\otimes W,
      \end{align*}
      meaning that $V'\otimes W$ is invariant under the action of $\rho$ on simple tensors, so it is invariant under the action of $\rho$ on all tensors, meaning $V'\otimes W\leq V\otimes W$ is a subrepresentation.
    \item We see that if $\left( \alpha,V \right)$ is an irreducible representation, then by endowing $V$ with the structure of an $F[G]$-module that acts via $\alpha$, then $V$ is simple. We will label this structure as $V_{\alpha}$, and we observe that for any nonzero $v\in V$, we have $F[G]\cdot v = V_{\alpha}$. 

      Similarly, since $W$ is a one-dimensional representation, it is necessarily irreducible, so endowing $W$ with the structure of an $F[G]$-module with the action via $\beta$, we have that $W$ is also simple. Furthermore, we view $\beta(g)w = \lambda_g w$ as multiplication by some nonzero scalar $\lambda_g$ dependent on $g\in G$.

      Now, since $W$ is one-dimensional, all elements of $W$ are of the form $\lambda w$ for some $\lambda\in F$ and $0\neq w\in W$. In particular, if $u = \sum_{i=1}^{n}v_i\otimes w_i$ is an element of $V\otimes W$, it is possible to consolidate these into one simple tensor via the process
      \begin{align*}
        \sum_{i=1}^{n}v_i\otimes w_i &= \sum_{i=1}^{n}v_i\otimes \lambda_i w\\
                                     &= \sum_{i=1}^{n}\lambda_iv_i\otimes w\\
                                     &= \left( \sum_{i=1}^{n}\lambda_iv_i \right)\otimes w\\
                                     &= v\otimes w,
      \end{align*}
      where we define $v = \sum_{i=1}^{n}\lambda_iv_i$. In particular, this means that $V\otimes W$ consists of all the simple tensors of the form $v\otimes w$ where $w$ is our distinguished generator for $W$ and $v$ is any element of $V$.

      In order to show that $V\otimes W$ is simple as an $F[G]$-module when acted upon by $\rho$, it suffices to show that for any nonzero $v\in V$, we have that $\rho\left( F[G] \right)\left( v\otimes w \right) = V\otimes W$. For an arbitrary element $\sum_{g\in G}x_g \delta_g\in F[G]$, we have
      \begin{align*}
        \rho\left( \sum_{g\in G}x_g\delta_g \right)\left( v\otimes w \right) &= \left( \alpha\left( \sum_{g\in G}x_g\delta_g \right)v \right)\otimes \left( \beta\left( \sum_{g\in G}x_g \delta_g \right)w \right)\\
                                                                             &= \left( \sum_{g\in G}x_g\alpha(g)v \right)\otimes \left( \sum_{g\in G}x_g\lambda_g \right)w\\
                                                                             &= \left( \sum_{g,h\in G}x_h\lambda_hx_g\alpha(g) \right)v\otimes w.
      \end{align*}
      Since all the $\lambda_h$ are nonzero constants, by varying the $x_g$ over $F$, we have that $\rho\left( F[G] \right)\left( v\otimes w \right) = V\otimes W$, and since $v\in V$ was an arbitrary nonzero vector, it follows that every nonzero vector in $V\otimes W$ is a generator for $V\otimes W$ as an $F[G]$-module, so $V\otimes W$ is simple as an $F[G]$-module, meaning $V\otimes W$ is irreducible as a representation of $G$.
  \end{enumerate}
\end{solution}
