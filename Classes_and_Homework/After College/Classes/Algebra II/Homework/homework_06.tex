\begin{problem}[Problem 1]
  Assume that $\left( \rho,V \right)$ is a unitary representation of a group $G$. Prove that if $W\subseteq V$ is a subrepresentation, then so is $W^{\perp}$.
\end{problem}
\begin{solution}
  We must show that if $g\in G$ and $v\in W^{\perp}$, then $\rho(g)v\in W^{\perp}$. For this, observe that if $w\in W$ and $v\in W^{\perp}$, then $ \iprod{v}{w} = 0 $ by symmetry of the definitions of $W$ and $W^{\perp}$. Therefore, we have
  \begin{align*}
    \iprod{\rho(g)v}{w} &= \iprod{v}{\rho(g)^{\ast}w}\\
                        &= \iprod{v}{\rho\left( g^{-1} \right)w}\\
                        &= 0,
  \end{align*}
  where the second equality follows from the fact that $\rho(g)$ is unitary, and the last equality from the fact that $W$ is a subrepresentation of $V$.
\end{solution}
\begin{problem}[Problem 2]\hfill
  \begin{enumerate}[(a)]
    \item Prove that the following are equivalent for a given matrix $A$: 
      \begin{itemize}
        \item $A$ is unitary;
        \item the rows of $A$ form an orthonormal basis of $\C^n$ with respect to the Hermitian inner product;
        \item the columns of $A$ form an orthonormal basis of $\C^n$ with respect to the Hermitian inner product.
      \end{itemize}
    \item Assume $\left( \rho,V \right)$ is a representation of a group $G$, and let $\beta$ be an orthonormal basis for $V$. Prove that $\left( \rho,V \right)$ is unitary if and only if $\left[ \rho(g) \right]_{\beta}$ is unitary for each $g\in G$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Assume $A$ is unitary. Then, the $j$th column of $A$ is extracted via $c_j = Ae_j$. Therefore, since $A$ is unitary, we have
      \begin{align*}
        \iprod{Ae_j}{Ae_k} &= \iprod{A^{\ast}Ae_j}{e_k}\\
                           &= \iprod{e_j}{e_k}\\
                           &= \delta_{jk},
      \end{align*}
      so that the columns of $A$ form an orthonormal basis of $\C^n$. Now, if the columns of $A$ form an orthonormal basis, then we observe that
      \begin{align*}
        \left( A^{\ast}A \right)_{i,j} &= \left( \sum_{k=1}^{n}a_{ik} \overline{a_{kj}} \right)_{i,j}\\
                                       &= \left( \iprod{Ae_i}{Ae_j} \right)_{i,j}\\
                                       &= \left( \delta_{ij} \right)_{i,j},
      \end{align*}
      so that $A^{\ast}A$ is identity. Since $A$ is square, this also gives that $AA^{\ast}$ is identity.

      Now, if $A$ is unitary, then we know that both $A^{\ast}A$ is identity and $AA^{\ast}$ is identity. Expressing the latter as $\left( A^{\ast} \right)^{\ast}A^{\ast}$, we find that $A^{\ast}$ is also unitary, meaning that the columns of $A^{\ast}$ form an orthonormal basis. Yet, the columns of $A^{\ast}$ are the complex conjugates of the rows of $A$, so the rows of $A$ form an orthonormal basis.

      If the rows of $A$ form an orthonormal basis, then the complex conjugates of the columns of $A^{\ast}$ form an orthonormal basis, so $A^{\ast}$ is unitary, meaning $A$ is unitary by the discussion from earlier.
    \item 
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 3]
  The goal of this problem is to prove the general case of Maschke's Theorem: if $G$ is a finite group and $F$ is any field with $\operatorname{char}\left( F \right)\nmid \left\vert G \right\vert$, then any representation of $G$ over $F$ is completely reducible.

  We only need to show that every representation of $G$ over $F$ has the complement property. That is, if $\left( \rho,V \right)$ is a representation of $G$ over $F$, and $W$ is a $G$-invariant subspace, then there exists a $G$-invariant subspace $U$ such that $V = W\oplus U$.

  Consider the following maps. Choose any subspace $Z$ of $V$ such that $V = W\oplus Z$, and let $P\colon V\rightarrow W$ be the projection onto $W$ along $Z$. Define $Q\colon V\rightarrow V$ by
  \begin{align*}
    Qv &= \frac{1}{\left\vert G \right\vert} \sum_{g\in G} \left( \rho\left( g \right)^{-1}P\rho(g) \right) v.
  \end{align*}
  \begin{enumerate}[(i)]
    \item Show that $Qw = w$ for all $W\in W$ and $\img(Q) = W$. Deduce that $Q^2 = Q$.
    \item Show that $\ker\left( Q \right)$ is $G$-invariant.
  \end{enumerate}
  Deduce from (i) that $V = W\oplus \ker\left( Q \right)$, and thus $U = \ker\left( Q \right)$ is the desired complement.
\end{problem}
\begin{solution}
  We have, for any fixed $g\in G$ and any $w\in W$, we have $\rho(g)w\in W$, $P\rho(g)w = \rho(g)w$, so that $\rho\left( g \right)^{-1}\rho(g)w = w$. Averaging gives that $Qw = w$ for all $w\in W$, where the sum in the averaging formula cannot be equal to zero since $\operatorname{char}\left( F \right)\nmid \left\vert G \right\vert$.

  Now, for any $z\in Z$, we have that $\rho(g)z = w_1 + z_1$ for some $w_1\in W$ and $z_1\in Z$, so that $Qz= w_1\in W$. Since $Qw = w$ for all $w\in W$ and $Qz\in W$ for all $z\in Z$, it follows that $\img\left( Q \right) = W$. For any $v\in V$, we have some $w\in W$ such that $Qv = w$, meaning that $Q^2 v = Qw = w$, so since $v$ is arbitrary, we have $Q^2 = Q$. In particular, this means that $Q$ is a projection onto $\img\left( Q \right) = W$. Since the exact sequence
  \begin{align*}
    0 \rightarrow \ker\left( Q \right)\rightarrow V \rightarrow W \rightarrow 0
  \end{align*}
  splits, as all vector spaces are projective modules, it follows that $V = W \oplus \ker\left( Q \right)$.

  Now, suppose $v\in \ker\left( Q \right)$. Our goal is to show that for any $h\in G$, $\rho(h)v\in \ker\left( Q \right)$. Computing, we have
  \begin{align*}
    Q\rho(h)v &= \left( \frac{1}{\left\vert G \right\vert}\sum_{g\in G} \rho\left( g^{-1} \right)P\rho(g) \right)\rho(h)v\\
              &= \rho\left( h \right)\rho\left( h \right)^{-1}\left( \frac{1}{\left\vert G \right\vert}\sum_{g\in G}\rho\left( g^{-1} \right)P\rho(gh) \right)v\\
              &= \rho\left( h \right)\left( \frac{1}{\left\vert G \right\vert}\sum_{g\in G} \rho\left( \left( gh \right)^{-1} \right)P\rho\left( gh \right) \right)v\\
              &= \rho\left( h \right) \left( \frac{1}{\left\vert G \right\vert}\rho\left( g^{-1} \right)P\rho(g) \right)v\\
              &= 0.
  \end{align*}
  Thus, we find that $U = \ker\left( Q \right)$ is the desired complement to $W$.
\end{solution}
\begin{problem}[Problem 4]\hfill
  \begin{enumerate}[(a)]
    \item Let $M$ and $Q$ be modules over some ring $R$. Suppose $M$ is completely reducible and $Q$ is a quotient of $M$. Prove that $M$ has a submodule isomorphic to $Q$.
    \item Use (a) and Maschke's theorem to prove that if $G$ is a finite group and $F$ is a field with $\operatorname{char}\left( F \right)\nmid \left\vert G \right\vert$, then for any irreducible representation $V$ of $G$, there is a subrepresentation of the regular representation $F[G]$ equivalent to $V$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Let $N$ be a submodule of $M$ such that $M/N\cong Q$. Since $M$ is completely reducible, it has the complement property, so there is some module $S$ such that $M = N\oplus S$. Since $N$ is a direct summand of $M$, it follows that $Q = M/N\cong S$.
    \item Representing the action of $G$ on $F[G]$ by regular permutations gives a $F[G]$-module structure on $F[G]$, meaning it is free of rank $1$.

      Now, our goal is to show that if $V$ is an irreducible representation of $G$, then there is a surjection from $F[G]$ onto $V$, giving that $V$ is a quotient of $F[G]$ so by (a) and Maschke's theorem, there is a submodule of $F[G]$ that is isomorphic to $V$ as an $F[G]$-module, meaning that there is a subrepresentation of $F[G]$ that is isomorphic to $V$.

      Endowing $V$ with the structure of an $F[G]$-module, we have that $V$ does not have any submodules. In particular, this means that $V$ is cyclic as an $F[G]$-module, so that $F[G]$ surjects onto $V$.
  \end{enumerate}
\end{solution}

\begin{problem}[Problem 6]
  Let $G = D_{2n}$ be the dihedral group of order $2n$. Recall that $G$ has presentation given by $G = \left\langle r,s | r^n=s^2=srs^{-1}r=1 \right\rangle$.
  \begin{enumerate}[(a)]
    \item Let $\omega\in \C$ is an $n$th root of unity. Prove that $G$ has a representation $\rho_{\omega}\colon G\rightarrow \GL_2\left( \C \right)$ given by
      \begin{align*}
        \rho\left( r \right) &= \begin{pmatrix}\omega & 0 \\ 0 & \omega^{-1}\end{pmatrix}\\
        \rho\left( s \right) &= \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}.
      \end{align*}
    \item Prove that the representation $\rho_{\omega}$ in part (a) is irreducible if and only if $\omega\neq \pm 1$.
    \item Let $\omega_1,\omega_2$ be $n$th roots of unity different from $\pm 1$. Prove that $\rho_{\omega_1}\cong \rho_{\omega_2}$ as representations if and only if $\omega_1 = \omega_2$ or $\omega_1^{-1} = \omega_2$.
    \item If we assume that $r$ is the counterclockwise rotation by $2\pi/n$ and $s$ is the reflection with respect to the $x$-axis, the corresponding $\rho$ is given by
      \begin{align*}
        \rho(r) &= \begin{pmatrix}\cos\left( 2\pi/n  \right) & -\sin\left( 2\pi/n \right)\\\sin\left( 2\pi/n \right) & \cos\left( 2\pi/n \right)\end{pmatrix}\\
        \rho\left( s \right) &= \begin{pmatrix}1 & 0 \\ 0 & -1\end{pmatrix}.
      \end{align*}
      Technically $\rho$ is a real representation, but since every $2\times 2$ real matrix can be thought of as a $2\times 2$ complex matrix, we can think of $\rho$ as a complex representation. Prove that this $\rho$ is equivalent to some $\rho_{\omega}$ from part (a), explicitly find such an $\omega$, and compute the matrix $T\in \GL_2\left( \C \right)$ such that $T^{-1}\rho(g)T = \rho_{\omega}(g)$ for all $g\in G$.
    \item Prove that $\left[ G,G \right] = \left\langle r^2 \right\rangle$. Deduce that $\left\vert G/\left[ G,G \right] \right\vert = 2$ if $n$ is odd and $ \left\vert G/\left[ G,G \right] \right\vert = 4 $ if $n$ is even. Then, describe all one-dimensional complex representations of $G$ explicitly.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item We want to verify that $\rho_{\omega}(s)$ and $\rho_{\omega}(r)$ satisfy the relations. First, observe that since $\omega$ is an $n$th root of unity and $\rho(r)$ is a diagonal matrix, that $\rho_{\omega}(r)^{n} = \1 = \rho_{\omega}\left( r^{n} \right)$. Similarly, $\rho_{\omega}(s)^2 = \1$ as Finally, we verify that $\rho(s)\rho(r) = \rho(r)^{-1}\rho(s)$, or
      \begin{align*}
        \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}^2 &= \begin{pmatrix}1 & 0 \\ 0 & 1\end{pmatrix}.
      \end{align*}
      Finally,
      \begin{align*}
        \begin{pmatrix}0 & 1\\ 1 & 0\end{pmatrix} \begin{pmatrix}\omega & 0 \\ 0 & \omega^{-1}\end{pmatrix} \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix} \begin{pmatrix}\omega & 0 \\ 0 & \omega^{-1}\end{pmatrix} &=  \begin{pmatrix}0 & \omega^{-1}\\\omega & 0\end{pmatrix} \begin{pmatrix}0 & \omega^{-1} \\ \omega & 0\end{pmatrix}\\
                         &= \begin{pmatrix}1 & 0 \\ 0 & 1\end{pmatrix}.
      \end{align*}
    \item We will prove that the $\rho_{\omega}$ of (a) are irreducible if and only if $\omega \neq \pm 1$. We start by finding proper $G$-invariant subspaces of $\C^2$. Let $W\leq \C^2$ be a $G$-invariant subspace. Since $V$ has dimension $2$, this means $W$ has dimension less than $2$, meaning there exists some vector
      \begin{align*}
        W &= \Span\set{ \begin{pmatrix}\alpha\\\beta\end{pmatrix} }.
      \end{align*}
      It is sufficient to check the generators $\rho_{\omega}(r)$ and $\rho_{\omega}(s)$. Starting with $\rho_{\omega}(s)$, we have
      \begin{align*}
        \rho_{\omega}(s) \begin{pmatrix}\alpha \\ \beta\end{pmatrix} &= \begin{pmatrix}\beta\\\alpha\end{pmatrix}\\
                                                                     &= \lambda \begin{pmatrix}\alpha\\\beta\end{pmatrix}.
      \end{align*}
      From this relation, we have that $\beta = \lambda \alpha$ and $\alpha = \lambda\beta$, or $\alpha = \lambda^2\alpha$, so we may write
      \begin{align*}
        W &= \Span\set{ \begin{pmatrix}\alpha\\\lambda\alpha\end{pmatrix} }
      \end{align*}
      with $\lambda = \pm 1$ as $\alpha\neq 0$. Evaluating $\rho(r)$, we have
      \begin{align*}
        \rho(r) \begin{pmatrix}\alpha\\\lambda\alpha\end{pmatrix} &= \begin{pmatrix}\omega\alpha \\ \pm\omega^{-1}\alpha\end{pmatrix}\\
                                                                  &= \begin{pmatrix}\zeta \alpha\\\pm \zeta \alpha\end{pmatrix},
      \end{align*}
      which holds if and only if $\omega = \omega^{-1}$, meaning that this holds if and only if $\omega = \pm 1$.

      If $\omega = \pm 1$, then we are guaranteed a $G$-invariant subspace generated by the vector 
      \begin{align*}
        v &= \begin{pmatrix}1\\1\end{pmatrix}.
      \end{align*}

    \item  If $\omega_1 = \omega_2$, then clearly $\rho_{\omega_1} = \rho_{\omega_2}$. Meanwhile, if $\omega_1 = \omega_2^{-1}$, then
      \begin{align*}
        T &= \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}.
      \end{align*}
      Observe that $\rho_{\omega_1}(s) = T$. This gives
      \begin{align*}
        \rho_{\omega_2}(r) &= T\rho_{\omega_1}(r)T^{-1}\\
                           &= \rho_{\omega_1}(s)\rho_{\omega_1}(r)\rho_{\omega_1}(s)^{-1}\\
                           &= \rho_{\omega_2}(r).
      \end{align*}
      Now, suppose $\rho_{\omega_1} \cong \rho_{\omega_2}$. Then, there exists some matrix $T\in \GL_2\left( \C \right)$ such that $\rho_{\omega_2}(g)T = T\rho_{\omega_1}(g)$ for all $g\in G$.

      In particular, this holds on the generators. Taking
      \begin{align*}
        T &= \begin{pmatrix}a & b \\ c & d\end{pmatrix}\\
        \rho_{\omega_2}(s) &= \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix},
      \end{align*}
      so
      \begin{align*}
        \rho_{\omega_2}(s)T &= \begin{pmatrix}0 & 1\\0 & 1\end{pmatrix} \begin{pmatrix}a & b \\ c & d\end{pmatrix}\\
                            &= \begin{pmatrix}b & a \\ d & c\end{pmatrix}\\
                            &= \begin{pmatrix}a & b \\ c & d\end{pmatrix} \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}\\
                            &= \begin{pmatrix}c & d \\ a &b\end{pmatrix},
      \end{align*}
      so $a = d$ and $b = c$, meaning
      \begin{align*}
        T &= \begin{pmatrix}a & b \\ b & a\end{pmatrix}.
      \end{align*}
      Now, we have
      \begin{align*}
        \rho_{\omega_2}(r)T &= \begin{pmatrix}a\omega_2 & b\omega_2 \\ b\omega_2^{-1} & a\omega_2^{-1}\end{pmatrix}\\
        T\rho_{\omega_1}(r) &= \begin{pmatrix}a\omega_1 & b\omega_1^{-1} \\ b\omega_{1} & a\omega_1^{-1}\end{pmatrix}.
      \end{align*}
      If $a\neq 0$, then $\omega_1 = \omega_2$, and if $a = 0$, then $b\neq 0$ and $\omega_1 = \omega_2^{-1}$.
    \item We observe that $\rho_{\omega}(s)$ has eigenvalues $\pm 1$ and eigenvectors given by
      \begin{align*}
        E_1 &= \Span\set{ \begin{pmatrix}1\\1\end{pmatrix} }\\
        E_{-1} &= \Span\set{ \begin{pmatrix}1\\-1\end{pmatrix} }.
      \end{align*}
      In particular, this means there is a unitary matrix $P\in \GL_2\left( \C \right)$ such that $P^{\ast}\rho(s)P = \rho_{\omega}(s)$ where the first column is in $E_1$ and the second column is in $E_{-1}$. Now, if we let $\theta = 2\pi/n$ and $\omega = \cos\left( \theta \right) + i\sin\left( \theta \right)$, then we observe that the matrix
      \begin{align*}
        \rho(r) &= \begin{pmatrix}\cos\left( \theta \right) & -\sin\left( \theta \right) \\ \sin\left( \theta \right) & \cos\left( \theta \right)\end{pmatrix}
      \end{align*}
      is unitarily equivalent to
      \begin{align*}
        R &= \begin{pmatrix}\omega & 0 \\ 0 & \omega^{-1}\end{pmatrix}.
      \end{align*}
      To find the desired matrix, observe that the matrix $\rho(r)$ has eigenvalues given by $\lambda_{1,2} = a \pm bi$, where $a = \cos\left( \theta \right)$ and $b = \sin\left( \theta \right)$. We then obtain eigenvectors
      \begin{align*}
        E_{\lambda_1} &= \Span\set{ \begin{pmatrix}i\\1\end{pmatrix} }\\
        E_{\lambda_2} &= \Span\set{ \begin{pmatrix}i\\-1\end{pmatrix} }.
      \end{align*}
    \item First, we show that $\left[ G,G \right] = \left\langle r^2 \right\rangle$. We will show that $r^2\in \left[ G,G \right]$. Since $\left( srs^{-1} \right)r^{-1}=r^{-2}$, we have $r^{-2}\in \left[ G,G \right]$ so $r^2\in \left[ G,G \right]$. In the reverse direction, note that $\left\langle r^2 \right\rangle = n/2$ if $n$ is even and $n$ if $n$ is odd. Since $\left\langle r^2 \right\rangle\trianglelefteq G$ is normal, as $sr^{k}r^2r^{-k}s = sr^2s = r^{-2}\in \left\langle r^2 \right\rangle$, we have that the quotient group $G/\left\langle r^2 \right\rangle$ has order either $2$ or $4$, meaning that $G/\left\langle r^2 \right\rangle$ is abelian, so $\left[ G,G \right]\subseteq \left\langle r^2 \right\rangle$ by the definition of the abelianization.

      As a result, we have already shown that $ \left\vert G/\left[ G,G \right] \right\vert = 2 $ if $n$ is odd and $ \left\vert G/\left[ G,G \right] \right\vert = 4 $ if $n$ is even.

      Describing all the $1$-dimensional complex representations of $G$, we have that if $\rho\colon G\rightarrow \C^{\times}$ is a $1$-dimensional representation, we have
      \begin{align*}
        \rho(s)\rho(r)\rho(s)^{-1} &= \rho(r)\\
                                   &= \rho\left( r^{-1} \right),
      \end{align*}
      meaning that $\rho(r)^2 = 1$, and we already have $\rho(s)^2 = 1$. These relations give representations either $\rho(r) = \pm 1$ or $\rho(s) = \pm 1$. If $n$ is even, then these are fully satisfactory of the relations in the definition of $G$. In the case when $n$ is odd, we must have $\rho(r)^{n} = 1$, and since $\left( -1 \right)^{n} \neq 1$, we have $\rho(r) = 1$.
  \end{enumerate}
\end{solution}
