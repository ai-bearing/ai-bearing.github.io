\begin{problem}[Problem 1]
  Let $p$ be a prime, $n$ a positive integer, and let $\Phi_n = x^{p^{n}}-x$ be an element of $\F_p[x]$. Prove that $\Phi_n$ is equal to the product of all the monic irreducible polynomials in $\F_p[x]$ whose degrees divide $n$, with each polynomial appearing with multiplicity one.
\end{problem}
\begin{solution}
  Fix an algebraic closure $ \overline{\F_p} $, and view any polynomial as $\F_p[x]$ as splitting into linear factors with roots in $ \overline{\F_p} $.

  The splitting field of $\Phi_n(x)$ is then given by $\F_{p^{n}}$, and we may write
  \begin{align*}
    \Phi_n(x) &= \prod_{\alpha\in \F_{p^{n}}} \left( x-\alpha \right),
  \end{align*}
  observing that each factor $\left( x-\alpha \right)$ is distinct. Any irreducible factor $f_i$ of $\Phi_n$ then defines a subfield of the splitting field of $\Phi_n(x)$, given by $\F_{p^{m}}\subseteq \F_{p^{n}}$ for some $n$. In particular, we have $m | n$, so that $\deg\left( f_i \right) | n$. Since $f_i$ divides $\Phi_n(x)$, it follows that over $ \overline{\F_p} $, we have
  \begin{align*}
    f_i(x) &= \prod_{\alpha\in \F_{p^{m}}} \left( x-\alpha \right),
  \end{align*}
  and since $\F_{p^{m}}\subseteq \F_{p^{n}}\subseteq \overline{\F_p}$ is unique, it follows that $f_i$ is unique.

  Additionally, if $f_j(x)$ is any (monic) irreducible polynomial in $\F_p[x]$ of degree $m | n$, we have that $f_j(x)$ defines a field extension $F\subseteq \F_{p^{n}}\subseteq \overline{\F_p}$ with $\left[ F:\F_p \right] = m$, whence $F = \F_{p^{m}}$. Since $\F_{p^{m}}\subseteq \F_{p^{n}}$, it follows that every root of $f_j$ in $ \overline{\F_p} $ is a root of $ \Phi_n(x) $, whence $f_j | \Phi_n(x)$, and $f_j$ is unique by the earlier discussion. Therefore, we have
  \begin{align*}
    \Phi_n(x) &= \prod_{m | n} f_m(x),
  \end{align*}
  where $\deg\left( f_m \right) | n$ and $f_m(x)$ is the unique monic irreducible factor of degree $m$ for $\Phi_n(x)$.
\end{solution}
\begin{problem}[Problem 2]
  Let $K\subseteq \C$ be the splitting field of $f(x) = x^4-2$ over $\Q$.
  \begin{enumerate}[(a)]
    \item Choose an order on the set of roots of $x^{4}-2$ and describe the associated embedding of $\gal\left( K/\Q \right)$ into $S_4$.
    \item Describe all subgroups of $\gal\left( K/\Q \right)$ and the corresponding subfields of $K$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item We observe that $K = \Q\left( \sqrt[4]{2},i \right)$ has degree $8$ over $\Q$, with basis given by
      \begin{align*}
        A &= \set{1,\sqrt[4]{2},\sqrt[4]{4},\sqrt[4]{8},i,i\sqrt[4]{2},i\sqrt[4]{4},i\sqrt[4]{8}}.
      \end{align*}
      Applying a $\Q$-automorphism to any element $k\in K$ written in its basis gives
      \begin{align*}
        \sigma\left( k \right) &= k_1 + k_2\sigma\left( \sqrt[4]{2} \right) + k_3\left( \sigma\left( \sqrt[4]{2} \right) \right)^2 + k_4\left( \sigma\left( \sqrt[4]{2} \right) \right)^{3}\\
                               &+ k_4\sigma\left( i \right) + k_5\sigma\left( i \right)\sigma\left( \sqrt[4]{2} \right) + k_6 \sigma\left( i \right)\left( \sigma\left( \sqrt[4]{2} \right) \right)^2 + k_7 \sigma\left( i \right)\left( \sigma\left( \sqrt[4]{2} \right) \right)^3.
      \end{align*}
      We have that $\gal\left( K/\Q \right)$ acts transitively on the roots of $f(x)$, which are given by $\set{\sqrt[4]{2},i\sqrt[4]{2},-\sqrt[4]{2},-i\sqrt[4]{2}}$, so we consider the automorphisms defined by $\sqrt[4]{2}\xmapsto{\sigma}i\sqrt[4]{2}$ and $i\xmapsto{\tau}-i$, where $\tau\left( \sqrt[4]{2} \right) = \sqrt[4]{2}$. Observe that under this ordering, we have representations of $\sigma = \left( 1,2,3,4 \right)$ and $\tau = \left( 2,4 \right)$ as cycle decompositions in $S_4$.
    \item The subgroups of $D_4 = \left\langle \sigma,\tau|\sigma^4,\tau^2,\tau\sigma\tau\sigma \right\rangle$, where $\sigma$ and $\tau$ are defined as above, are given by
      \begin{align*}
        H_0 &= \left\langle e \right \rangle\\
        H_1 &= \left\langle \sigma \right\rangle\\
        H_2 &= \left\langle \sigma^2 \right\rangle\\
        H_3 &= \left\langle \tau \right\rangle\\
        H_4 &= \left\langle \sigma\tau \right\rangle\\
        H_5 &= \left\langle \sigma^2\tau \right\rangle\\
        H_6 &= \left\langle \sigma^3\tau \right\rangle\\
        H_7 &= \left\langle \sigma^2,\tau \right\rangle\\
        H_8 &= \left\langle \sigma^2,\sigma\tau \right\rangle\\
        G &= \left\langle \sigma,\tau \right\rangle.
      \end{align*}
      Now, we compute fixed fields.
      \begin{itemize}
        \item First, we observe that $K^{H_0} = K$ almost by definition.
        \item To find $K^{H_1}$, we observe the action of $\sigma$ on the basis $A$ is given by
          \begin{align*}
            \sigma &= \begin{cases}
              1\mapsto 1\\
              \sqrt[4]{2} \mapsto i\sqrt[4]{2}\\
              \sqrt[4]{4} \mapsto -\sqrt[4]{4}\\
              \sqrt[4]{8} \mapsto -i\sqrt[4]{8}\\
              i \mapsto i\\
              i\sqrt[4]{2} \mapsto -\sqrt[4]{2}\\
              i\sqrt[4]{4} \mapsto -i\sqrt[4]{4}\\
              i\sqrt[4]{8} \mapsto \sqrt[4]{8}.
            \end{cases}
          \end{align*}
          Therefore, $\Q\left( i \right)\subseteq K^{H_1}$. Since $\left[ G:H_1 \right] = 2 = \left[ \Q\left( i \right):\Q \right]$, it follows that $K^{H_1} = \Q\left( i \right)$.
        \item For $K^{H_2}$, the action of $\sigma^2$ on the basis $A$ is given by
          \begin{align*}
            \sigma^2 &= \begin{cases}
              1\mapsto 1\\
              \sqrt[4]{2} \mapsto -\sqrt[4]{2}\\
              \sqrt[4]{4} \mapsto \sqrt[4]{4}\\
              \sqrt[4]{8} \mapsto -\sqrt[4]{8}\\
              i\mapsto i\\
              i\sqrt[4]{2}\mapsto -i\sqrt[4]{2}\\
              i\sqrt[4]{4} \mapsto i\sqrt[4]{4}\\
              i\sqrt[4]{8} \mapsto -i\sqrt[4]{8}.
            \end{cases}
          \end{align*}
          Therefore, we have $\Q\left( i,\sqrt{2} \right)\subseteq K^{H_2}$. Since $\left[ G:H_2 \right] = 4 = \left[ \Q\left( i,\sqrt{2} \right):\Q \right]$, equivalence follows.
        \item For $K^{H_3}$, the action of $\tau$ on the basis $A$ is given by
          \begin{align*}
            \tau &= \begin{cases}
              1\mapsto 1\\
              \sqrt[4]{2} \mapsto \sqrt[4]{2}\\
              \sqrt[4]{4} \mapsto \sqrt[4]{4}\\
              \sqrt[4]{8} \mapsto \sqrt[4]{8}\\
              i\mapsto -i\\
              i\sqrt[4]{2} \mapsto -i\sqrt[4]{2}\\
              i\sqrt[4]{4} \mapsto -i\sqrt[4]{4}\\
              i\sqrt[4]{8}\mapsto -i\sqrt[4]{8}.
            \end{cases}
          \end{align*}
          Therefore, $\Q\left( \sqrt[4]{2} \right)\subseteq K^{H_3}$, while since $\left[ G:H_3 \right]=4=\left[ \Q\left( \sqrt[4]{2} \right):\Q \right]$, we have $K^{H_3} = \Q\left( \sqrt[4]{2} \right)$.
        \item For $K^{H_4}$, the action of $\sigma\tau$ on the basis $A$ is given by
          \begin{align*}
            \sigma\tau &= \begin{cases}
              1\mapsto 1\\
              \sqrt[4]{2}\xmapsto{\tau}\sqrt[4]{2}\xmapsto{\sigma}i\sqrt[4]{2}\\
              \sqrt[4]{4}\xmapsto{\tau}\sqrt[4]{4}\xmapsto{\sigma}-\sqrt[4]{4}\\
              \sqrt[4]{8}\xmapsto{\tau}\sqrt[4]{8}\xmapsto{\sigma}-i\sqrt[4]{8}\\
              i\xmapsto{\tau}-i\xmapsto{\sigma}-i\\
              i\sqrt[4]{2}\xmapsto{\tau}-i\sqrt[4]{2}\xmapsto{\sigma}\sqrt[4]{2}\\
              i\sqrt[4]{4}\xmapsto{\tau}-i\sqrt[4]{4}\xmapsto{\sigma} i\sqrt[4]{4}\\
              i\sqrt[4]{8}\xmapsto{\tau}-i\sqrt[4]{8}\xmapsto{\sigma} -\sqrt[4]{8}.
            \end{cases}
          \end{align*}
          To find the fixed field in this scenario, we let $x = \sum_{j=1}^{8}a_je_j$ be an element of $\Q\left( \sqrt[4]{2},i \right)$, where
          \begin{align*}
            \set{e_1,\dots,e_8} = \set{1,\sqrt[4]{2},\dots,i\sqrt[4]{8}}
          \end{align*}
          is our preferred ordering of the basis over $\Q$. Then, we have
          \begin{align*}
            x &= a_1 e_1 + a_2 e_2 + a_3e_3 + a_4e_4 + a_5e_5 + a_6e_6 + a_7e_7 + a_8 e_8\\
            \sigma\tau(x) &= a_1 e_1 + a_6 e_2 - a_3e_3 -a_8e_4 - a_5e_5 + a_2 e_6 + a_7e_7 -a_4e_8
          \end{align*}
          For equality to hold, we must have $a_5 = -a_5$, $a_3 = -a_3$, $a_2 = a_6$, and $a_4 = -a_8$. Writing this out explicitly, we have
          \begin{align*}
            x &= a_1 + a_2 \left( \sqrt[4]{2} + i\sqrt[4]{2} \right) + a_7 \left( i\sqrt[4]{4} \right) + a_4 \left( \sqrt[4]{8}-i\sqrt[4]{8} \right)\\
              &= a_1 + a_2 \left( \sqrt[4]{2}\left( 1+i \right) \right) + \frac{1}{2}a_7 \left( \sqrt[4]{2}\left( 1+i \right) \right)^2 - \frac{1}{2} a_4\left( \sqrt[4]{2}\left( 1+i \right) \right)^{3}.
          \end{align*}
          In particular, this means that $\Q\left( \sqrt[4]{2}\left( 1+i \right) \right)\subseteq K^{H_4}$. Since $\left[ \Q\left( \sqrt[4]{2}\left( 1+i \right) \right):\Q \right] = 4 = \left[ G:H_4 \right]$, equality follows.
        \item Similarly for $H_5$, the action of $\sigma^2\tau$ on the basis $A$ is given by
          \begin{align*}
            \sigma^2\tau &= \begin{cases}
              1\mapsto 1\\
              \sqrt[4]{2}\xmapsto{\tau}\sqrt[4]{2}\xmapsto{\sigma^2}-\sqrt[4]{2}\\
              \sqrt[4]{4}\xmapsto{\tau}\sqrt[4]{4}\xmapsto{\sigma^2}\sqrt[4]{4}\\
              \sqrt[4]{8}\xmapsto{\tau}\sqrt[4]{8}\xmapsto{\sigma^2}-\sqrt[4]{8}\\
              i\xmapsto{\tau}-i\xmapsto{\sigma^2}-i\\
              i\sqrt[4]{2}\xmapsto{\tau}-i\sqrt[4]{2}\xmapsto{\sigma^2}i\sqrt[4]{2}\\
              i\sqrt[4]{4}\xmapsto{\tau}-i\sqrt[4]{4}\xmapsto{\sigma^2}-i\sqrt[4]{4}\\
              i\sqrt[4]{8}\xmapsto{\tau}-i\sqrt[4]{8}\xmapsto{\sigma^2}i\sqrt[4]{8}.
            \end{cases}
          \end{align*}
          In particular, we observe that the action of $\sigma^2\tau$ fixes $i\sqrt[4]{2}$, so that $\Q\left( i\sqrt[4]{2} \right)\subseteq K^{H_5}$. Since $\left[ \Q\left( i\sqrt[4]{2} \right):\Q \right] = 4 = \left[ G:H_5 \right]$, it follows that this is indeed the fixed field.
        \item Since $H_6 = \sigma H_4 \sigma^{-1}$, it follows that $K^{H_6} = \sigma\left( K^{H_4} \right)$. Since $\sigma\left( \sqrt[4]{2}\left( 1+i \right) \right) = \sqrt[4]{2}\left( i-1 \right) = -\sqrt[4]{2}\left( 1-i \right)$, it follows that $\sigma\left( \Q\left( \sqrt[4]{2}\left( 1+i \right) \right) \right) = \Q\left( \sqrt[4]{2}\left( 1-i \right) \right)$, so that $K^{H_6} = \Q\left( \sqrt[4]{2}\left( 1-i \right) \right)$.
        \item We observe that $\left[ G:H_7 \right] = 2$, meaning that $H_7$ is normal, and in particular, $K^{H_7}/\Q$ is a Galois extension. Since $\sigma^2$ and $\tau$ both fix $\sqrt[4]{4}$, it follows that $\Q\left( \sqrt{2} \right)\subseteq K^{H_7}$, but since $\left[ \Q\left( \sqrt{2} \right):\Q \right] = 2 = \left[ G:H_7 \right]$, it follows that $K^{H_7} = \Q\left( \sqrt{2} \right)$, which is the splitting field of the irreducible (by Eisenstein) polynomial $p(x) = x^2 - 2$, which verifies that this is in fact a Galois extension.
        \item Similarly, we observe that $\left[ G:H_8 \right] = 2$, and since both $\sigma^2$ and $\sigma\tau$ fix $i\sqrt[4]{4}$, it follows that $\Q\left( i\sqrt{2} \right)\subseteq K^{H_8}$. Since $\left[ \Q\left( i\sqrt{2} \right):\Q \right] = 2 = \left[ G:H_8 \right]$, this gives that $K^{H_8} = \Q\left( i\sqrt{2} \right)$, which is the splitting field for the irreducible (by Eisenstein) polynomial $p(x) = x^2 + 2$, verifying that this is indeed a Galois extension.
        \item Finally, $K^{G} = \Q$ by definition.
      \end{itemize}

\begin{problem}[Problem 3]
  Let $p$ and $q$ be distinct primes with $q > p$, and let $K/F$ be a Galois extension of degree $pq$. Prove that
  \begin{enumerate}[(a)]
    \item there exists a field $L$ with $F\subseteq L\subseteq K$ and $\left[ L:F \right] = q$;
    \item there exists a unique field $M$ with $F\subseteq M\subseteq K$ and $\left[ M:F \right] = p$.
  \end{enumerate}
\end{problem}
\begin{solution}
  From the Sylow theorems, we know that there is a $p$-Sylow subgroup $H_p$ and a $q$-Sylow subgroup $H_q$, giving existence of both $L = K^{H_p}$ and $M = K^{H_q}$ by the Galois correspondence.
  
  To show the uniqueness of $M$, it suffices to show the uniqueness of the $q$-Sylow subgroup. This follows from the fact that the number of $q$-Sylow subgroups must divide $p$ and is equivalent to $1$ mod $q$, but since $p < q$, this means the number of $q$-Sylow subgroups is equal to $1$.
\end{solution}
\begin{problem}[Problem 4]
  Let $K/F$ and $L/F$ be Galois extensions.
  \begin{enumerate}[(a)]
    \item Prove that the extension $KL/F$ is also Galois and there is a natural embedding $\iota\colon \gal\left( KL/F \right)\rightarrow \gal\left( K/F \right)\times \gal\left( L/F \right)$.
    \item Assume now that both $K/F$ and $L/F$ are finite. Prove that the map $\iota$ in (a) is an isomorphism if and only if $K\cap L = F$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item We have established already that, since $K/F$ and $L/F$ are normal, so too is $KL/F$. Therefore, it suffices to show separability. For this, observe that if $\alpha\in KL$, then we have $L(\alpha)/L/F$ is a tower of extensions with $L(\alpha)/L$ since $KL/L$ is separable and $L/F$ separable by assumption, so that $L\left( \alpha \right)/F$ is separable, meaning that $\alpha$ is separable over $F$. Since $\alpha$ is arbitrary, it follows that $KL/F$ is separable.

      Given $\sigma\in \gal\left( KL/F \right)$, we observe that since $KL/K/F$ is a tower of Galois extensions, we have that $\sigma|_{K}\in \gal\left( K/F \right)$, and similarly, since $KL/L/F$ is a tower of Galois extensions, we have $\sigma|_{L}\in \gal\left( L/F \right)$. Define a map $\iota\colon \gal\left( KL/F \right)\rightarrow \gal\left( K/F \right)\times \gal\left( L/F \right)$ given by
      \begin{align*}
        \sigma\mapsto \left( \sigma|_{K},\sigma|_{L} \right).
      \end{align*}
      For any $\sigma,\tau\in \gal\left( KL/F \right)$, we have
      \begin{align*}
        \sigma\tau &\mapsto \left( \sigma\tau|_{K},\sigma\tau|_{L} \right)\\
                   &= \left( \sigma|_{K}\tau|_{K},\sigma|_{L}\tau|{L} \right)\\
                   &= \left( \sigma|_{K},\sigma|_{L} \right) \left( \sigma|_{L},\tau|_{L} \right),
      \end{align*}
      whence this is a homomorphism.

      We observe that $\sigma\in \ker\left( \iota \right)$ if and only if $\sigma|_{K} = \id_K$ and $\sigma|_{L} = \id_L$. Viewing $\sigma\in \operatorname{Emb}_F\left( KL, \overline{F} \right)$, we have that $\sigma$ fixes both $K$ and $L$. Yet, this means $\sigma$ must fix $KL$, so that $\iota$ is an embedding.
    \item Since both $K/F$ and $L/F$ are finite extensions, so too is $KL/F$ as we have
      \begin{align*}
        \left[ KL:F \right] &= \left[ KL:K \right]\left[ L:F \right]\\
                            &= \left[ K:K\cap L \right]\left[ L:F \right]\\
                            &= \frac{\left[ K:F \right]}{\left[ K\cap L:F \right]}\left[ L:F \right].
      \end{align*}
      Since $\iota$ is injective, it follows that $\iota$ is an isomorphism if and only if $\left\vert \gal\left( KL/F \right) \right\vert = \left\vert \gal\left( K/F \right) \right\vert\left\vert \gal\left( L/F \right) \right\vert$, which holds if and only if
      \begin{align*}
        \left[ KL:F \right] &= \left[ K:F \right]\left[ L:F \right],
      \end{align*}
      which holds if and only if $\left[ K\cap L:F \right] = 1$, which holds if and only if $K\cap L = F$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 5]\hfill
  \begin{enumerate}[(a)]
    \item Let $L/F$ be a separable extension of degree $n$, and let $K$ be the Galois closure of $L$ over $F$. Prove that $K$ can be written as a compositum $L_1L_2\cdots L_n$, where $L_1,\dots,L_n$ are (not necessarily distinct) conjugates of $L$ over $F$.
    \item Let $K/F$ and $L/F$ be finite $p$-extensions. Prove that $KL/F$ is also a $p$-extension.
    \item Suppose $K/L$ and $L/F$ are $p$-extensions, and let $M$ be the Galois closure of $K$ over $F$. Prove that $M/F$ is also a $p$-extension.
    \item Now assume only that $L/F$ is a separable extension with $\left[ L:F \right]$ a power of $p$, and let $M$ be the Galois closure of $L$ over $F$. Prove that $\left[ M:F \right]$ need not be a power of $p$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item If $ \overline{F}\supseteq L$ is an algebraically closed field containing some finite separable extension $L/F$, then a subfield $L'$ is called conjugate to $L$ if $L' = \sigma\left( L \right)$ for some $F$-embedding $\sigma\colon L\rightarrow \overline{F}$.

      Let $n = \left\vert \operatorname{Emb}_F\left( L, \overline{F} \right) \right\vert$, and let $L_i = \sigma_i\left( L \right)$ where $ \operatorname{Emb}_F\left( L, \overline{F} \right) = \set{\sigma_1,\dots,\sigma_n} $. Then, if $\widetilde{L} = L_1\cdots L_n$, then we claim that $K = \widetilde{L}$.

      We start by showing that $K\subseteq \widetilde{L}$, which is equivalent to showing that $ \widetilde{L}/F $ is Galois. Let $\tau\in \operatorname{Emb}_F\left( \widetilde{L}, \overline{F} \right)$. Then, $\tau|_{L_i} = L_j$ for some $j$, meaning that $\tau\left( L_i \right)\subseteq \widetilde{L}$, so that $\tau\left(  \widetilde{L} \right)\subseteq \widetilde{L}$. Since $\tau$ is an injective map of finite-dimensional $F$-vector spaces, it is a surjection, so $\tau\left( \widetilde{L} \right) = \widetilde{L}$. 

      To show that $ \widetilde{L}\subseteq K $, it suffices to show that $L_i\subseteq K$ for each $i$. For this, observe that $\sigma_i\left( L \right) = L_i$. By the embedding lemma, there is an extension of $\sigma_i$ to $ \widetilde{\sigma}\colon K\rightarrow \overline{F} $ satisfying $ \widetilde{\sigma}|_{L} = \sigma_i $. Since $ \widetilde{\sigma}\left( K \right) = K $, it follows that $ \widetilde{\sigma}\left( L \right) = L_i\subseteq K $.
    \item A finite extension $K/F$ is called a $p$-extension if it is Galois and $\gal\left( K/F \right)$ is a $p$-group.

      If $K/F$ and $L/F$ are $p$-extensions, then we know that $KL/F$ is Galois, so we have
      \begin{align*}
        \left[ KL:F \right] &= \left[ KL:K \right] \left[ K:F \right]\\
                            &= \left[ L:K\cap L \right] \left[ K:F \right],
      \end{align*}
      but since $K\cap L$ corresponds to a subgroup of $\gal\left( L/F \right)$, it follows that $\left[ KL:F \right]$ is a $p$-extension.
    \item Since $M/F$ is Galois, it suffices to show that $M/L$ is a $p$-extension. From part (a), we know that $M = K_1K_2\cdots K_n$, where each $K_i$ is a conjugate subfield of $K$. Since $L/F$ is Galois and any embedding of $K$ into $M$ restricts to $L$, with $L$ Galois, it follows that each $K_i$ contains $L$, whence $M$ $M/L$ is a $p$-extension.
    \item Consider the extension $\Q\left( \sqrt[3]{2} \right)/\Q$, which is a extension of degree $3$ contained in the Galois extension $\Q\left( \sqrt[3]{2},\zeta_3 \right)/\Q$. Observe that the extension $\Q\left( \sqrt[3]{2} \right)/\Q$ is not Galois since the minimal polynomial of $\sqrt[3]{2}$ is given by $q(x) = x^3-2$, but $\Q\left( \sqrt[3]{2} \right)$ does not contain all the roots of $q(x)$. In particular, adjoining all the roots of $p$ gives the Galois closure is $\Q\left( \sqrt[3]{2},\zeta_3 \right)$, but this is a degree $6$ extension over $\Q$, which is not a $p$-extension where $p = 3$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 6]
  Let $p$ be a prime, $K = \F_p(t)$, the field of rational functions in one variable over $\F_p$. Let $\sigma\colon K\rightarrow K$ be the unique automorphism of $K$ such that $\sigma(t) = t + 1$ and $G = \left\langle \sigma \right\rangle$. Observe that $G$ is cyclic of order $p$. Let $F = K^{G}$. Find an explicit element $s$ such that $F = \F_p(s)$ and prove your answer.
\end{problem}
\begin{solution}
  We start by observing that $\left( \sigma(t) \right)^{p} = \left( t+1 \right)^{p} = t^{p} + 1$, so that
  \begin{align*}
    s(t) &= t^{p} - t
  \end{align*}
  is fixed by $\sigma$.

  To show that $\F_p\left( s \right)$ is in fact our desired $F$, it suffices to show that $\left[ \F_p\left( t \right):\F_p\left( s \right) \right] = p$, as we already have $\F_p\left( s \right)\subseteq K^{G}$. We observe then that
  \begin{align*}
    p&\leq \left[ \F_p\left( t \right):\F_p\left( s \right) \right]\\
     &= \left[ \F_p\left( t \right):K^{G} \right]\left[ K^{G}: \F_p\left( s \right)\right]\\
     &= p\left[ K^{G}:\F_p\left( s \right) \right],
  \end{align*}
  whence $\left[ K^{G}:\F_p\left( s \right) \right] = 1$.
\end{solution}
\begin{problem}[Problem 7]
  Let $E/F$ be a field extension, and let $K_1$ and $K_2$ be subfields of $E$ containing $F$ with $K_1/F$ and $K_2/F$ finite. Let $K_1K_2$ be the composite field of $K_1$ and $K_2$. Prove that the $F$-algebra $K_1\otimes_{F}K_2$ is a field if and only if
  \begin{align*}
    \left[ K_1K_2:F \right] &= \left[ K_1:F \right]\left[ K_2:F \right].
  \end{align*}
\end{problem}
\begin{solution}
  Letting $\iota_1$ and $\iota_2$ be the embeddings of $K_1$ and $K_2$ into $E$, we define the bilinear map $b\colon K_1\times K_2\rightarrow K_1K_2$ by $b\left( \alpha_1,\alpha_2 \right) = \iota_1\left( \alpha_1 \right)\iota_2\left( \alpha_2 \right)$. Observe that this is an $F$-bilinear map, so by the universal property, there is a linear map $ B\colon K_1\otimes_{F}K_2\rightarrow K_1K_2 $ defined by $B\left( \alpha_1\otimes \alpha_2 \right) = \iota_1\left( \alpha_1 \right)\iota_2\left( \alpha_2 \right)$. 

  Now, suppose $K_1\otimes_{F}K_2$ is a field. Then, $B$ is injective since it is a nonzero field homomorphism. Since the image of $B$ contains both $K_1$ by mapping elements of the form $\alpha_1\otimes 1$ for $\alpha_1\in K_1$ and $K_2$ by mapping elements of the form $1\otimes \alpha_2$ for $\alpha_2\in K_2$, it follows that the image of $B$ contains $K_1K_2$, hence it is equal to $K_1K_2$, meaning $B$ is an isomorphism. Since $\Dim\left( K_1\otimes_{F}K_2 \right) = \Dim\left( K_1 \right)\Dim\left( K_2 \right)$, it follows that
  \begin{align*}
    \left[ K_1K_2:F \right] &= \Dim_{F}\left( K_1K_2 \right)\\
                            &= \Dim_{F} \left( K_1\otimes_{F}K_2 \right)\\
                            &= \Dim_{F}\left( K_1 \right)\Dim_{F}\left( K_2 \right)\\
                            &= \left[ K_1:F \right]\left[ K_2:F \right].
  \end{align*}
  Now, suppose we have
  \begin{align*}
    \left[ K_1K_2:F \right] &= \left[ K_1:F \right]\left[ K_2:F \right].
  \end{align*}
  Observe by the earlier discussion that the map $B$ is surjective since the image of $B$ contains both $K_1$ and $K_2$, hence contains $K_1K_2$, meaning that in general, we have $\left[ K_1K_2:F \right]\leq \left[ K_1:F \right]\left[ K_2:F \right]$. If we have equality, then this means $B$ is a surjective linear map between vector spaces of the same dimension, meaning that it is an isomorphism of $F$-algebras. In particular, this means that given any sum of simple tensors
  \begin{align*}
    x &=\sum_{j=1}^{n}\alpha_j\otimes \beta_j
  \end{align*}
  in $K_1\otimes_{F}K_2$ with $x\neq 0$, we may find a unique inverse also in $K_1\otimes_{F}K_2$ given by
  \begin{align*}
    y &= B^{-1} \left( \left( B(x) \right)^{-1} \right),
  \end{align*}
  where $B(x)$ admits an inverse by virtue of $K_1K_2$ being a field. Since
  \begin{align*}
    xy &= x B^{-1}\left( \left( B\left( x \right) \right)^{-1} \right)\\
       &= B^{-1}\left( B(x) \right)B^{-1}\left( \left( B\left( x \right) \right)^{-1} \right)\\
       &= B^{-1} \left( B\left( x \right)\left( B\left( x \right) \right)^{-1} \right)\\
       &= B^{-1}\left( 1 \right)\\
       &= 1\otimes 1,
  \end{align*}
  it follows that this is indeed an inverse.
\end{solution}
\begin{problem}[Problem 8]
  Let $K/F$ be a finite Galois extension of degree $n$. Prove that
  \begin{align*}
    K\otimes_{F}K &\cong \underbrace{K\times\cdots\times K}_{\text{$n$ times}}
  \end{align*}
  as $K$-algebras.
\end{problem}
\begin{solution}
  Since $K/F$ is a finite Galois extension, it follows that there is some $\alpha\in K$ such that $K = F\left( \alpha \right)$, where $\alpha$ has degree $n$. Furthermore, $\mu_{\alpha,F}(x)$ splits into distinct linear factors in $K$. Using the Chinese remainder theorem and the extension of scalars properties, we get
  \begin{align*}
    K\otimes_{F}K &= K\otimes_{F} F\left( \alpha \right)\\
                  &\cong K\otimes_{F} \frac{F\left[ x \right]}{\left( \mu_{\alpha,F}(x) \right)}\\
                  &\cong \frac{K\left[ x \right]}{\left( \mu_{\alpha,F}(x) \right)}\\
                  &= \bigoplus_{i=1}^{n} \frac{K\left[ x \right]}{\left( x-\alpha_i \right)}\\
                  &\cong K^{n}.
  \end{align*}
\end{solution}
