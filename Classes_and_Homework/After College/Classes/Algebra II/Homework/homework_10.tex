\begin{problem}[Problem 1]
  Prove the interesting part of Corollary 18.7 from the Spring 2010 notes: if $K/L/F$ is a tower of algebraic extensions, and $K/L$ and $L/F$ are separable, then $K/F$ is separable.
\end{problem}
\begin{solution}
  Suppose $K/L$ and $L/F$ are separable, with $K/F$ a finite extension. Then, since $\left[ K:F \right] = \left[ K:L \right]\left[ L:F \right]$, we must have that both $K/L$ and $L/F$ are finite extensions. It follows that there are towers of separable simple extensions $L_0 =L \subseteq L_1\subseteq\cdots\subseteq L_m = K$, and $F_0 = F \subseteq F_1\subseteq\cdots\subseteq F_n = L$. Combined, this yields a tower of separable simple extensions 
  \begin{align*}
    F_0 = F \subseteq F_1\subseteq \cdots \subseteq F_n\subseteq L_1\subseteq\cdots\subseteq L_m = K,
  \end{align*}
  so that $K/F$ is separable.

  If $K/F$ is not a finite extension, we let $\alpha\in K$ be any element, and choose some polynomial $p\in K[x]$ given by
  \begin{align*}
    p(x) &= a_0 + a_1 x + \cdots + a_{m-1}x^{m-1} + x^{m}
  \end{align*}
  that has $p(\alpha) = 0$. We then have that $M = F\left( a_0,\dots,a_{m-1} \right)/F$ is a finite extension. Since we have either $M\subseteq L$ or $L\subseteq M$, we have either $L/M/F$ is a tower of separable extensions or $K/M/L$, following from the fact that $L/F$ and $K/L$ respectively are separable. Let $M' = M\left( \alpha \right)$, so that $M'/F$ is a finite extension. If $M'\subseteq L$, then any element of $M'$ is an element of $L$, so the minimal polynomial of any element of $M'$ is separable over $F$, meaning $M'/F$ is separable. If $L\subseteq M'$, then the tower $M'/L/F$ has $\left[ M':F \right]$ finite, so that $\left[ M':L \right]$ and $\left[ L:F \right]$ are finite separable extensions, reducing to the previous case of $K/F$ as finite.

  Therefore, any element $\alpha\in K$ is contained in a separable extension over $F$, so $K$ is separable over $F$.
\end{solution}

\begin{problem}[Problem 4]
  Let $f(x)\in \Q[x]$ be an irreducible polynomial of degree $n$, and let $K\subseteq \C$ be the splitting field for $f(x)$ over $\Q$. Label the roots of $f(x)$ by $1,\dots,n$ in some order, and let $\iota\colon \operatorname{Gal}\left( K/\Q \right)\rightarrow S_n$ be the associated embedding.
  \begin{enumerate}[(a)]
    \item Prove that $\left\vert \operatorname{Gal}\left( K/\Q \right) \right\vert$ is a multiple of $n$ and divides $n!$.
    \item Assume $f(x)$ has at least one non-real root. Prove that complex conjugation is an element of $ \operatorname{Gal}\left( K/\Q \right) $ of order $2$.
    \item Assume that $f(x)$ has precisely two non-real roots. Prove that the image of the complex conjugation under the embedding $\iota$ is a transposition.
    \item Suppose that $n = \operatorname{deg}\left( f \right)$ is prime, and assume that $f(x)$ has exactly two non-real roots. Prove that $\operatorname{Gal}\left( K/\Q \right)$ is isomorphic to $S_n$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item We have that $\iota\left( \operatorname{Gal}\left( K/\Q \right) \right)\subseteq S_n$ is a subgroup, so Lagrange's theorem immediately gives that $\left\vert \operatorname{Gal}\left( K/\Q \right) \right\vert$ divides $n!$.

      Label the roots of $f$ by $\set{\alpha_1,\dots,\alpha_n}$. Note that the Galois group acts transitively on this set, so by orbit--stabilizer, we have
      \begin{align*}
        n &= \left\vert O_{\alpha_1} \right\vert\\
          &= \left\vert G \right\vert/\left\vert G_{\alpha_1} \right\vert,
      \end{align*}
      where we have $G = \operatorname{Gal}\left( K/\Q \right)$, so that $n$ divides $\left\vert \operatorname{Gal}\left( K/\Q \right) \right\vert$.
    \item Let $\alpha$ be one of these non-real roots. Since $K/\Q$ is a normal extension, it follows that any irreducible polynomial over $\Q$ that has $\alpha$ as a root must split fully over $K$. It suffices to show then that the minimal polynomial of $\alpha$ over $\Q$ also must have $ \overline{\alpha} $ as a root. Then, the map of $K$ taking every element to its conjugate would be an automorphism of $K$ that fixes $\Q$, meaning that it would be an element of $\operatorname{Gal}\left( K/\Q \right)$, and it would have degree $2$ as we are assuming that $K$ has an element with nontrivial imaginary part.

      For this, we have
      \begin{align*}
        \mu_{\alpha,F} \left( \overline{\alpha} \right) &= \overline{\alpha}^{n} + a_{n-1} \overline{\alpha}^{n-1} + \cdots + a_1\overline{\alpha} + a_0\\
                                                        &= \overline{\alpha^{n} + a_{n-1}\alpha^{n-1} + \cdots + a_1 \alpha + a_0}\\
                                                        &= 0,
      \end{align*}
      so we have our desired result.
    \item The image of complex conjugation under $\iota$ is an element of order $2$, meaning that it has a cycle type given by a sum
      \begin{align*}
        n &= 2 + 2 + \cdots + 2 + 1 + \cdots + 1.
      \end{align*}
      Next, we observe that the action of conjugation $c\in \operatorname{Gal}\left( K/\Q \right)$ does not affect any of the real roots, meaning that for every real root $\alpha_i$, $c\cdot \alpha_i = \alpha_i$. This means there are $\left( n-2 \right)$ fixed points of this action, so that the cycle type consists of exactly one instance of $2$ in the partition of $n$, or that the image of conjugation under $\iota$ is a transposition.
    \item We start by observing that $ W = \iota\left( \operatorname{Gal}\left( K/\Q \right) \right)\subseteq S_n $ is a group with order $kp$ that divides $p!$, hence $1\leq k \leq \left( p-1 \right)!$. Since $p$ does not divide each of $1,\dots,p-1$, it follows that $W$ admits a $p$-Sylow subgroup since $p\nmid k$, meaning there is a $p$-cycle in $W$.

      By rearranging the roots if necessary, we may assume that the automorphism corresponding to the $p$ cycle maps to $\gamma_p = \left( 1,\dots,p \right)$, and we have that complex conjugation in $K$ corresponds to some transposition given by $\tau = \left( a,b \right)$. We will show now that the group $\left\langle \gamma_p,\tau \right\rangle$ is equal to $S_p$.

      For this, observe that $S_p$ is generated by $\tau_1 = \left( 1,2 \right)$ and $\gamma_p$, since $\gamma_p^{k}\tau_1\gamma_p^{-k} = \left( 1+k,2+k \right) \eqcolon \tau_k$, from which we may construct $\left( 1,s \right)$ by conjugating by $\tau_2,\dots,\tau_{s-1}$ in order. Eventually, this will yield all transpositions of the form $\left( r,s \right)$ with $r < s$ by some mixture of cycling through conjugations with $\gamma_p$ or $\tau_k$. Since the symmetric group is generated by transpositions, it follows that $S_p$ is generated by $\tau_1$ and $\gamma_p$.

      Without loss of generality, we may assume that $\tau = \left( 1,c \right)$ by conjugating with sufficiently many copies of $\gamma_p$. It follows that if we are able to show that $\left( 1,2 \right)\in \left\langle \gamma_p,\tau \right\rangle$, then we are done by the previous paragraph. For this, observe that by conjugating $\tau$ with powers of $\gamma_p$, we may find all transpositions of the form $\left( k,c + k-1 \right)$ for any $1\leq k \leq p$. Setting $k = c$, we have the transposition $\zeta = \left( c,2c-1 \right)$, so that $\zeta\tau\zeta^{-1} = \left( 1,2c-1 \right)$. Performing this procedure repeatedly gives that we may find all transpositions of the form $\left( 1,\ell c - \ell + 1 \right)$. Modulo $p$, this means we want to find $\ell$ such that
      \begin{align*}
        \ell c - \ell + 1 &= 2,
      \end{align*}
      and since $p$ is prime, and $c\neq 1$, we have $\left( c-1 \right)$ is invertible mod $p$, we have $\ell = \left( c-1 \right)^{-1}$, which gives our desired transposition $\left( 1,2 \right)\in \left\langle \gamma_p,\tau \right\rangle$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 5]
  Let $p$ be a prime, $n\geq 2$ an integer, and $f(x) = x^{n} - p$. Let $K\subseteq \C$ be the splitting field for $f(x)$ over $\Q$.
  \begin{enumerate}[(a)]
    \item Prove that $K = \Q\left( \sqrt[n]{p},\zeta_n \right)$ where $\zeta_n = e^{2\pi i /n}$.
    \item Prove that $\left[ K:\Q \right]\leq n\varphi(n)$, where $\varphi$ is the Euler totient function. Then, describe the elements of $\operatorname{Gal}\left( K/\Q \right)$ by their actions on $\sqrt[n]{p}$ and $\zeta_n$.
    \item Let $G = \Z/n\Z\rtimes \left( \Z/n\Z \right)^{\times}$, where $\left( \Z/n\Z \right)^{\times}$ acts on $\Z/n\Z$ by multiplication. Prove that $\operatorname{Gal}\left( K/\Q \right)$ is isomorphic to a subgroup of $G$ and describe an explicit embedding.
    \item Let $p = 3$ and $n = 12$. Prove that the inequality in (b) is strict, and find $\left[ K:\Q \right]$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item Let
      \begin{align*}
        A &= \set{\sqrt[n]{p},\zeta_n\sqrt[n]{p},\dots,\zeta_n^{n-1}\sqrt[n]{p}}.
      \end{align*}
      We then have that $K = \Q\left( A \right)\subseteq \Q\left( \sqrt[n]{p},\zeta_n \right)$, and since
      \begin{align*}
        \zeta_n &= \frac{\zeta_n\sqrt[n]{p}}{\sqrt[n]{p}},
      \end{align*}
      it follows that $\Q\left( \sqrt[n]{p},\zeta_n \right)\subseteq \Q\left( A \right) = K$, whence $K = \Q\left( \sqrt[n]{p},\zeta_n \right)$.
    \item We have
      \begin{align*}
        \left[ K:\Q \right] &= \left[ \Q\left( \sqrt[n]{p},\zeta_n \right):\Q \right]\\
                            &= \left[ \Q\left( \sqrt[n]{p},\zeta_n \right):\Q\left( \zeta_n \right) \right]\left[ \Q\left( \zeta_n \right):\Q \right]\\
                            &\leq n\varphi(n),
      \end{align*}
      where the second inequality follows from the fact that $\mu_{\sqrt[n]{p},\Q} = x^{n} - p$ by the Eisenstein criterion, so $\mu_{\sqrt[n]{p},\Q\left( \zeta_n \right)}(x) | x^{n} - p$.

      We have that the Galois group $\gal\left( K/\Q \right)$ may act on the primitive $n$-th roots of unity $\zeta_n$ by permutation, meaning that it may map $\zeta_n\mapsto \zeta_n^{d}$ for any $d\in \left( \Z/n\Z \right)^{\times}$. Similarly, the Galois group may act on the roots of $f(x)$ by cyclic permutation, taking $\sqrt[n]{p}$ to $\zeta^{a}\sqrt[n]{p}$, for any $a\in \Z/n\Z$.
    \item Consider an element $\varphi\in \gal\left( K/\Q \right)$. Then, $\varphi$ is defined by two maps; one taking $\zeta_n\mapsto \zeta_n^{d}$ for some $d\in \left( \Z/n\Z \right)^{\times}$ and one taking $\sqrt[n]{p}\mapsto \zeta_n^{dk}\sqrt[n]{p}$, for some $k\in \Z/n\Z$. This allows us to define a map
      \begin{align*}
        \theta\colon \gal\left( K/\Q \right)&\rightarrow \Z/n\Z\rtimes \left( \Z/n\Z \right)^{\times}\\
        \varphi &\mapsto \left( dk,d \right).
      \end{align*}
      We start by showing that this is a homomorphism. First, let
      \begin{align*}
        \varphi_1 &\coloneq \begin{cases}
          \sqrt[n]{p} \mapsto \zeta_n^{d_1k_1}\sqrt[n]{p}\\
          \zeta_n \mapsto \zeta_n^{d_1}
        \end{cases}\\
          \varphi_2 &\mapsto \begin{cases}
            \sqrt[n]{p}\mapsto \zeta_n^{d_2k_2}\sqrt[n]{p}\\
            \zeta_n \mapsto \zeta_n^{d_2}
          \end{cases}.
      \end{align*}
      We have
      \begin{align*}
        \theta\left( \varphi_1 \right)\theta\left( \varphi_2 \right) &= \left( d_1k_1,d_1 \right)\left( d_2k_2,d_2 \right)\\
                                                                     &= \left( d_1k_1 + d_1d_2k_2,d_1d_2 \right)
                                                                       \intertext{and}
        \varphi_1\circ\varphi_2 &= \begin{cases}
          \sqrt[n]{p}\mapsto \zeta_{n}^{d_2k_2}\sqrt[n]{p}\mapsto \left( \zeta_n^{d_2k_2} \right)^{d_1}\zeta_n^{d_1k_1}\sqrt[n]{p}=\zeta_n^{d_1k_1 + d_1d_2k_2}\sqrt[n]{p}\\
          \zeta_n\mapsto \zeta_n^{d_2}\mapsto \zeta_n^{d_1d_2},
        \end{cases}
          \intertext{whence}
          \theta\left( \varphi_1\circ\varphi_2 \right) &= \left( d_1k_1 + d_1d_2k_2,d_1d_2 \right).
      \end{align*}
      This is injective since we can only have $\left( dk,d \right)\mapsto \left( 0,1 \right)$ when $d = 1$ and $k = 0$, so that $\ker\left( \theta \right) = \id$.
    \item We will compute $\zeta_{12}$ explicitly, finding that it is equal to
      \begin{align*}
        \zeta_{12} &= \frac{1}{2} \left( \sqrt{3} + i \right),
      \end{align*}
      so that
      \begin{align*}
        \Q\left( \sqrt[12]{3},\zeta_{12} \right) &= \Q\left( \sqrt[12]{3},i \right),
      \end{align*}
      where we observe that $\sqrt{3}\in \Q\left( \sqrt[12]{3} \right)$. We may then compute
      \begin{align*}
        \left[ K:\Q \right] &= \left[ \Q\left( \sqrt[12]{3},i \right):\Q \right]\\
                            &= \left[ \Q\left( \sqrt[12]{3},i \right):\Q\left( \sqrt[12]{3} \right) \right]\left[ \Q\left( \sqrt[12]{3} \right):\Q \right]\\
                            &= \left( 2 \right)\left( 12 \right)\\
                            &= 24,
      \end{align*}
      while since $\varphi\left( 12 \right) = 4$, as $1,5,7,11$ are coprime to $12$, it follows that $n\varphi(n) = 48$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 6]
  Prove that if $n = p^{k}m$ where $p$ is a prime and $m$ is relatively prime to $p$, then there are precisely $m$ distinct $n$ th roots of unity over a field of characteristic $p$.
\end{problem}
\begin{solution}
  Let
  \begin{align*}
    q(x) &= x^{n}-1,
  \end{align*}
  whose roots are the $n$th roots of unity. Since we are working over characteristic $p$, we may write
  \begin{align*}
    q(x) &= \left( x^{m} \right)^{p^k}-1\\
         &= \left( x^{m}-1 \right)^{p^{k}}\\
         &\eqqcolon r(x)^{p_k},
  \end{align*}
  and in particular, the $n$th roots of unity are the roots of $r(x)$. Observe that $r'(x) = mx^{m-1}\neq 0$ as $p\nmid m$, so that $\gcd\left( r,r' \right) = 1$, meaning that $r(x) = x^{m}-1$ is separable in the algebraic closure $ \overline{F} $ of $F$.
\end{solution}
