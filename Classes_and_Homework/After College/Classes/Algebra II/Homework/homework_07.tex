\begin{problem}[Problem 1]
  Let $G$ be a group, $N$ a normal subgroup of $G$, and let $\pi\colon G\rightarrow G/N$ the natural projection.

  Given a representation $\rho\colon G/N\rightarrow \GL(V)$ of $G/N$, define the representation $ \widetilde{\rho}\colon G\rightarrow \GL(V) $ of $G$ by
  \begin{align*}
    \widetilde{\rho}(g) &= \rho\circ \pi\left( g \right)\\
                        &= \rho\left( gN \right).
  \end{align*}
  Prove that $\widetilde{\rho}$ is irreducible if and only if $\rho$ is irreducible. Also, prove that two representations $\rho_1$ and $\rho_2$ of $G/N$ are equivalent if and only if the corresponding representations $\widetilde{\rho}_1$ and $\widetilde{\rho}_2$ are equivalent.
\end{problem}
\begin{problem}[Problem 2]
  Let $\left( \rho,V \right)$ be a representation of a group $G$. Recall that the dual representation $\left( \rho^{\ast},V^{\ast} \right)$ is defined by $\rho^{\ast}\left( g \right)f = f\circ \rho\left( g^{-1} \right)$ for all $f\in V^{\ast}$. Prove that
  \begin{enumerate}[(i)]
    \item $\left( \rho^{\ast},V^{\ast} \right)$ is a representation;
    \item if $\beta$ is a basis for $V$, and $\beta^{\ast}$ is the dual basis for $V^{\ast}$, then $\left[ \rho^{\ast}\left( g \right) \right]_{\beta^{\ast}} = \left[ \rho\left( g^{-1} \right) \right]_{\beta}^{T}$. 
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(i)]
    \item If $g,h\in G$, and $f\in V^{\ast}$ is arbitrary, then
      \begin{align*}
        \rho^{\ast}\left( gh \right)f &= f\circ \rho\left( \left( gh \right)^{-1} \right)\\
                                      &= f\circ \rho\left( h^{-1}g^{-1} \right)\\
                                      &= \left( f\circ \rho\left( h^{-1} \right) \right)\circ \rho\left( g^{-1} \right)\\
                                      &= \rho^{\ast}\left( g \right)\left( f\circ \rho\left( h^{-1} \right) \right)\\
                                      &= \rho^{\ast}\left( g \right)\rho^{\ast}\left( h \right)f.
      \end{align*}
      Therefore, we have $\rho^{\ast}\left( gh \right) = \rho^{\ast}\left( g \right)\rho^{\ast}\left( h \right)$ for all $g,h\in G$.
    \item Our goal is to show that $\left[ A^{\ast} \right]_{\beta^{\ast}} = \left[ A \right]_{\beta}^{T}$. We note that if $\left[ A \right]_{\beta}$ is a matrix on some basis $\beta$, then the matrix coefficient $a_{ij}$ is extracted by evaluating $v_i^{\ast}\left( Av_j \right)$, where $v_i^{\ast}$ is the dual basis element for $v_i$. In particular, this means that, if $b_{ij}$ denotes the $\left( i,j \right)$ entry of $\left[ A^{\ast} \right]_{\beta^{\ast}}$, we have $b_{ij} = \hat{v}_i\left( A^{\ast}v_j^{\ast} \right)$, where $\hat{v}_i$ denotes the double dual element for $v_i$. Yet, this gives
      \begin{align*}
        b_{ij} &= \hat{v}_i\left( A^{\ast}v_j^{\ast} \right)\\
               &= \left( A^{\ast}v_j^{\ast} \right)\left( v_i \right)\\
               &= v_j^{\ast}\left( Av_i \right)\\
               &= a_{ji}.
      \end{align*}
      Thus, $\left[ A^{\ast} \right]_{\beta^{\ast}} = \left[ A \right]_{\beta}^{T}$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 3]
  Describe all irreducible complex representations of $S_4$ and compute its character table.
\end{problem}
\begin{solution}
  The following partitions of $4$ give that there are $5$ conjugacy classes:
  \begin{enumerate}[(i)]
    \item $4 = 1 + 1 + 1 + 1$ (the trivial cycle type);
    \item $4 = 2 + 1 + 1$ (transpositions);
    \item $4 = 2 + 2$ (the Klein $4$-group, $V_4$);
    \item $4 = 3 + 1$ (the $3$-cycles);
    \item $4 = 4$ (the $4$-cycles).
  \end{enumerate}
  We will label these classes by $K_1,\dots,K_5$. We already know of three representations for $S_4$:
  \begin{enumerate}[(i)]
    \item $\rho_{t}$, which is the trivial representation (which has dimension 1);
    \item $\rho_{\sgn}$, the sign representation (which has dimension 1);
    \item $\rho_{\operatorname{std}}$, the standard representation (which has dimension 3).
  \end{enumerate}
  Since $S_4/V_4\cong S_3$, we know that these three representations also define the three irreducible representations for $S_3$; using Problem 1, we may then find a representation which we will call $\rho_{q}$ (quotient) for $S_4$ by taking $\rho(g) = \rho_{\operatorname{std}}\left(gV_4\right)$. This representation has dimension 2.

  Finally, we know that the tensor product of an irreducible representation by a one-dimensional representation is also irreducible, so taking the tensor product of $\rho_{\operatorname{std}}$ with $\rho_{\operatorname{sgn}}$ gives the representation $\rho_{\operatorname{twist}} = \rho_{\operatorname{std}}\otimes \rho_{\operatorname{sgn}}$. We thus get the following character table.
  \begin{center}
    \begin{tabular}{c|c|c|c|c|c}
      & $K_1$ & $K_2$ & $K_3$ & $K_4$ & $K_5$\\
      \hline
      $\chi_{t}$ & $ 1 $ & $1$ & $1$ & $1$ & $1$ \\
      $\chi_{\operatorname{sgn}}$ & $1$ & $-1$ & $1$ & $1$ & $-1$ \\
      $\chi_{q}$ & $2$ & $0$ & $2$ & $-1$ & $0$\\
      $\chi_{\operatorname{std}}$ & $3$ & $1$ & $-1$ & $ 0 $ & $-1$\\
      $\chi_{\operatorname{twist}}$ & $3$ & $-1$ & $-1$ & $0$ & $1$ \\
      \hline
    \end{tabular}
  \end{center}
\end{solution}
\begin{problem}[Problem 4]
  Now, describe all irreducible complex representations of the alternating group $A_4$, and compute its character table. First, prove that $\left[ A_4,A_4 \right] = V_4$, where
  \begin{align*}
    V_4 &= \set{e,\left( 1,2 \right)\left( 3,4 \right),\left( 1,3 \right)\left( 2,4 \right),\left( 1,4 \right)\left( 2,3 \right)}
  \end{align*}
  is the Klein $4$-group.
\end{problem}
\begin{solution}
  There are three cycle types contained in $A_4$ --- namely, the identity, cycles of type $ 2 + 2 $, and the $3$-cycles. Observe that, by the pigeonhole principle, there are no odd permutations $Z_{S_4}\left( \left( 1,2,3 \right) \right)$, meaning there are a total of four conjugacy classes in $A_4$.

  To find the commutator subgroup for $A_4$, it suffices to show that the commutator of any permutations of these cycle types is given by some cycle of type $2 + 2$. Computing, we have
  \begin{align*}
    \left( a,b,c \right)\left( a,b \right)\left( c,d \right)\left( a,c,b \right)\left( a,b \right)\left( c,d \right) &= \left( a,b,c \right)\left( b,d,a \right)\\
                                                                                                                                &= \left( a,c \right)\left( b,d \right)\\
    \left( a,b,c \right)\left( b,c,d \right)\left( a,c,b \right)\left( b,d,c \right) &= \left( a,b,c \right)\left( a,d,c \right)\\
                                                                                     &= \left( a,d \right)\left( b,c \right).
  \end{align*}
  The fact that the commutator of two cycles of type $2 + 2$ is a cycle of type $2 + 2$ is clear from the fact that the cycles of type $2 + 2$ form a subgroup of $A_4$. Therefore, since every commutator is a cycle of type $ 2 + 2 $, we have by varying all of $a,b,c,d$ among $1,2,3,4$ that $\left[ A_4,A_4 \right] = V_4$.

  In particular, this gives three one-dimensional representations: one is the trivial representation, one is the representation that takes the cycle $\left( 1,2,3 \right)$ to a cube root of unity denoted $\omega$, and one is the representation that takes the cycle $\left( 1,2,3 \right)$ to $\omega^2$.

  Finally, there is a $3$-dimensional representation that emerges from taking the standard representation of these permutations viewing them in $S_4$. Note that the character of the standard representation is given by subtracting $1$ from the character of the defining representation viewing them as permutation matrices over $\C$. Thus, we get the following character table.
  \begin{center}
    \begin{tabular}{c|c|c|c|c}
      & $\left( 1,2 \right)\left( 3,4 \right)$ & $\left( 1,2,3 \right)$ & $\left( 1,3,2 \right)$\\
      \hline
      $\chi_{t}$ & $1$ & $1$ & $1$ & $1$\\
      $\chi_{\omega}$ & $1$ & $1$ & $\omega$ & $\omega^2$\\
      $\chi_{\omega^2}$ & $1$ & $1$ & $\omega^2$ & $\omega$\\
      $\chi_{\operatorname{std}}$ & $3$ & $-1$ & $0$ & $0$\\
      \hline
    \end{tabular}
  \end{center}
\end{solution}
\begin{problem}[Problem 5]
  A representation $\left( \rho,V \right)$ of a group $G$ over a field $F$ is called \textit{cyclic} if $V$ is cyclic as an $F[G]$-module.
  \begin{enumerate}[(a)]
    \item Prove that any irreducible representation is cyclic.
    \item If $\left( \rho,V \right)$ is a cyclic representation of a finite group $G$, prove that $\Dim\left( V \right)\leq \left\vert G \right\vert$.
    \item Now, let $\left( \rho,V \right)$ be an irreducible representation of a finite group $G$. Prove that $\Dim\left( V \right)\leq \left\vert G \right\vert-1$.
  \end{enumerate}
\end{problem}
\begin{solution}\hfill
  \begin{enumerate}[(a)]
    \item If $v_0\in V$ is any element, then $F[G]\cdot v_0\leq V$ is a submodule, but since $V$ is irreducible as an $F[G]$-module, it follows that $F[G]\cdot v_0 = V$, so $\rho$ is cyclic.
    \item If $v\in V$ is a cyclic vector for $\rho$, then we observe that the set
      \begin{align*}
        S &= \set{\rho(g)v}_{g\in G}
      \end{align*}
      is a spanning set for $V$, meaning that $\Dim\left( V \right)\leq \left\vert S \right\vert\leq \left\vert G \right\vert$.
    \item Since irreducible representations are cyclic, we know that the set $S$ determined as in part (b) is a spanning set for $V$, where $v$ is our cyclic vector. If it were the case that $S$ were linearly independent, meaning that $\Dim\left( V \right) = \left\vert G \right\vert$, then we observe that the one-dimensional subspace
      \begin{align*}
        L &= \set{\lambda \sum_{g\in G} \rho(g)v | \lambda\in F}
      \end{align*}
      is then $G$-invariant, which contradicts the assumption that $\left( \rho,V \right)$ is irreducible. Therefore, it must be the case that $\Dim\left( V \right)\leq \left\vert S \right\vert\leq \left\vert G \right\vert-1$.
  \end{enumerate}
\end{solution}
\begin{problem}[Problem 6]
  Let $G$ be a group, $N$ a normal subgroup of $G$, and $\left( \rho,V \right)$ a representation of $G$ over a field $F$.
  \begin{enumerate}[(a)]
    \item Prove that $V^{N} = \set{v\in V | \rho(n)v = v\text{ for all }n\in N}$ is a subrepresentation of $\left( \rho,V \right)$.
    \item Given a homomorphism $\lambda\colon N\rightarrow F^{\times}$, define
      \begin{align*}
        V_{\lambda}\left( N \right) &= \set{v\in V | \rho(n)v = \lambda(n)v\text{ for all }n\in N}.
      \end{align*}
      Prove that for every $g\in G$ and every $\lambda\colon N\rightarrow F^{\times}$, we have $\rho(g)\left( V_{\lambda} \right) = V_{\mu}$ for some $\mu$ depending on $\lambda$ and $g$.
  \end{enumerate}
\end{problem}
