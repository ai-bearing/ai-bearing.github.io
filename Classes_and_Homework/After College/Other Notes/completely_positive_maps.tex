\section{Positive Maps}%
We will start by focusing our discussion of positive maps on a subclass of linear subspaces of $C^{\ast}$-algebras.
\begin{definition}
  Let $\mathcal{A}$ be a $C^{\ast}$-algebra, and let $\mathcal{S}\subseteq \mathcal{A}$ be a self-adjoint linear subspace that contains $1$. We call such an $\mathcal{S}$ an \textit{operator system}.
\end{definition}
Note that if $h$ is a self-adjoint element of $\mathcal{S}$, then it is possible to write $h$ as the difference of two positive elements in $\mathcal{S}$,
\begin{align*}
  h &= \frac{1}{2}\left( \norm{h}1 + h \right) - \frac{1}{2}\left( \norm{h}1-h \right).
\end{align*}
\begin{definition}
  If $\mathcal{S}\subseteq \mathcal{A}$ is an operator system, $\mathcal{B}$ is a $C^{\ast}$-algebra, and $\phi\colon \mathcal{S}\rightarrow \mathcal{B}$ is a linear map, then we say $\phi$ is positive if it maps positive elements of $\mathcal{S}$ to positive elements of $\mathcal{B}$.
\end{definition}
In the special case where the $C^{\ast}$-algebra $\mathcal{B}$ is the complex numbers (i.e., $\phi$ is a positive linear functional), then we know from results in $C^{\ast}$-algebra theory that $\norm{\phi} = \phi(1)$. If $\mathcal{B}$ is an arbitrary $C^{\ast}$-algebra, it turns out that $\phi$ is still positive, but that the bound is different.
\begin{proposition}
  If $\phi\colon \mathcal{S}\rightarrow \mathcal{B}$ is a positive map, then $\norm{\phi}\leq 2\norm{\phi(1)}$.
\end{proposition}
\begin{proof}
  If $p$ is positive, then since $0\leq p \leq \norm{p}1$, it follows that $0\leq \phi(p)\leq \norm{p}\phi(1)$, so that $\norm{\phi(p)}\leq \norm{p}\norm{\phi(1)}$.

  If $p_1$ and $p_2$ are positive, then $\norm{p_1-p_2}\leq \max\left( \norm{p_1},\norm{p_2} \right)$, so if $h$ is self-adjoint in $\mathcal{S}$, we have
  \begin{align*}
    \phi(h) &= \frac{1}{2}\phi\left( \norm{h}1 + h \right) - \frac{1}{2}\phi\left( \norm{h}1 - h \right),
  \end{align*}
  giving
  \begin{align*}
    \norm{\phi(h)} &\leq \frac{1}{2}\max\left( \norm{\phi\left( \norm{h}1 + h \right)}, \norm{\phi\left( \norm{h}1 - h \right)}\right)\\
                   &\leq \norm{h}\norm{\phi(1)}.
  \end{align*}
  Finally, if $a$ is an arbitrary element of $\mathcal{S}$, then we may write the Cartesian decomposition $a = h + ik$, and find
  \begin{align*}
    \norm{\phi(a)} &\leq \norm{\phi(h)} + \norm{\phi(k)}\\
                   &\leq 2\norm{a}\norm{\phi(1)}.
  \end{align*}
\end{proof}
It turns out that this bound cannot be improved upon.
\begin{example}
  Consider the subspace $\mathcal{S}\subseteq C\left( S^1 \right)$ spanned by $1,z,\bar{z}$. Then, we may define $\phi\colon \mathcal{S}\rightarrow \M_2$ given by
  \begin{align*}
    \phi\left( a + bz + c \bar{z} \right) &= \begin{pmatrix}a & 2b \\ 2c & a\end{pmatrix}.
  \end{align*}
  It follows that an element of $\mathcal{S}$ is positive if and only if $c = \overline{b}$ and $a \geq 2\left\vert b \right\vert$, while a self-adjoint element of $\M_2$ is positive if and only if its diagonal entries and determinant are positive real numbers. Therefore, it follows that $\phi$ is a positive map.

  Yet,
  \begin{align*}
    2\norm{\phi(1)} &= 2\\
                    &= \norm{\phi(z)}\\
                    &\leq \norm{\phi},
  \end{align*}
  meaning that $\norm{\phi} = 2\norm{\phi(1)}$.
\end{example}
\begin{lemma}
  Let $\mathcal{A}$ be a unital $C^{\ast}$-algebra, and let $p_1,\dots,p_n$ be positive elements of $\mathcal{A}$ satisfying
  \begin{align*}
    \sum_{i=1}^{n } p_i &\leq 1.
  \end{align*}
  If $\lambda_1,\dots,\lambda_n$ are scalars with $\left\vert \lambda_i \right\vert\leq 1$, then we have
  \begin{align*}
    \norm{\sum_{i=1}^{n}\lambda_ip_i} &\leq 1.
  \end{align*}
\end{lemma}
\begin{proof}
  We have
  \begin{align*}
    \begin{pmatrix}\sum_{i=1}^{n}\lambda_ip_i & 0 & \cdots & 0 \\ 0 & 0 & \cdots & 0 \\ \vdots & \vdots & \ddots & \vdots \\ 0 & 0 & \cdots & 0\end{pmatrix} &= \begin{pmatrix}p_1^{1/2} & \cdots & p_n^{1/2}\\0 & \cdots & 0 \\ \vdots & \ddots & \vdots \\ 0 & \cdots & 0\end{pmatrix} \operatorname{diag}\left( \lambda_1,\dots,\lambda_n \right) \begin{pmatrix}p_1^{1/2} & 0 & \cdots & 0 \\ \vdots & \vdots & \ddots & \vdots \\ p_n^{1/2} & 0 & \cdots & 0\end{pmatrix}.
  \end{align*}
  The norm of the matrix on the left is $\norm{\sum_{i=1}^{n}\lambda_ip_i}$, while each of the three matrices on the right have norm less than $1$, since $\norm{a^{\ast}a} = \norm{a}^2$.
\end{proof}
This allows us to find some further restrictions on the norms of positive maps.
\begin{theorem}
  Let $\mathcal{B}$ be a unital $C^{\ast}$-algebra, and let $X$ be a compact Hausdorff space with $C(X)$ the space of continuous functions on $X$. Let $\phi\colon C(X)\rightarrow \mathcal{B}$ be a positive map. Then, $\norm{\phi} = \norm{\phi(1)}$.
\end{theorem}
\begin{proof}
  We may assume that $\phi(1)\leq 1$. Let $f\in C(X)$, $\norm{f}\leq 1$, and let $\ve > 0$. Choose a finite open cover $\set{U_1,\dots,U_n}$ with respective elements $x_1,\dots,x_j$ such that $\left\vert f(x) - f\left( x_j \right) \right\vert < \ve$ for all $x\in U_j$, and find a partition of unity $\set{p_1,\dots,p_n}$ subordinate to this open cover. Then, setting $\lambda_j = f\left( x_j \right)$, then if $p_i(x)\neq 0$ for some $i$, then $x\in U_i$, and $\left\vert f(x)-\lambda_i \right\vert < \ve$. Therefore, for any $x\in X$,
  \begin{align*}
    \left\vert f(x) - \sum_{i=1}^{n}\lambda_ip_i(x) \right\vert &\leq \sum_{i=1}^{n} \left\vert f\left( x \right)-\lambda_i \right\vert p_i(x)\\
                                                                &< \ve.
  \end{align*}
  By the lemma, we have that $\norm{\sum_{i=1}^{n}\lambda_i\phi\left( p_i \right)}\leq 1$, so
  \begin{align*}
    \norm{\phi(f)} &\leq \norm{\phi\left( f-\sum_{i=1}^{n}\lambda_ip_i \right)} + \norm{\sum_{i=1}^{n}\lambda_i\phi\left( p_i \right)}\\
                   &< 1 + \ve\norm{\phi}.
  \end{align*}
  Therefore, $\norm{\phi}\leq 1$.
\end{proof}
\section{Completely Positive Maps}%
\begin{definition}
  If $\mathcal{B}$ is a $C^{\ast}$-algebra and $\phi\colon \mathcal{S}\rightarrow \mathcal{B}$ is a linear map, then we may define $\phi_n\colon \M_n\left( \mathcal{S} \right)\rightarrow \M_n\left( \mathcal{B} \right)$ by $\phi_n\left( \left( a_{ij} \right)_{i,j} \right) = \left( \phi\left( a_{ij} \right) \right)_{i,j}$.

  We call $\phi$ $n$-positive if $\phi_n$ is positive, and we call $\phi$ completely positive if $\phi$ is $n$-positive for all $n$.

  We call $\phi$ completely bounded if
  \begin{align*}
    \norm{\phi}_{\operatorname{cb}} &\coloneq \sup_{n}\norm{\phi_n}
  \end{align*}
  is finite. If $\norm{\phi}_{\text{cb}}\leq 1$, then we call $\phi$ completely contractive.
\end{definition}
\begin{lemma}
  Let $\mathcal{A}$ be a unital $C^{\ast}$-algebra, and let $a,b\in \mathcal{A}$. Then,
  \begin{enumerate}[(i)]
    \item $\norm{a}\leq 1$ if and only if
      \begin{align*}
        \begin{pmatrix}1 & a \\ a^{\ast} & 1\end{pmatrix}
      \end{align*}
      is positive in $\M_2\left(\mathcal{A}\right)$.
    \item The matrix
      \begin{align*}
        \begin{pmatrix}1 & a \\ a^{\ast} & b\end{pmatrix}
      \end{align*}
      is positive in $\M_2(\mathcal{A})$ if and only if $a^{\ast}a\leq b$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  Faithfully represent $\mathcal{A}$ on $\mathcal{H}$ via $\pi\colon \mathcal{A}\rightarrow B\left( \mathcal{H} \right)$, and set $A = \pi(a)$. Then, if $\norm{A}\leq 1$, for any $x,y\in \mathcal{H}$, we have
  \begin{align*}
    \iprod{ \begin{pmatrix}1 & A \\ A^{\ast} & 1\end{pmatrix} \begin{pmatrix}x\\y\end{pmatrix} }{ \begin{pmatrix}x\\y\end{pmatrix} } &= \iprod{x}{y} + \iprod{Ay}{x} + \iprod{x}{Ay} + \iprod{y}{y}\\
                             &\geq \norm{x}^2 - 2\norm{A}\norm{x}\norm{y} + \norm{y}^2\\
                             &\geq 0.
  \end{align*}
  Conversely, if $\norm{A} > 1$< then there are unit vectors $x$ and $y$ such that $ \iprod{Ay}{x} < -1 $, so the inner product above would be negative.

  Now, if we let $B = \pi(b)$, then if we let $B\geq A^{\ast}A$, then $B-A^{\ast}A\geq 0$, so that $ \iprod{By}{y}\geq \iprod{Ay}{Ay} $ for all $y\in \mathcal{H}$, meaning that for all $x,y\in \mathcal{H}$, we have
  \begin{align*}
    \iprod{\begin{pmatrix}1 & A \\ A^{\ast} & B\end{pmatrix} \begin{pmatrix}x\\y\end{pmatrix}}{ \begin{pmatrix}x\\y\end{pmatrix} } &= \iprod{x}{x} + \iprod{Ax}{y} + \iprod{A^{\ast}x}{y} + \iprod{By}{y}\\
                            &\geq \iprod{x}{x} + 2\re \iprod{Ax}{y} + \iprod{Ay}{Ay}\\
                            &\geq \norm{x}^2 + \norm{Ay}^2 - 2\norm{Ay}\norm{x}\\
                            &\geq 0.
  \end{align*}
  In the case that $B\not\geq A^{\ast}A$, then there is some unit vector $y$ such that $ \iprod{By}{y} < \norm{Ay}^2 $, which would yield the analogous outcome as in the proof of (i).
\end{proof}
\begin{proposition}
  Let $\mathcal{S}$ be an operator system, $\mathcal{B}$ a unital $C^{\ast}$-algebra, and $\phi\colon \mathcal{S}\rightarrow \mathcal{B}$ a unital $2$-positive map. Then, $\phi$ is contractive.
\end{proposition}
\begin{proof}
  Let $a\in \mathcal{S}$ be with $\norm{a}\leq 1$. Then,
  \begin{align*}
    \phi_2 \left( \begin{pmatrix}1 & a \\ a^{\ast} & 1\end{pmatrix} \right) &= \begin{pmatrix}1 & \phi(a)\\\phi(a)^{\ast} & 1\end{pmatrix}
  \end{align*}
  is positive, so $\norm{\phi(a)}\leq 1$.
\end{proof}
\begin{proposition}
  Let $\mathcal{A}$ and $\mathcal{B}$ be unital $C^{\ast}$-algebras, $\phi\colon \mathcal{A}\rightarrow \mathcal{B}$ a unital $2$-positive map. Then, $\phi(a)^{\ast}\phi(a)\leq \phi\left(a^{\ast}a\right)$ for all $a\in \mathcal{A}$.
\end{proposition}
\begin{proof}
  We have
  \begin{align*}
    \begin{pmatrix}1 & a \\ 0 & 0\end{pmatrix}^{\ast} \begin{pmatrix}1 & a \\ 0 & 0\end{pmatrix} &= \begin{pmatrix}1 & a \\ a^{\ast} & a^{\ast}a\end{pmatrix}\\
                     &\geq 0,
  \end{align*}
  so
  \begin{align*}
    \begin{pmatrix}1 & \phi(a) \\ \phi(a) & \phi\left( a^{\ast}a \right)\end{pmatrix} &\geq 0.
  \end{align*}
  Using part (ii) of the above lemma gives our desired result.
\end{proof}
\begin{proposition}
  Let $\mathcal{A}$ and $\mathcal{B}$ be unital $C^{\ast}$-algebras, and let $\mathcal{M}\leq \mathcal{A}$ be a unital subspace, with $\mathcal{S} = \mathcal{M} + \mathcal{M}^{\ast}$. If $\phi\colon \mathcal{M}\rightarrow \mathcal{B}$ is unital and $2$-contractive, then the map $\varphi\colon \mathcal{S}\rightarrow \mathcal{B}$ given by $\varphi\left( a + b^{\ast} \right) = \phi(a) + \phi(b)^{\ast}$ is contractive and $2$-positive.
\end{proposition}
\begin{proof}
  Since $\phi$ is contractive, $\varphi$ is well-defined. Since $\M_2\left( \mathcal{S} \right) = \M_2\left( \mathcal{M} \right) + \M_2\left( \mathcal{M} \right)^{\ast}$, and $ \left( \varphi \right)_2 = \left( \varphi_2 \right) $, then since $\phi_2$ is contractive, $\varphi_2$ is positive and $\varphi$ is contractive.
\end{proof}
\begin{proposition}
  Let $\mathcal{A}$ and $\mathcal{B}$ be unital $C^{\ast}$-algebras, $\mathcal{M}$ a unital subspace of $\mathcal{A}$, $\mathcal{S} = \mathcal{M} + \mathcal{M}^{\ast}$. If $\phi\colon \mathcal{M}\rightarrow \mathcal{B}$ is unital and completely contractive, then $\varphi\colon \mathcal{S}\rightarrow \mathcal{B}$ as defined above is completely contractive and completely positive.
\end{proposition}
\begin{proof}
  Since $\phi_n$ is unital and contractive, $\varphi_n$ is positive, and since $\varphi_{2n} = \left( \varphi_n \right)_2$, we have $\varphi_n$ is contractive.
\end{proof}
\begin{proposition}
  Let $\mathcal{S}\subseteq \mathcal{A}$ be an operator system, and let $\mathcal{B}$ be a $C^{\ast}$-algebra, with $\phi\colon \mathcal{S}\rightarrow \mathcal{B}$ completely positive. Then, $\phi$ is completely bounded, satisfying $\norm{\phi(1)} = \norm{\phi} = \norm{\phi}_{\operatorname{cb}}$.
\end{proposition}
\begin{proof}
  We have $\norm{\phi(1)}\leq \norm{\phi}\leq \norm{\phi}_{\operatorname{cb}}$, so it suffices to show that $\norm{\phi}_{\operatorname{cb}}\leq \norm{\phi(1)}$. If $A = \left( a_{ij} \right)_{i,j}$ is in $\M_n\left( \mathcal{S} \right)$ with $\norm{A}\leq 1$, and $I_n$ is the unit matrix of $\M_n\left( \mathcal{A} \right)$, then the block matrix
  \begin{align*}
    P &= \begin{pmatrix}I_n & A \\ A^{\ast} & I_n\end{pmatrix}
  \end{align*}
  is positive, so
  \begin{align*}
    \phi_{2n}(p) &= \begin{pmatrix}\phi_n\left( I_n \right) & \phi_n\left( A \right) \\ \phi_n\left( A \right)^{\ast} & \phi_n\left( I_n \right)\end{pmatrix}
  \end{align*}
  is positive. Therefore, $\norm{\phi_n\left( \mathcal{A} \right)}\leq \norm{\phi_n\left( I_n \right)} = \norm{\phi(1)}$.
\end{proof}

\section{Dilations and Extensions}%
We discuss a characterization of completely positive maps, as well as a way to understand extensions of completely positive maps.
\subsection{Stinespring's Dilation Theorem}%
\begin{theorem}[Stinespring's Dilation Theorem]
  Let $\mathcal{A}$ be a unital $C^{\ast}$-algebra, and let $\phi\colon \mathcal{A}\rightarrow B\left( \mathcal{H} \right)$ be a completely positive map. Then, there exists a Hilbert space $\mathcal{K}$, a unital $\ast$-homomorphism $\pi\colon \mathcal{A}\rightarrow B\left( \mathcal{K} \right)$, and a bounded operator $V\colon \mathcal{H}\rightarrow \mathcal{K}$ with $\norm{\phi(1)} = \norm{V}^2$ such that
  \begin{align*}
    \phi(a) &= V^{\ast}\pi(a)V.
  \end{align*}
\end{theorem}
\begin{proof}
  Let $A\odot \mathcal{H}$ be the algebraic tensor product, and define a symmetric bilinear map
  \begin{align*}
    \left( a\otimes x,b\otimes y \right) &= \iprod{\phi\left( b^{\ast}a \right)x}{y},
  \end{align*}
  and extend linearly, where $ \iprod{\cdot}{\cdot} $ is the inner product on $\mathcal{H}$. Then, since $\phi$ is completely positive, it follows that $ \left( \cdot,\cdot \right) $ is positive semidefinite, with
  \begin{align*}
    \left( \sum_{j=1}^{n}a_j\otimes x_j,\sum_{i=1}^{n}a_i\otimes x_i \right) &= \iprod{\phi_n\left( \left( a_i^{\ast}a_j \right)_{i,j} \right) \begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix}}{ \begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix} }\\
                                                                             &\geq 0.
  \end{align*}
  Since positive semidefinite bilinear forms satisfy the Cauchy--Schwarz inequality, we may define the subspace
  \begin{align*}
    \mathcal{N} &= \set{u\in \mathcal{A}\odot \mathcal{H} | \left( u,u \right) = 0}\\
                &= \set{u\in \mathcal{A}\odot \mathcal{H} | \left( u,v \right) = 0\text{ for all }v\in \mathcal{A}\odot \mathcal{H}},
  \end{align*}
  with an induced bilinear form on the quotient space $A\odot \mathcal{H}/\mathcal{N}$ defined by
  \begin{align*}
    \iprod{u + \mathcal{N}}{ v + \mathcal{N} } &= \left( u,v \right).
  \end{align*}
  Define $\mathcal{K}$ to be the Hilbert space completion of $\mathcal{A}\odot \mathcal{H}/\mathcal{N}$. Now, define a linear map $\pi(a)\colon \mathcal{A}\odot \mathcal{H}\rightarrow \mathcal{A}\odot \mathcal{H}$ by
  \begin{align*}
    \pi(a)\left( \sum_{i=1}^{n}a_i\otimes x_i \right) &= \sum_{i=1}^{n} \left( aa_i \right)\otimes x_i.
  \end{align*}
  We have that the inequality in $\M_n\left( \mathcal{A} \right)$ given by
  \begin{align*}
    \left( a_i^{\ast}a^{\ast}aa_j \right)_{i,j} &\leq \norm{a^{\ast}a}\left( a_i^{\ast}a_j \right)_{i,j}
  \end{align*}
  is satisfied, giving
  \begin{align*}
    \left( \pi(a)\left( \sum_{j=1}^{n}a_j\otimes x_j \right),\pi(a)\left( \sum_{i=1}^{n}a_i\otimes x_i \right) \right) &= \sum_{i,j=1}^{n} \iprod{\phi\left( a_i^{\ast}a^{\ast}aa_j \right)x_j}{x_i}\\
                                                                                                                                &\leq \norm{a^{\ast}a}\sum_{i,j=1}^{n} \iprod{\phi\left( a_i^{\ast}a_j \right)x_j}{x_i}\\
                                                                                                                                                                                                                                &= \norm{a}^2 \left( \sum_{j=1}^{n}a_j\otimes x_j,\sum_{i=1}^{n}a_i\otimes x_i \right).
  \end{align*}
  Therefore, $\pi(a)$ is invariant under $\mathcal{N}$, so induces a quotient map on $\mathcal{A}\otimes \mathcal{H}/\mathcal{N}$, which we will also denote by $\pi(a)$. We have that (this new) $\pi(a)$ is bounded with $\norm{\pi(a)}\leq \norm{a}$, meaning that it extends to a bounded linear map on $\mathcal{K}$.

  We define $V\colon \mathcal{H}\rightarrow \mathcal{K}$ by
  \begin{align*}
    V(x) &= 1\otimes x + \mathcal{N}.
  \end{align*}
  Then,
  \begin{align*}
    \norm{Vx}^2 &= \left( 1\otimes x,1\otimes x \right)\\
                &= \iprod{\phi(1)x}{x}\\
                &\leq \norm{\phi(1)}\norm{x}^2,
  \end{align*}
  and $\norm{V}^2 = \norm{\phi(1)}$.

  Finally, we observe that
  \begin{align*}
    \iprod{V^{\ast}\pi(a)Vx}{y} &= \left( \pi(a)1\otimes x,1\otimes y \right)\\
                                &= \iprod{\phi(a)x}{y}
  \end{align*}
  for all $x$ and $y$, so $V^{\ast}\pi(a)V = \phi(a)$.
\end{proof}
We observe that if $\phi$ is unital, then $V$ is an isometry, and we may identify $\mathcal{H}$ with the subspace $V\mathcal{H}$ of $\mathcal{K}$, and that $\phi(a) = P\pi(a)P$, where $P$ is the projection onto $\mathcal{H}$. In particular, this means that every unital completely positive map is the compression of a $\ast$-homomorphism.

This construction is very similar to the GNS representation, and we call the triple $\left( \pi,V,\mathcal{K} \right)$ a Stinespring representation for $\phi$.

If $\mathcal{K}_1 = \left[ \pi(\mathcal{A})V\mathcal{H} \right]$ denotes the closed linear span of the image of $\pi(\mathcal{A})V$, then the restriction of $\pi$ to $\mathcal{K}_1$ defines a $\ast$-homomorphism. If there is a Stinespring representation $\left( \pi_1,V,\mathcal{K}_1 \right)$ that satisfies $\mathcal{K}_1 = \left[ \pi_1\left( \mathcal{A} \right)V\mathcal{H} \right]$, then we call such a triple a \textit{minimal} Stinespring representation.

The minimal Stinespring representation is unique up to unitary equivalence.
\begin{proposition}
  Let $\left( \pi,V,\mathcal{H} \right)$ be a minimal Stinespring dilation of a completely positive contraction $\varphi\colon A\rightarrow B\left( \mathcal{H} \right)$. The, there exists a $\ast$-homomorphism $\rho\colon \varphi\left( A \right)'\rightarrow \pi\left( A \right)'$ such that $\varphi\left( a \right)x = V^{\ast}\pi(a)\rho(x)V$ for all $a\in A$ and all $x\in \varphi\left( A \right)'$.
\end{proposition}
One of the most important applications of the Stinespring theorem is showing a version of the Cauchy--Schwarz inequality, as well as a bimodularity property.
\begin{proposition}
  Let $A$ and $B$ be $C^{\ast}$-algebras, with $\varphi\colon A\rightarrow B$ a completely positive contraction.
  \begin{enumerate}[(i)]
    \item The inequality $\varphi\left( a \right)^{\ast}\varphi\left( a \right)\leq \varphi\left( a^{\ast}a \right)$ holds for all $a\in A$.
    \item If, for some $a\in A$, we have $\varphi\left( a^{\ast}a \right) = \varphi\left( a \right)^{\ast}\varphi\left( a \right)$ and $\varphi\left( aa^{\ast} \right) = \varphi\left( a \right)\varphi\left( a \right)^{\ast}$, then we have $\varphi\left( ba \right) = \varphi\left( b \right)\varphi\left( a \right)$ and $\varphi\left( ab \right) = \varphi\left( a \right)\varphi\left( b \right)$ for all $b\in A$.
    \item The subspace
      \begin{align*}
        A_{\varphi} &= \set{a\in A | \varphi\left( a^{\ast}a \right) = \varphi\left( a \right)^{\ast}\varphi\left( a \right)\text{ and }\varphi\left( aa^{\ast} \right) = \varphi\left( a \right)\varphi\left( a \right)^{\ast}}
      \end{align*}
      is a $C^{\ast}$-subalgebra of $A$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  Let $B\subseteq B\left( \mathcal{H} \right)$ be a faithful $\ast$-representation, and $\left( \pi,V,\mathcal{K} \right)$ a Stinespring dilation of $\varphi\colon A\rightarrow B\subseteq B\left( \mathcal{H} \right)$. For every $a\in A$, we have
  \begin{align*}
    \varphi\left( a^{\ast}a \right) - \varphi\left( a \right)^{\ast}\varphi\left( a \right) &= V^{\ast}\pi\left( a \right)^{\ast}\left( 1-VV^{\ast} \right)\pi\left( a \right)V\\
                                                                                            &\geq 0
  \end{align*}
  as $V$ is a contraction. Therefore, (i) follows.

  Moreover, equality holds if and only if $\left( 1-VV^{\ast} \right)^{1/2}\pi(a)V = 0$, meaning that
  \begin{align*}
    \varphi\left( ba \right) - \varphi\left( b \right)\varphi\left( a \right) &= V^{\ast}\pi(b)\left( 1-VV^{\ast} \right)\pi(a)V\\
                                                                              &= 0
  \end{align*}
  for every $b\in A$. Symmetry gives (ii), and (iii) follows from (ii).
\end{proof}
The $C^{\ast}$-subalgebra $A_{\varphi}$ is called the \textit{multiplicative domain} of $\varphi$. 
\subsection{Arveson's Extension Theorem}%
We now begin by understanding relations between completely positive maps and the matrix algebras $\M_n\left( \C \right)$.
\begin{proposition}
  Let $A$ be a $C^{\ast}$-algebra, and let $\left( e_{ij} \right)_{i,j}$ denote the matrix units of $\M_n\left( \C \right)$. A map $\varphi\colon \M_n\left( \C \right)\rightarrow A$ is completely positive if and only if the matrix $\left( \varphi\left( e_{ij} \right) \right)_{i,j}$ is positive in $\M_n\left( A \right)$.
\end{proposition}
\begin{proof}
  Since $\left( \left( e_{ij} \right)_{i,j} \right)_{i,j}\in \M_n\left( \M_n\left( \C \right) \right)$ (the matrix of matrix units) is positive, as it is a multiple of a rank one projection, the forward direction follows immediately.

  In the reverse direction, let $a = \left( a_{ij} \right)_{i,j} = \left( \varphi\left( e_{ij} \right) \right)_{i,j}$ in $\M_n\left( A \right)$ be positive, and let $\left( b_{ij} \right)_{i,j} = a^{1/2}$ be the square root. By definition, then have
  \begin{align*}
    \varphi\left( e_{ij} \right) &= \sum_{k=1}^{n}b_{ki}^{\ast}b_{kj}.
  \end{align*}
  If $A\subseteq B\left( \mathcal{H} \right)$ is a faithful representation, define $V\colon \mathcal{H}\rightarrow \ell_2^n\otimes \ell_2^n\otimes \mathcal{H}$ by
  \begin{align*}
    V\xi &= \sum_{j,k=1}^{n}\zeta_j\otimes\zeta_k\otimes b_{kj}\xi,
  \end{align*}
  where $\set{\zeta_j}_{j=1}^{n}$ is the standard basis for $\ell_2^n$. Then, if $T = \left( t_{ij} \right)_{i,j}\in \M_n\left( \C \right)$, we have
  \begin{align*}
    \iprod{V^{\ast}\left( T\otimes 1 \otimes 1 \right)V\eta}{\xi} &= \iprod{\left( T\otimes1\otimes 1 \right)V\eta}{V\xi}\\
                                                                  &= \sum_{i,j,k,\ell=1}^{n} \iprod{T\zeta_j}{\zeta_i} \iprod{\zeta_k}{\zeta_{\ell}} \iprod{b_{kj}\eta}{b_{\ell i}\xi}\\
                                                                  &= \iprod{\varphi\left( \sum_{i,j=1}^{n}t_{ij}e_{ij} \right)\eta}{\xi}\\
                                                                  &= \iprod{\varphi(T)\xi}{\eta},
  \end{align*}
  for every $\xi,\eta\in \mathcal{H}$. Thus, $\varphi\left( T \right) = V^{\ast}\left( T\otimes1\otimes 1 \right)V$ for every $T\in \M_n\left( \C \right)$, so $\varphi$ is completely positive.
\end{proof}
In particular, for fixed $a_1,\dots,a_n\in A$, we may define $\varphi\colon \M_n\left( \C \right)\rightarrow A$ by $\varphi\left( e_{ij} \right) = a_ia_j^{\ast}$, and observe that this gives a completely positive map, as
\begin{align*}
  \left( \varphi\left( e_{ij} \right) \right) &= \begin{pmatrix}a_1a_1^{\ast} & a_1a_2^{\ast} & \cdots & a_1a_n^{\ast}\\a_2a_1^{\ast} & a_2a_2^{\ast} & \cdots & a_2a_n^{\ast} \\ \vdots & \vdots & \ddots & \vdots \\ a_na_1^{\ast} & a_na_2^{\ast} & \cdots & a_na_n^{\ast}\end{pmatrix}\\
                                              &= \begin{pmatrix}a_1 & 0 & \cdots & 0 \\ a_2 & 0 & \cdots & 0 \\ \vdots & \vdots & \ddots & \vdots \\ a_n & 0 & \cdots & 0\end{pmatrix} \begin{pmatrix}a_1 & 0 & \cdots & 0 \\ a_2 & 0 & \cdots & 0 \\ \vdots & \vdots & \ddots & \vdots \\ a_n & 0 & \cdots & 0\end{pmatrix}^{\ast}
\end{align*}
There is a similar reverse characterization. Given a linear map $\varphi\colon A\rightarrow \M_n\left( \C \right)$, define a linear functional $ \hat{\varphi}\colon \M_n\left( A \right)\rightarrow \C $ by
\begin{align*}
  \hat{\varphi}\left( \left( a_{ij} \right)_{i,j} \right) &= \sum_{i,j=1}^{n} \left( \varphi\left( a_{ij} \right) \right)_{i,j}
\end{align*}
\begin{proposition}
  A map $\varphi\colon A\rightarrow \M_n\left( \C \right)$ is completely positive if and only if $\hat{\varphi}$ is positive on $\M_n\left( A \right)$.
\end{proposition}
\begin{proof}
  Let $\set{\zeta_i}_{i=1}^{n}$ be the standard orthonormal basis for $\ell_2^n$, and let
  \begin{align*}
    \zeta &= \begin{pmatrix}\zeta_1\\\vdots\\\zeta_n\end{pmatrix}\\
          &\in \left( \ell_2^n \right)^n.
  \end{align*}
  We have
  \begin{align*}
    \hat{\varphi}\left( \left( a_{ij} \right)_{i,j} \right) &= \iprod{\varphi_n\left( \left( a_{ij} \right)_{i,j} \right)\zeta}{\zeta}
  \end{align*}
  whenever $\left( a_{ij} \right)_{i,j}\in \M_n\left( A \right)$, so if $\varphi_n$ is positive, then so too is $\hat{\varphi}$.

  In the reverse direction, let $\hat{\varphi}$ be positive, and let $L_2\left( \M_n\left( A \right),\hat{\varphi} \right)$ be the GNS representation with cyclic vector $\xi$. If $\left( e_{ij} \right)_{i,j}$ denotes the system of matrix units for $\M_n\left( \C \right)$, we may view them as elements in $\M_n\left( A \right)$ by tensoring with $1_A$. Define the operator $V\colon \ell_2^n\rightarrow L_2\left( \M_n\left( A \right),\hat{\varphi} \right)$ by
  \begin{align*}
    V\zeta_j &= \pi\left( e_{1j} \right)\xi.
  \end{align*}
  Then, we have
  \begin{align*}
    \varphi(a) &= V^{\ast}\pi\left( \operatorname{diag}\left( a,\dots,a \right) \right)V,
  \end{align*}
  so that $\varphi$ is completely positive.
\end{proof}
\begin{lemma}
  Let $E\subseteq A$ be an operator system, and let $\psi\colon E\rightarrow \C$ be a positive linear functional. Then, $\norm{\psi} = \psi(1)$, and any norm-preserving extension of $\psi$ to $A$ is positive.
\end{lemma}
\begin{proof}
  Let $x\in E$ be such that $\norm{x}\leq 1$ and $\left\vert \psi(x) \right\vert$ is close to $\norm{\psi}$. We may assume that $0 < \psi(x)\in \R$ by multiplying by the sign of $\psi(x)$. Since positive maps are self-adjoint, we have
  \begin{align*}
    \psi(x) &= \frac{1}{2}\psi\left( x + x^{\ast} \right),
  \end{align*}
  so we may assume $x$ is self-adjoint. Yet, the operator inequality $x\leq \norm{x}1$ induces
  \begin{align*}
    \psi(x) &\leq \psi(1)\norm{x}.
  \end{align*}
  Since any functional satisfying $\psi(1) = \norm{\psi}$ is positive, we have that any extension to $A$ is positive.
\end{proof}
\begin{corollary}
  If $E\subseteq A$ is an operator system, and $\varphi\colon E\rightarrow \M_n\left( \C \right)$ is a completely positive map, then $\varphi$ extends to a completely positive map $A\rightarrow \M_n\left( \C \right)$.
\end{corollary}
\begin{proof}
  We may define the positive linear functional $\hat{\varphi}$ on $\M_n\left( E \right)$, which extends to a positive linear functional on $\M_n\left( A \right)$. Applying the correspondence in reverse gives our desired extension.
\end{proof}
\begin{theorem}
  Let $A$ be a unital $C^{\ast}$-algebra, and let $E\subseteq A$ be an operator system. Then, every contractive completely positive map $\varphi\colon E\rightarrow B\left( \mathcal{H} \right)$ extends to a contractive completely positive map $ \overline{\varphi}\colon A\rightarrow B\left( \mathcal{H} \right) $.
\end{theorem}
\begin{proof}
  Let $\left( P_i \right)_{i\in I}$ be an increasing net of finite-rank projections that converge to identity in SOT. For each $i$, consider the map $\varphi_i\colon E\rightarrow P_iB\left( \mathcal{H} \right)P_i$ given by $\varphi_i\left( e \right) = P_i\varphi\left( e \right)P_i$, which can be viewed as taking values in a matrix algebra.

  We may extend $ \varphi_i $ to be defined on all of $A$, and letting $\varphi_i$ take values in $B\left( \mathcal{H} \right)$, we use point-ultraweak compactness of the unit ball in $B\left( A,B\left( \mathcal{H} \right) \right)$ to find a cluster point $\Phi\colon A\rightarrow B\left( \mathcal{H} \right)$. This map $\Phi$ is completely positive and extends $\varphi$
\end{proof}
\section{Nuclearity and Exactness}%
\begin{definition}
  We call a map $\theta\colon A\rightarrow B$ between $C^{\ast}$-algebras \textit{nuclear} if there are contractive completely positive maps $\varphi_n\colon A\rightarrow \M_{k(n)}\left( \C \right)$ and $\psi_n\colon \M_{k(n)}\left( \C \right)\rightarrow B$ such that $\psi_n\circ\varphi_n\rightarrow\theta$ pointwise in the norm topology:
  \begin{align*}
    \norm{\psi_n\circ\varphi_n(a)\rightarrow\theta(a)}&\rightarrow 0
  \end{align*}
  for all $a\in A$.
\end{definition}
\begin{definition}
  If $A$ is a $C^{\ast}$-algebra, and $N$ is a von Neumann algebra, then a map $\theta\colon A\rightarrow N$ is called \textit{weakly nuclear} if there exist contractive completely positive maps $\varphi_n\colon A\rightarrow \M_{k(n)}\left( \C \right)$ and $\psi_n\colon \M_{k(n)}\left( \C \right)\rightarrow N$ such that $\psi_n\circ\varphi_n\rightarrow \theta$ pointwise in the ultraweak topology:
  \begin{align*}
    \eta\left( \psi_n\circ\varphi_n(a) \right)\rightarrow\eta\left( \theta(a) \right)
  \end{align*}
  for all $a\in A$ and normal functionals $\eta\in N_{\ast}$.
\end{definition}
By uniqueness of preduals, if $N\subseteq B\left(\mathcal{H}\right)$ is a faithful normal representation, then it suffices to observe that $\psi_n\circ\varphi_n\rightarrow\theta$ pointwise in the ultraweak topology if and only if
\begin{align*}
  \iprod{\psi_n\circ\varphi_n(a)v}{w} &\rightarrow \iprod{\theta(a)v}{w}
\end{align*}
for all $a\in A$ and $v,w\in \Omega$ for some collection $\Omega$ of vectors whose linear span is dense in $\mathcal{H}$.

One of the interesting aspects of nuclear maps is that whether a map is nuclear or not depends on the range.
\begin{proposition}
  Let $M\subseteq B(H)$ be a von Neumann algebra. The natural inclusion map $M\hookrightarrow B(H)$ is always weakly nuclear.
\end{proposition}
\begin{proof}
  Let $\set{P_i}_{i\in I}$ be a net of finite rank projections increasing to the identity. If $P_i$ has rank $k(i)$, then we may define maps $\varphi_i\colon M\rightarrow \M_{k(i)}\left( \C \right)\cong P_i B(H)P_i$ by compression, and let $\psi_i\colon \M_{k(i)}\left( \C \right) \rightarrow B(H)$ be natural inclusion maps.

  Since the predual of $B(H)$ is the trace class operators, we have that these maps converge weakly to the identity on $B(H)$, it follows that $M\hookrightarrow B(H)$ is weakly nuclear.
\end{proof}
\begin{proposition}
  A map $\theta\colon A\rightarrow B$ is nuclear if and only if for every finite $F\subseteq A$ and every $\ve > 0$, there exist $n\in \N$ and contractive completely positive maps $\varphi\colon A\rightarrow \M_n\left( \C \right)$, $\psi\colon \M_n\left( \C \right)\rightarrow B$ such that $\norm{\theta(a)-\psi\circ\varphi(a)} < \ve$ for all $a\in F$.
\end{proposition}
\begin{proof}
  Define the set $\mathcal{F} = \set{\left( F,\ve \right) | F\subseteq A\text{ finite, }\ve > 0}$, directed by
  \begin{align*}
    \left( F_1,\ve_1 \right)\preceq \left( F_2,\ve_2 \right) &\Leftrightarrow F_1\subseteq F_2\text{ and }\ve_2 \leq \ve_1.
  \end{align*}
  It can then be verified that convergence in the point-norm topology in the definition for nuclearity is equivalent to convergence via this directed set, and vice versa.
\end{proof}
Note that the point-norm topology is (effectively) equivalent to the strong operator topology, while the point-weak topology is (effectively) equivalent to the weak operator topology.
\begin{definition}
  A $C^{\ast}$-algebra $A$ is called \textit{nuclear} if the identity map is nuclear.
\end{definition}
\begin{definition}
  A $C^{\ast}$-algebra $A$ is called \textit{exact} if there exists a faithful representation $\pi\colon A\rightarrow B(H)$ such that $\pi$ is nuclear.
\end{definition}
\begin{definition}
   A von Neumann algebra $M$ is called \textit{semidiscrete} if the identity map is weakly nuclear.
\end{definition}
We will show now that if the double dual of a $C^{\ast}$-algebra is semidiscrete, then the $C^{\ast}$-algebra is nuclear.
\begin{lemma}
  Let $A$ be a Banach space, and let $B(A)$ be the space of all bounded linear maps from $A$ to $A$, and let $C\subseteq B(A)$ be a convex set. Then, the point-weak and point-norm closures of $C$ coincide.
\end{lemma}
\begin{proof}
  Let $T\in B(A)$ be a map such that $\left( T_i \right)_i\subseteq \mathcal{C}$ converges in the point-weak topology --- i.e., for all $a\in A$ and all $\eta\in A^{\ast}$,
  \begin{align*}
    \eta\left( T_ia \right) &\rightarrow \eta\left( Ta \right).
  \end{align*}
  We must show that for any finite subset $F = \set{a_1,\dots,a_k}$ of $A$ and every $\ve > 0$, there is $S\in \mathcal{C}$ such that $\norm{Sa_j - Ta_j} < \ve$ for all $1\leq j \leq k$.

  Since the net of maps $\left( T_i\oplus\cdots\oplus T_i \right)_{i}$ converges in the point-weak topology to $ T\oplus\cdots\oplus T $, we have that the element $ Ta_1\oplus\cdots\oplus Ta_k $ belongs to the weak closure of $\left( T_ia_1\oplus\cdots\oplus T_ia_k \right)$, so by the Hahn--Banach separation, it belongs to the norm closure of the convex hull. Therefore, we can find a convex combination of the $T_i$ that is close in norm to $T$ on all of $F$.
\end{proof}
\begin{definition}
  We say a completely positive map $\theta\colon A\rightarrow A$ is \textit{factorable} if there exist completely positive maps $\varphi\colon A\rightarrow \M_n\left( \C \right)$ and $\psi\colon \M_n\left( \C \right)\rightarrow A$ such that $\theta = \psi\circ\varphi$.
\end{definition}
\begin{proposition}
  The set of factorable maps is convex.
\end{proposition}
\begin{proof}
  Let $\theta_1,\theta_2\colon A\rightarrow A$ be factorable maps with factorizations given by $\left( \varphi_i,\psi_i,\M_{n_i}\left(\C\right) \right)$. We observe then that $\lambda \theta_1 + \left( 1-\lambda \right)\theta_2$ factorizes through $\M_{n_1}\left( \C \right)\oplus \M_{n_2}\left( \C \right)$, given by $\varphi_1\oplus \varphi_2$ and $\lambda \psi_1 + \left( 1-\lambda \right)\psi_2$ respectively.

  The set of positive operators is a cone, so this map is completely positive, and we may embed $\M_{n_1}\left( \C \right)\oplus \M_{n_2}\left( \C \right)\hookrightarrow \M_{n_3}\left( \C \right)$ to find this desired factorization.
\end{proof}
Note that this also shows that the space of contractive factorable maps is also convex.
\begin{proposition}
  If $A^{\ast\ast}$ is semidiscrete, then $A$ is nuclear.
\end{proposition}
\begin{proof}
  Suppose $A$ is unital, fix finite subsets $ F\subseteq A $ and $ Z\subseteq \left( A^{\ast\ast} \right)_{\ast}=A^{\ast} $, and $\ve > 0$. By some previous lemma,s it suffices to show that $ \id_A $ belongs to the point-weak closure of the contractive factorable maps. That is, there are contractive completely positive maps $\varphi\colon A\rightarrow \M_n\left( \C \right)$ and $\psi\colon \M_n\left( \C \right)\rightarrow A$ such that, for all $a\in F$ and $\eta\in Z$, we have
  \begin{align*}
    \left\vert \eta\left( \psi\circ\varphi(a) \right)-\eta(a) \right\vert < \ve.
  \end{align*}
  Since $A^{\ast\ast}$ is semidiscrete, we may find unital completely positive maps $\varphi\colon A^{\ast\ast}\rightarrow \M_n\left( \C \right)$ and $\psi'\colon \M_n\left( \C \right)\rightarrow A^{\ast\ast}$ such that $\left\vert \eta\left( \psi'\circ\varphi(a) \right) - \eta(a) \right\vert < \ve$ for all $a\in F$ and $\eta\in Z$.
\end{proof}

\subsection{Conditional Expectations}%
This subsection is copied from \href{https://ai-bearing.github.io/Classes_and_Homework/After\%20College/Other\%20Notes/vna_projections.pdf}{projections} notes.
\begin{definition}
  If $A$ is a unital $C^{\ast}$-algebra, and $B\subseteq A$ is a unital $C^{\ast}$-subalgebra, then a conditional expectation from $A$ to $B$ is a unital completely positive map $E\colon A\rightarrow B$ with $E|_{B} = \id$. 
\end{definition}
From \cite[Corollary 3.19]{paulsen_completely_bounded_maps}, every conditional expectation is bimodular --- that is, $E\left( axb \right) = aE(x)b$ for all $a,b\in B$ and $x\in A$.

It turns out that the stated properties are defining for conditional expectations.
\begin{theorem}
  Let $A$ be a unital $C^{\ast}$-algebra, $B\subseteq A$ a unital $C^{\ast}$-subalgebra, and $E\colon A\rightarrow B$ with $E|_{B} = \id$ and $\norm{E}\leq 1$. Then, $E$ is a conditional expectation.
\end{theorem}
\begin{proof}
  Start by considering the case that $A$ is a von Neumann algebra and $B$ is a von Neumann subalgebra.

  Then, for any $p\in P(B)$ and $x\in A$, we have $\left( 1-p \right)E\left( px \right) = E\left( \left( 1-p \right)E\left( px \right) \right)$, so for all $t > 0$,
  \begin{align*}
    \left( 1+t \right)^2 \norm{pE\left( \left( 1-p \right)x \right)}^2 &= \norm{pE\left( \left( 1-p \right)x + tpE\left( \left( 1-p \right)x \right) \right)}^2\\
                                                                       &\leq \norm{\left( 1-p \right)x + tpE\left( \left( 1-p \right)x \right)}^2\\
                                                                       &\leq \norm{\left( 1-p \right)x}^2 + t^2 \norm{pE\left( \left( 1-p \right)x \right)}^2,
  \end{align*}
  so for all $t > 0$, we have $\left( 1+2t \right)\norm{pE\left( \left( 1-p \right)x \right)}^2\leq \norm{\left( 1-p \right)x}^2$, so that $pE\left( \left( 1-p \right)x \right) = 0$. Since both $p$ and $1-p$ are in $B$, we have that $pE(x) = pE(px) = E(px)$. Since the span of the projections is norm-dense in $B$, we have $E\left( yx \right) = yE(x)$ for all $y\in B$ and all $x\in A$. Therefore, we have that $E$ is $B$-bimodular.

  Since $ E $ is a unital contraction, we have that $E$ is positive. If $\left( a_{ij} \right)_{i,j}\in \M_n\left( A \right)_{+}$ and $x_1,\dots,x_n\in B$, then we have
  \begin{align*}
    \sum_{i,j=1}^{n}x_i^{\ast}E\left( a_{ij} \right)x_j &= E\left( \sum_{i,j=1}^{n}x_i^{\ast}a_{ij}x_j \right)\\
                                                        &\geq 0,
  \end{align*}
  so $E^{(n)}\left( \left( a_{ij} \right)_{i,j} \right)\geq 0$, meaning $E$ is completely positive.

  In the general case, we embed $A$ into its double dual, $A^{\ast\ast}$, which is a von Neumann algebra, and similarly for $B$, and use the fact that $E^{\ast\ast}\colon A^{\ast\ast}\rightarrow B^{\ast\ast}$ is equal to $E$ on the subalgebras $A$ and $B$.
\end{proof}
\begin{theorem}
  Let $M$ be a finite von Neumann algebra with trace $\tau$, and let $N\subseteq M$ be a von Neumann subalgebra. Then, there exists a unique normal conditional expectation $E\colon M\rightarrow N$ with $\tau\circ E = \tau$.
\end{theorem}
\begin{proof}
  Let $e_N\in B\left( L_2\left( M,\tau \right) \right)$ be the projection onto $L_2\left( N,\tau \right)\subseteq L_2\left( M,\tau \right)$. Let $J$ be the conjugation operator on $L_2\left( M,\tau \right)$, which we can also view as the conjugation operator on $L_2\left( N,\tau \right)$.

  We have $N'\cap B\left( L_2\left( N,\tau \right) \right)$ from a basic result on the standard representation (see \href{https://ai-bearing.github.io/Classes_and_Homework/After\%20College/Other\%20Notes/orbit_equivalence.pdf}{orbit equivalence notes}). Since $L_2\left( N,\tau \right)$ is invariant under the operation $JNJ$, we must have $e_N\left( JyJ \right) = \left( JyJ \right)e_N$ for all $y\in N$.

  For any $x\in M$ and $y\in N$, we get
  \begin{align*}
    e_Nxe_NJyJ &= e_NxJyJ\\
               &= e_NxJyJxe_N\\
               &= JyJe_Nxe_N,
  \end{align*}
  so $e_Nxe_N\in \left( JNJ \right)' = N$, and we may denote this operator by $E(x)$. We have that $E\colon M\rightarrow N$ is normal, unital, completely positive, and $E|_{N} = \id$, so $E$ is a normal conditional expectation. Furthermore, for any $x\in M$, we have
  \begin{align*}
    \tau\left( E(x) \right) &= \iprod{e_Nxe_n1_{\tau}}{1_{\tau}}\\
                            &= \iprod{x1_{\tau}}{1_{\tau}}\\
                            &= \tau(x),
  \end{align*}
  where $1_{\tau}$ denotes the cyclic vector on the GNS representation.

  If $\hat{E}$ were another trace-preserving conditional expectation, then for any $x\in M$ and $y\in N$, we would have
  \begin{align*}
    \tau\left( \hat{E}(x)y \right) &=\tau\left( \hat{E}\left( xy \right) \right)\\
                                   &= \tau\left( xy \right)\\
                                   &= \tau\left( E\left( xy \right) \right)\\
                                   &= \tau\left( E(x)y \right),
  \end{align*}
  meaning that $\hat{E} = E$ as $x$ is arbitrary and $\tau$ is faithful.
\end{proof}
\subsection{Amenability}%
Now we can discuss how completely positive maps relate to amenability.

Recall that a (discrete, countable) group $G$ is called \textit{amenable} if there is a state $m\colon \ell_{\infty}\left( G \right)\rightarrow \C$ satisfying $m\left( \lambda_sf \right) = m(f)$for all $s\in G$, where $\lambda_sf(t) = f\left( s^{-1}t \right)$. There are a variety of equivalent characterizations, which are discussed in depth in \href{https://ai.avinash-iyer.com/Classes_and_Homework/College/Y4/Honors\%20Thesis/Amenability\%20Text/amenability.pdf}{this document}.
\begin{proposition}
  Let $G$ be a group. The following are equivalent:
  \begin{enumerate}[(i)]
    \item $G$ is amenable;
    \item there exists a net $\left( \xi_i \right)_{i\in I}$ of unit vectors in $\ell_2\left( G \right)$ such that, for all $g\in G$, we have
      \begin{align*}
        \lim_{i\in I}\norm{\lambda_g\xi_i-\xi_i}_{\ell_2} &= 0;
      \end{align*}
    \item there exists a net of finitely supported positive definite functions on $G$ that converge pointwise to $1$;
    \item there exists a net $\left( E_i \right)_{i\in I}$ of nonempty finite subsets of $G$ such that
      \begin{align*}
        \lim_{i\in I} \frac{\left\vert gE_i\triangle E_i \right\vert}{\left\vert E_i \right\vert} &= 0.
      \end{align*}
  \end{enumerate}
\end{proposition}
Our goal will be to use the theory of completely positive maps to introduce other characterizations of amenability both for groups and von Neumann algebras.
\begin{proposition}
  Let $G$ be a group, and let $M = L(G)$ be its group von Neumann algebra. Then, $G$ is amenable if and only if there exists a conditional expectation $E\colon B\left( L_2\left( M \right) \right)\rightarrow M$.
\end{proposition}
\begin{proof}
  Suppose $E$ exists. If $f\in \ell_{\infty}\left( G \right)$, let $M_f$ be the operator of multiplication by $f$ on $\ell_2\left( G \right)$. Set $m(f) = \tau\left( E\left( M_f \right) \right)$, where $\tau = \iprod{\cdot\delta_e}{\delta_e}$ is the canonical trace on $M$. Since $u_sM_fu_s^{\ast} = M_{\lambda_s f}$ for every $s\in G$, we have that $m$ is left invariant.

  Now, assume $G$ is amenable, and let $m$ be a mean on $\ell_{\infty}\left( G \right)$. Given $\xi,\eta\in \ell_2\left( G \right)$, and $T\in B\left( \ell_2\left( M \right) \right)$, define the function
  \begin{align*}
    f_{\xi,\eta}^{T}\left( s \right) &= \iprod{\xi}{\rho(s)T\rho\left( s^{-1} \right)\eta},
  \end{align*}
  where $\rho$ is the right regular representation of $G$. We have that $f_{\xi,\eta}^{T}$ is a bounded function on $G$ satisfying
  \begin{align*}
    \left\vert f_{\xi,\eta}^{T}(s) \right\vert &\leq \norm{T}\norm{\xi}\norm{\eta}.
  \end{align*}
  This allows us to define a continuous sesquilinear form
  \begin{align*}
    \left( \xi,\eta \right) &= m\left( f_{\xi,\eta}^{T} \right)
  \end{align*}
  on $\ell_2\left( G \right)$. By the structure of sesquilinear forms, we have a unique operator $E(T)$ satisfying
  \begin{align*}
    \iprod{\xi}{E(T)\eta} &= m\left( f_{\xi,\eta}^{T} \right).
  \end{align*}
  Since $m$ is invariant under the left-regular action, it is invariant under the right-regular action, so 
  \begin{align*}
    \rho(s)E(T)\rho\left( s^{-1} \right) = E(T),
  \end{align*}
  so $E(T)$ commutes with $\rho(G)$, meaning $E(T)\in L(G)$, and $E(T)$ is a conditional expectation as it is equal to identity on $L(G)$ with norm less than $1$.
\end{proof}
This property motivates the following definition.
\begin{definition}
  We say a von Neumann algebra is \textit{amenable} (or \textit{injective}) if $M$ can be faithfully represented as a von Neumann subalgebra of some $B\left( H \right)$ such that there exists a conditional expectation $E\colon B\left( H \right)\rightarrow M$.
\end{definition}
\begin{proposition}
  Let $M$ be a von Neumann algebra. The following are equivalent:
  \begin{enumerate}[(i)]
    \item $M$ is injective;
    \item for every inclusion $A\subseteq B$ of unital $C^{\ast}$-subalgebras, any unital completely positive map $\phi\colon A\rightarrow M$ extends to a completely positive map from $B$ to $M$;
    \item for any $B(H)$ that contains $M$ as a von Neumann subalgebra, there is a conditional expectation from $B(H)$ onto $M$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  To show (i) implies (ii), assume $M$ is a von Neumann subalgebra of $B(H)$, and let $E\colon B(H)\rightarrow M$ be a conditional expectation. Extend $\phi$ to a completely positive map $ \widetilde{\phi}\colon B\rightarrow B(H) $ by using Arveson's extension theorem. Then, $E\circ \widetilde{\phi}\colon B\rightarrow M$ is a completely positive extension of $\phi$.

  For (ii) implies (iii), if $M\subseteq B(H)$, then apply (ii) with $A = M$, $B = B(H)$, and the identity map $\id_M$. Then, there exists a completely positive map $\phi\colon B(H)\rightarrow M$ whose restriction to $M$ is $\id_M$, which is necessarily a conditional expectation.
\end{proof}
\subsection{Bimodules and Completely Positive Maps}%
There is a significant connection between completely positive maps and the notion of mbimodules over von Neumann algebras.
\begin{definition}
  Let $M$ and $N$ be finite von Neumann algebras. A Hilbert space $H$ is said to be a $M$-$N$-bimodule if it is equipped with two commuting normal $\ast$-representations $\pi\colon M\rightarrow B\left( H \right)$ and $\rho\colon N^{\op}\rightarrow B\left( H \right)$. We will write
  \begin{align*}
    x\cdot \xi \cdot y &= \pi(x)\rho\left( y^{\op} \right)\xi
  \end{align*}
  for all $\xi\in H$, $x\in M$, and $y\in N$.
\end{definition}
\begin{example}
  The following are important examples of bimodules.
  \begin{enumerate}[(i)]
    \item The standard representation $L_2\left( M \right)$ is an $M$-$M$-bimodule with representations of $M$ and $M^{\op}$ given by $x\cdot \xi \cdot y = x\xi y$.
    \item The coarse, denoted $L_2\left( M \right)\otimes L_2\left( N \right)$, is an $M$-$N$-bimodule given by $x\cdot \left( \xi\otimes \eta \right)\cdot y = \left( x\xi \right)\otimes \left( \eta y \right)$. We have that
      \begin{align*}
        L_2\left( M \right)\otimes L_2\left( N \right) &\cong \operatorname{HS}\left( L_2\left( M \right),L_2\left( N \right) \right),
      \end{align*}
      where $\operatorname{HS}\left( L_2\left( M \right),L_2\left( N \right) \right)$ is the bimodule of Hilbert--Schmidt operators from $L_2\left( M \right)$ to $L_2\left( N \right)$.
    \item If $\tau\in \aut\left( M \right)$ is a trace-preserving automorphism, then $L_2\left( M \right)$ can be equipped with a twisted representation $x\cdot \xi \cdot y = x\xi \theta\left( y \right)$.
    \item If $B\subseteq M$ is a von Neumann subalgebra, denote by $e_B\colon L_2\left( M \right)\rightarrow L_2\left( B \right)$ the orthogonal projection. The \textit{basic construction}, $ \left\langle M,e_B \right\rangle $ is the von Neumann subalgebra of $B\left( L_2\left( M \right) \right)$ generated by $M$ and $e_B$. Endowing $ \left\langle M,e_B \right\rangle $ with the semifinite trace
      \begin{align*}
        \Tr\left( xe_By \right) &= \tau\left( xy \right),
      \end{align*}
      then $L_2\left( \left\langle M,e_B \right\rangle,\Tr \right)$ is endowed with the structure of a $M$-$M$-bimodule. 
  \end{enumerate}
\end{example}
