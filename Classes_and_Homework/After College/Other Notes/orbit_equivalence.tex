\section{Polish Spaces and Standard Borel Spaces}%
We go over some preliminaries from descriptive set theory.
\begin{definition}
  If $\left( X,\tau \right)$ is a topological space, the \textit{Borel sets} of $X$ are elements of the $\sigma$-algebra generated by $\tau$. We denote the resulting measurable space as $\left( X,\mathcal{B}_X \right)$.
\end{definition}
Recall that a topological space is called \textit{Polish} if it is homeomorphic to a separable complete metric space.
\begin{proposition}
  Let $\left( X,\Omega \right)$ be a measurable space. Then, the following are equivalent:
  \begin{enumerate}[(i)]
    \item $\left( X,\Omega \right)$ is isomorphic to some $\left( Y,\mathcal{B}_Y \right)$, where $Y$ is separable and metrizable;
    \item $\left( X,\Omega \right)$ is isomorphic to some $\left( Y,\mathcal{B}_Y \right)$, where $Y\subseteq Z$ for some uncountable Polish space $Z$;
    \item $\left( X,\Omega \right)$ is countably generated and separates points --- that is, if $x,y\in X$ are distinct, then there is $A\in \Omega$ with $x\in A$ and $y\notin A$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  We have that (ii) implies (i) by definition, and (i) implies (iii) since separable metric spaces are Hausdorff. Therefore, we only need to show that (iii) implies (ii).

  Now, let $\set{A_n}_{n\in \N}$ generate $\Omega$ and separate the points of $X$. Define a function $f\colon X\rightarrow 2^{\N}$ by $f(x) = \left( \chi_{A_n}(x) \right)_{n}$, where $\chi_A$ is the indicator function for $A$. Then, $f$ is injective since $\set{A_n}_{n\in \N}$ separates points, and it is measurable since $f(x)_n = 1$ if and only if $x\in A_n$.

  Defining $Y = f(X)\subseteq 2^{\N}$, we have that since $f\left( A_n \right) = \set{y\in 2^{\N} | y(n) = 1}\cap Y$, $f^{-1}$ is measurable, so $\left( X,\Omega \right)$ and $\left( Y,\mathcal{B}_Y \right)$ are Borel isomorphic.
\end{proof}
\begin{theorem}[Kuratowski's Theorem]
  Let $X$ be a measurable space, and let $Y$ be a nonempty Polish space. If $Z\subseteq X$ and $f\colon Z\rightarrow Y$ is measurable, then there is a measurable function $\hat{f}\colon X\rightarrow Y$ extending $f$.
\end{theorem}
\begin{proof}
  It suffices to find a measurable subset $Z^{\ast}\subseteq X$ with $Z\subseteq Z^{\ast}$ and a measurable function $f^{\ast}\colon Z^{\ast}\rightarrow Y$ extending $f$.

  Let $\set{V_n}_{n\in \N}$ be a basis of nonempty open subsets of $Y$. There are measurable subsets $B_n$ in $X$ with $f^{-1}\left( V_n \right) = Z\cap B_n$. For any $z\in Z$, we have $z\in B_n$ if and only if $f(z)\in V_n$.

  Define
  \begin{align*}
    Z^{\ast} &= \set{x\in X | \text{there exists $y$ in $Y$ such that for all $n\in \N$, $x\in B_n$ if and only if $y\in V_n$}}.
  \end{align*}
  For any $x\in Z^{\ast}$, let $f^{\ast}(x) = y$, where
  \begin{align*}
    \set{y} &= \bigcap\set{V_n | x\in B_n}.
  \end{align*}
  We have $Z\subseteq Z^{\ast}$, so that $f^{\ast}$ extends $f$, and $f^{\ast}\colon Z^{\ast}\rightarrow Y$ is measurable, as $ \left( f^{\ast} \right)^{-1}\left( V_n \right) = B_n\cap Z^{\ast} $.
\end{proof}
\begin{definition}
  We say a measurable space $\left( X,\Omega \right)$ is a \textit{standard Borel space} if it is isomorphic to $\left( Y,\mathcal{B}_Y \right)$ for some Polish space $Y$. Equivalently, this holds if there is some Polish topology $\tau$ such that $\Omega$ is the $\sigma$-algebra generated by $\tau$.
\end{definition}
\begin{theorem}
  Let $\left( X,\tau \right)$ be a Polish space, and let $A\subseteq X$ be a Borel set. Then, there is a Polish topology $\tau_A\supseteq \tau$ such that $\sigma\left( \tau_A \right) = \sigma\left( \tau \right)$ and $A$ is clopen in $\tau_A$.
\end{theorem}
We will prove this with two lemmas.
\begin{lemma}
  Let $\left( X,\tau \right)$ be a Polish space, and let $F\subseteq X$ be closed. Let $\tau_F$ be the topology generated by $\tau$ and $F$. Then, $\tau_F$ is a Polish topology, $F$ is clopen in $\tau_F$, and $\sigma\left( \tau_F \right) = \sigma\left( \tau \right)$, where $\sigma\left( \tau \right)$ denotes the $\sigma$-algebra generated by the topology $\tau$.
\end{lemma}
\begin{proof}
  We observe that $\tau_F$ is the union of the subspace topologies on $F$ and $F^{c}$, so since the subspace topology on $F$ is Polish, it follows that $\tau_F$ is Polish.
\end{proof}
\begin{lemma}
  Let $\left( X,\tau \right)$ be a Polish space, and let $\left( \tau_n \right)_{n\in \N}$ be a sequence of Polish topologies with $\tau\subseteq \tau_n$ for each $n$. Then, the topology $\tau_{\infty}$ generated by the union of the $\tau_n$ is Polish. Furthermore, if $\tau_n\subseteq \sigma\left( \tau \right)$, then $\sigma\left( \tau_{\infty} \right)\subseteq \sigma\left( \tau \right)$.
\end{lemma}
\begin{proof}
  Let $X_n = X$ for each $n$. Define the map $\varphi\colon X\rightarrow \prod_{n}X_n$ given by $\varphi(x) = \left( x,x,\dots \right)$.

  Observe that $\varphi(X)$ is closed in $\prod_{n}\left( X_n,\tau_n \right)$. If we have $\left( x_n \right)_n\notin \varphi(x)$, then for some $i < j$, we have $x_i\neq x_j$, so there are disjoint open subsets $U,V$ in $\tau$ (hence in $\tau_i,\tau_j$ resp.) with $x_i\in U$ and $x_j\in V$. The basic open subset defined by $\pi_i^{-1}\left( U \right)\cap \pi_j^{-1}\left( V \right)\subseteq \prod_{n}X_n$ is thus not contained in $\varphi(X)$.

  In particular, this means $\varphi(X)$ is Polish. Yet, $\varphi$ is a homeomorphism of $\left( X,\tau_{\infty} \right)$ with $\varphi(X)$, so $\left( X,\tau_{\infty} \right)$ is Polish.

  Finally, if $\tau_n\subseteq \sigma\left( \tau \right)$, and $\set{U_i^{(n)}}_{i\in \N}$ is a basis for $\tau_n$, then $\set{U_i^{(n)}}_{i,n\in \N}$ is a subbasis for $\tau_{\infty}$, so $\tau_{\infty}\subseteq \sigma\left( \tau \right)$.
\end{proof}
\begin{proof}[Proof of Original Theorem]
  Let $\mathcal{S}$ be the set of subsets $A\subseteq X$ for which there exists a Polish topology $\tau_A\supseteq \tau$ with $\sigma\left( \tau_A \right) = \sigma\left( \tau \right)$ and $A$ is clopen in $\tau_A$.

  It is enough to show that $\tau\subseteq \mathcal{S}$ with $\mathcal{S}$ a $\sigma$-algebra. We observe that $\tau\subseteq \mathcal{S}$ by the first lemma since we may use any closed subsets. Furthermore, $\mathcal{S}$ is closed under complements. Finally, if $\set{A_n}_{n\in \N}\subseteq \mathcal{S}$, then we let $\tau_n = \tau_{A_n}$ satisfy the above condition. If $\tau_{\infty}$ is as in the previous lemma, then $A = \bigcup_{n}A_n$ is open in $\tau_{\infty}$, and applying the first lemma gives us our desired result.
\end{proof}
\begin{corollary}
  Let $X$ be a standard Borel space, $Y$ a polish space, and $f\colon X\rightarrow Y$ a Borel map. Then, there exists a Polish topology on $X$ which generates the same Borel structure such that $f$ is continuous with respect to this topology.
\end{corollary}
\begin{proof}
  Let $\set{E_n}_{n\in \N}$ be a countable basis for the topology on $Y$. There then exists a Polish topology on $X$ which generates the same Borel structure and such that $f^{-1}\left( E_n \right)$ is clopen for each $n$. In this topology, $f$ is continuous.
\end{proof}
\section{Orbit Equivalence}%
We will assume that $\left( X,\mu \right)$ is a nonatomic standard Borel probability space (i.e., a Borel probability space over a separable completely metrizable space $X$).
\begin{definition}
  A \textit{countable Borel equivalence relation} $\mathcal{R}\subseteq X\times X$ is a Borel subset and equivalence relation satisfying the assumption that
  \begin{align*}
    \left[ x \right]_{\mathcal{R}} &\coloneq \set{y\in X | \left( x,y \right)\in \mathcal{R}} 
  \end{align*}
  is countable.
\end{definition}
The set $\left[ \mathcal{R} \right]$ is known as the \textit{full group} of the equivalence relation $\mathcal{R}$ and consists of all Borel isomorphisms $\phi\colon X\rightarrow X$ with its graph satisfying $\operatorname{graph}\left(\phi\right)\subseteq \mathcal{R}$.

The set of all \textit{partial} Borel isomorphisms $\phi\colon \operatorname{dom}\left( \phi \right)\rightarrow \operatorname{ran}\left( \phi \right)$ with $\operatorname{graph}\left( \phi \right)\subseteq \mathcal{R}$ will be denoted $\left[ \left[ \mathcal{R} \right] \right]$.

The most basic example of a countable Borel equivalence relation is the space of orbits from the action of a countable discrete group $\Gamma\curvearrowright X$. We will write this as $\mathcal{R}\left( \Gamma\curvearrowright X \right)$. From now on, we will use ``equivalence relation'' to refer to countable Borel equivalence relations.

In fact, one can show that all such equivalence relations arise in this form.
\begin{theorem}[Feldman--Moore Theorem]
  If $\mathcal{R}$ is a countable Borel equivalence relation on $X$, then there exists a countable group $\Gamma$ and a Borel action $\Gamma\curvearrowright X$ such that $\mathcal{R} = \mathcal{R}\left( \Gamma\curvearrowright X \right)$. Moreover, both $\Gamma$ and the action can be chosen such that $\left( x,y \right)\in \mathcal{R}$ if and only if there is $g\in \Gamma$ with order $2$ such that $y = g\cdot x$.
\end{theorem}
We say the measure $\mu$ on $X$ is $\mathcal{R}$-invariant if
\begin{align*}
  \phi_{\ast}\mu &= \mu
\end{align*}
for all $\phi\in \left[ \mathcal{R} \right]$, where $\phi_{\ast}\mu\left( U \right) = \mu\left( \phi^{-1}\left( U \right) \right)$ for all Borel $U\subseteq X$.
\begin{proposition}
  The following are equivalent:
  \begin{enumerate}[(a)]
    \item $\mu$ is $\mathcal{R}$-invariant;
    \item $\mu$ is invariant under any $\Gamma$ where $\Gamma$ is such that $\mathcal{R} = \mathcal{R}\left( \Gamma\curvearrowright X \right)$;
    \item $\mu$ is $\Gamma$-invariant for some $\Gamma$ with $ \mathcal{R} = \mathcal{R}\left( \Gamma\curvearrowright X \right) $;
    \item for all $\phi\in \left[ \left[ \mathcal{R} \right] \right]$, we have $\mu\left( \operatorname{dom}\left( \phi \right) \right) = \mu\left( \operatorname{ran}\left( \phi \right) \right)$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  We have (d) implies (a) since $\left[ \left[ \mathcal{R} \right] \right]\supseteq \left[ \mathcal{R} \right]$, while (a) implies (b) since the action of $g$ is an element of $\left[ \mathcal{R} \right]$, and (b) implies (c) by definition. To see (c) implies (d), if $A = \operatorname{dom}\left( \phi \right)$ and $B = \operatorname{ran}\left( \phi \right)$, then there is a Borel partition $A = \bigcup_{n\geq 1}A_n$ and some $\set{\gamma_n}_{n\geq 1}$ such that for any $x\in A_n$, we have $\phi(x) = \gamma_n\cdot x$, meaning $B = \bigcup_{n\geq 1}\gamma_n\cdot A_n$.
\end{proof}

\begin{definition}
  A measure $\mu$ is called $\mathcal{R}$-ergodic if every $\mathcal{R}$-invariant Borel set is either null or co-null. The measure $\mu$ is ergodic with respect to the action of a group $\Gamma$ if it is ergodic for $\mathcal{R}\left( \Gamma\curvearrowright X \right)$.
\end{definition}
We discuss an example.

Let $\mu$ be the product measure on the space $X = 2^{\Gamma}$, where $\Gamma$ is a countably infinite group, and $2 = \set{0,1}$ assigns weight $1/2$ to each element. For each finite $S\subseteq \Gamma$, and map $s\colon S\rightarrow 2$, set
\begin{align*}
  \mathcal{N}_s &= \set{f | f|_{S} = s}.
\end{align*}
Then, the set $\mathcal{N}_s$ forms a clopen basis for $2^{\Gamma}$, and we may define $\mu\left( \mathcal{N}_s \right) = 2^{-\left\vert S \right\vert}$. Since
\begin{align*}
  \gamma\cdot \mathcal{N}_s &= \mathcal{N}_{\gamma\cdot s},
\end{align*}
we have that $\mu$ is shift-invariant.

In fact, we claim that $\mu$ is mixing, in that
\begin{align*}
  \lim_{n\rightarrow\infty} \mu\left( \gamma\cdot \left( A\cap B \right) \right) &= \mu\left( A \right)\mu\left( B \right)
\end{align*}
for any Borel $A,B\subseteq X$. Given any $A,B\subseteq X$ and $\ve > 0$, we may find finite unions of clopen sets $A'$ and $B'$ such that $\mu\left( A\triangle A' \right),\mu\left( B\triangle B' \right) < \ve$. In particular, this means we only need to show the mixing condition when $A$ and $B$ are finite unions of basic clopen sets. If $A$ is supported on $S$ and $B$ is supported on $T$, where $S,T$ are finite, then for all but finitely many $\gamma$, we have $\gamma\cdot S\cap T = \emptyset$, meaning $\gamma\cdot A$ and $B$ are independent, meaning
\begin{align*}
  \mu\left( \gamma\cdot A\cap B \right) &= \mu\left( \gamma\cdot A \right)\mu\left( B \right)\\
                                        &= \mu\left( A \right)\mu\left( B \right).
\end{align*}
\begin{definition}
  We say $\left( X,\mathcal{R} \right)$ and $\left( Y,\mathcal{S} \right)$ are Borel isomorphic if there is a Borel bijection $\pi\colon X\rightarrow Y$ such that $\left( x,y \right)\in \mathcal{R}$ if and only if $\left( \pi(x),\pi(y) \right)\in \mathcal{S}$.

  If $\mu$ and $\nu$ are invariant Borel probability measures on $X$ and $Y$ respectively, then we say they are almost everywhere Borel isomorphic if there are conull invariant Borel subsets $A\subseteq X$ and $B\subseteq Y$ and a Borel isomorphism $\pi\colon \left( A,\mathcal{R}|_{A} \right) \rightarrow \left( B,\mathcal{S}|_{B} \right)$ with $\pi_{\ast}\mu=\nu$.
\end{definition}
\begin{definition}
  We say two actions $\Gamma\curvearrowright X$ and $\Lambda\curvearrowright Y$ are \textit{orbit equivalent} if the respective orbit equivalence relations $ \mathcal{R}\left( \Gamma\curvearrowright X \right) $ and $\mathcal{R}\left( \Lambda\curvearrowright Y \right)$ are Borel isomorphic.

  If the actions of $\Gamma$ and $\Lambda$ are both measure-preserving actions on finite Borel measure spaces, then they are orbit equivalent if the equivalence relations are almost everywhere Borel isomorphic with $\pi_{\ast}\mu = \nu$.
\end{definition}
\section{Amenability and Hyperfiniteness}%
Our goal now is to show that non-atomic probability measure preserving ergodic actions of amenable groups are orbit equivalent. This is a conclusion of the combination of two theorems; a theorem of Dye says that any two non-atomic probability measure-preserving ergodic actions of $\Z$ are orbit equivalent, and a theorem of Ornstein and Weiss says that any probability measure-preserving action of an amenable group is orbit equivalent to one of $\Z$. This will eventually yield the celebrated Connes--Feldman--Weiss theorem, which states the following.
\begin{theorem}
  Let $\mathcal{R}$ be a countable Borel equivalence relation on $X$, and let $\mu$ be an $\mathcal{R}$-quasi-invariant probability measure. If $\mathcal{R}$ is $\mu$-amenable, then $\mathcal{R}$ is almost everywhere hyperfinite.
\end{theorem}
We will assume command of understanding of amenability in the case of groups, which is discussed in depth in \href{https://ai.avinash-iyer.com/Classes_and_Homework/College/Y4/Honors\%20Thesis/Amenability\%20Text/amenability.pdf}{this document}.
\subsection{Hyperfiniteness}%
We start by studying equivalence relations induced by $\Z$-actions.
\begin{definition}
  If $\mathcal{R}$ is a Borel equivalence relation on $X$, we say $\mathcal{R}$ is \textit{smooth} if it admits a family $\set{B_n}_{n\in \N}$ of Borel subsets of $X$ such that for all $x,y\in X$, $\left( x,y \right)\in \mathcal{R}$ if and only if, for all $n\in \N$, $x\in B_n$ if and only if $y\in B_n$.
\end{definition}
Via the maps $x\mapsto \set{\chi_{B_n}\left( x \right)}_{n\in \N}$, we see that this definition says $\mathcal{R}$ is smooth if and only if there is a Borel function $\phi\colon X\rightarrow Y$ with $Y$ a standard Borel space such that for all $x,y\in X$, we have $\left( x,y \right)\in \mathcal{R}$ if and only if $\phi(x) = \phi(y)$.
\begin{definition}
  If $\mathcal{R}$ is a Borel equivalence relation, we define $X/\mathcal{R}$ to be the set of equivalence classes of $\mathcal{R}$ equipped with the quotient Borel structure, wherein $A\subseteq X/\mathcal{R}$ is Borel if and only if $\bigcup A$ is Borel in $X$.
\end{definition}
\begin{proposition}
  A countable Borel equivalence relation $\mathcal{R}$ is smooth if and only if $X/\mathcal{R}$ is standard Borel.
\end{proposition}
\begin{proof}
  Let $\phi\colon X\rightarrow Y$ be a surjection that gives the smoothness of $\mathcal{R}$, and let $ \varphi $ be the induced bijection from $X/\mathcal{R}\rightarrow Y$. It suffices to verify that $\varphi$ induces an isomorphism o the quotient Borel structure $X/\mathcal{R}$ with the standard Borel structure of $Y$. Since $\phi$ is countable-to-one, we have that $\phi$ maps Borel sets to Borel sets, meaning that an $\mathcal{R}$-invariant set $B$ is Borel if and only if $ \varphi\left( B \right) = \phi(B) $ is Borel.

  In the reverse direction, if the quotient Borel structure on $X/\mathcal{R}$ is standard Borel, then there is a countable family of Borel sets $\set{B_n}_{n\in \N}$ in the quotient Borel structure that separates points. Yet, any such family induces a Borel separating family for $\mathcal{R}$.
\end{proof}
\begin{definition}
  A \textit{complete section} for $\mathcal{R}$ is a set $B\subseteq X$ that intersects every equivalence class in $\mathcal{R}$.

  A \textit{transversal} for $\mathcal{R}$ is a set $B\subseteq X$ that intersects every equivalence class in $\mathcal{R}$ exactly once.

  A \textit{selector} for $\mathcal{R}$ is a function $f\colon X\rightarrow X$ with $\operatorname{graph}\left( f \right)\subseteq \mathcal{R}$ and $f(X)\subseteq X$ is a transversal.
\end{definition}
A Borel equivalence relation admits a Borel transversal if and only if it admits a Borel selector; if it admits a Borel transversal, then it is smooth (but this is not true in reverse).
\begin{proposition}
  If $\mathcal{R}$ is a \textit{countable} smooth equivalence relation, then $\mathcal{R}$ admits a Borel selector. Moreover, if all the classes of $\mathcal{R}$ are of size $n\in \N\cup\set{\infty}$, then there is a sequence $\set{f_i}_{i < n}$ of Borel selectors for $\mathcal{R}$ whose images partition $X$.
\end{proposition}
\begin{proof}
  Let $\phi\colon X\rightarrow Y$ be a surjection that implements the smoothness of $\mathcal{R}$, and define a relation $\mathcal{S}\subseteq Y\times X$ by saying $\left( y,x \right)\in \mathcal{S}$ if and only if $\phi(x)=y$.

  Since $\mathcal{S}$ is Borel and has countable sections, it follows from the Lusin--Novikov theorem that there is a Borel uniformization $f\colon Y\rightarrow X$ of $\mathcal{S}$. It follows that $f\circ\phi$ is our desired Borel selector.

  Now, if all the classes of $\mathcal{R}$ are of size $n\leq \aleph_0$, there are then sequences $\set{f_i}_{i < n}$ of Borel uniformizations of $\mathcal{S}$ with disjoint ranges whose union is $X$, so $\set{f_i\circ\phi}_{i < n}$ is a sequence of Borel selectors whose images partition $X$.
\end{proof}
\begin{definition}
  A countable Borel equivalence relation $\mathcal{R}$ on $X$ is called \textit{hyperfinite} if there is an increasing exhaustive sequence $\set{\mathcal{S}_n}_{n\in \N}$ of finite Borel equivalence relations with $\mathcal{S}_0\subseteq \mathcal{S}_1\subseteq\cdots$, and
  \begin{align*}
    \bigcup_{n\in \N}\mathcal{S}_n &= \mathcal{R}.
  \end{align*}
\end{definition}
Here, a finite Borel equivalence relation is where each equivalence class is a finite set.
\begin{example}
  Define an equivalence relation $\mathcal{R}_0$ on $2^{\N}$ by saying $\left( x,y \right)\in \mathcal{R}_0$ if there exists $n\in \N$ such that for all $m\geq n$, we have $x_m = y_m$. This relation is hyperfinite, since we may define equivalence relations $\mathcal{R}_n$ by saying $\left( x,y \right)\in \mathcal{R}_n$ if, for all $m\geq n$, we have $x_m = y_m$.

  Yet, $\mathcal{R}_0$ is not smooth. For this, observe that if $\Delta = \bigoplus_{n\in \N}\Z/2\Z$ and $G = \prod_{n\in \N}\Z/2\Z$, with $\Delta$ acting on $G$ by translation (i.e., elementwise addition). The orbit equivalence relation is then $\mathcal{R}_0$, with the Haar measure $\mu$ being $\mathcal{R}_0$-ergodic. Yet, this means there can be no Borel transversal $T$ for $\mathcal{R}_0$, as then the probability measure $\nu$ on $T$ defined by $\nu\left( A \right) = \mu\left( \left[ A \right]_{\mathcal{R}_0} \right)$ would be non-atomic and discrete-valued, which cannot happen. Here, $\left[ A \right]_{\mathcal{R}_0}$ denotes the saturation of $A$, which is the set
  \begin{align*}
    \left[ A \right]_{\mathcal{R}} &= \set{x\in X | \text{there exists }y\in A\text{ such that }\left( x,y \right)\in \mathcal{R}}.
  \end{align*}
\end{example}
\begin{definition}
  If $\mathcal{R}$ is a Borel equivalence relation, we say $\mathcal{R}$ admits a \textit{vanishing sequence of markers} if there is a decreasing sequence of Borel complete sections for $\mathcal{R}$ whose intersection is the empty set.

  We say $\mathcal{R}$ is \textit{aperiodic} if all its classes are infinite.
\end{definition}
\begin{lemma}[Marker Lemma]
  If $\mathcal{R}$ is an aperiodic countable Borel equivalence relation, then $\mathcal{R}$ admits a vanishing sequence of markers.
\end{lemma}
\begin{proof}
  Since every uncountable standard Borel space is (Borel) isomorphic, we may assume that $\mathcal{R}$ is defined on $X = 2^{\N}$. If $s_n(x)$ is the lexicographically smallest $s\in 2^{n}$ such that $\left[ x \right]_{\mathcal{R}}\cap \mathcal{N}_s$ has infinite cardinality (where $\mathcal{N}_s$ is our neighborhood basis of clopen sets at $s$), we may define the sets $A_n$ by saying $x\in A_n$ if $x|_{n} = s_n(x)$.

  The set $\set{A_n}_{n\in \N}$ is a decreasing sequence of Borel sets that intersects every class of $\mathcal{R}$ at infinitely many points. Since
  \begin{align*}
    A &= \bigcap_{n\in \N} A_n
  \end{align*}
  intersects each class of $\mathcal{R}$ in at most one point, we have $B_n = A_n \setminus A$ is our desired vanishing sequence of markers.
\end{proof}
\begin{theorem}[Slaman--Steel, Weiss]
  Let $\mathcal{R}$ be a Borel equivalence relation. The following are equivalent:
  \begin{enumerate}[(i)]
    \item the relation $\mathcal{R}$ is hyperfinite;
    \item the relation $\mathcal{R}$ is induced by a Borel $\Z$-action.
  \end{enumerate}
\end{theorem}
\begin{proof}
  To see (i) implies (ii), we start with the case that $\mathcal{R}$ is smooth. Furthermore, breaking $\mathcal{R}$ into countably many invariant Borel subsets, we may assume that every class of $\mathcal{R}$ is of cardinality $n\in \N\cup\set{\infty}$. If $n\in \N$, we fix a Borel linear ordering of $X$, denoted $\prec$. If $x^{+}$ is the successor of $x$ when restricting $\prec$ to $\left[ x \right]_{\mathcal{R}}$, we have that
  \begin{align*}
    T(x) &= \begin{cases}
      x^{+} & \text{there exists $y\in \left[ x \right]_{\mathcal{R}}$ such that $x \prec y$}\\
      \min_{\prec}\left[ x \right]_{\mathcal{R}} & \text{else}
    \end{cases}
  \end{align*}
  gives our desired Borel $\Z$-action.

  If $n = \infty$, we partition $X$ into Borel transversals $\set{B_n}_{n\in \Z}$ of $\mathcal{R}$. Defining $T(x) = y$ whenever $\left( x,y \right)\in \mathcal{R}$, $x\in B_n$, and $y\in B_{n+1}$, we have that $T$ is a Borel automorphism of $X$ that induces $\mathcal{R}$.

  Now, let $\set{\mathcal{S}_n}_{n\in \N}$ be an increasing exhaustive sequence of finite Borel equivalence relations in $\mathcal{R}$, with $\mathcal{S}_0 = \Delta_{X}$ being the diagonal relation. Since $X/\mathcal{S}_n$ is a standard Borel space, we may find a Borel linear ordering $\prec_n$ of $X/\mathcal{S}_n$. Given distinct $\left( x,y \right)\in \mathcal{R}$, let $n\left( x,y \right)$ be the largest $n\in \N$ such that $\left( x,y \right)\notin \mathcal{S}_n$. Define $x \prec y$ to be such that $x\neq y$, $\left( x,y \right)\in \mathcal{R}$, and $\left[ x \right]_{\mathcal{S}_{n\left( x,y \right)}}\prec \left[ y \right]_{\mathcal{S}_{n\left( x,y \right)}}$.

  We see that $\prec$ discretely orders the classes of $\mathcal{R}$. The set of classes of $\mathcal{R}$ on which $\prec$ has a least or greatest element is a Borel invariant set on which $\mathcal{R}$ is smooth. Therefore, we may assume that $\prec$ is a $\Z$-ordering on each of the classes of $\mathcal{R}$, and the successor function for $\prec$ is the desired Borel automorphism that induces $\mathcal{R}$.

  In the reverse direction, suppose without loss of generality that $\mathcal{R}$ is induced by an aperiodic Borel automorphism $T$. On any smooth-in-$\mathcal{R}$ invariant Borel set, we may find an increasing and exhaustive sequence of finite Borel subequivalence relations. In each class $C$ in $\mathcal{R}$, define the $\Z$-ordering by saying $x\prec_{C}y$ if and only if there exists $n > 0$ such that $T^{n}(x) = y$.

  By the marker lemma, there is a vanishing sequence of markers $\set{B_n}_{n\in \N}$ that we may assume, for each $C$, that $B_n\cap C$ is unbounded in both directions in $\prec_{C}$. This follows from the fact that if $\set{A_n}_{n\in \N}$ is a vanishing sequence of markers for $\mathcal{R}$, then the union of the set of all classes $C$ for which $A_n\cap C$ has a least or largest element in $\prec_{C}$ for some $n$ is an invariant Borel set on which $\mathcal{R}$ is smooth.

  Let $\mathcal{F}_n$ be the finite equivalence relation contained in $\mathcal{R}$ defined by saying $\left( x,y \right)\in \mathcal{F}_n$ if and only if $x = y$, or $\left( x,y \right)\in \mathcal{R}$ and if $C = \left[ x \right]_{\mathcal{R}} = \left[ y \right]_{\mathcal{R}}$, the interval $\left[ x,y \right]$ avoids $B_n$. Then, $\set{\mathcal{F}_n}_{n\in \N}$ is an increasing and exhaustive sequence of finite Borel equivalence relations contained in $\mathcal{R}$ whose union is $\mathcal{R}$.
\end{proof}
