\section{The Standard Representation}%
Before we may start understanding the von Neumann algebra of countable Borel equivalence relations, we discuss the standard representation of a finite von Neumann algebra $\left( M,\tau \right)$.
\begin{proposition}
  Let $M$ be a finite von Neumann algebra with trace $\tau$. Then, the map $x\mapsto x^{\ast}$ extends to an anti-linear isometry $J\colon L_2\left( M,\tau \right)\rightarrow L_2\left( M,\tau \right)$ with $JMJ = M'\cap B\left( L_2\left( M,\tau \right) \right)$. Here, $L_2\left( M,\tau \right)$ refers to the GNS representation of $M$ with respect to the faithful state $\tau$.
\end{proposition}
\begin{proof}
  For any $x\in M$, we have
  \begin{align*}
    \norm{x^{\ast}}_{\tau}^2 &= \tau\left( xx^{\ast} \right)\\
                             &= \tau\left( x^{\ast}x \right)\\
                             &= \norm{x}_{\tau}^2.
  \end{align*}
  Thus, this extends to an anti-linear isometry $J\colon L_2\left( M,\tau \right)\rightarrow L_2\left( M,\tau \right)$. We have $ \iprod{J\xi}{J\eta} = \iprod{\xi}{\eta} $ for all $\xi,\eta\in L_2\left( M,\tau \right)$.

  For any $\xi\in L_2\left( M,\tau \right)$, define the unbounded operators $L_{\xi}^{0}\colon M1_{\tau}\rightarrow L_2\left( M,\tau \right)$ and $R_{\xi}^{0}\colon M1_{\tau}\rightarrow L_2\left( M,\tau \right)$ by
  \begin{align*}
    L_{\xi}^0\left( x1_{\tau} \right) &= \left( Jx^{\ast}J \right)\xi\\
    R_{\xi}^{0}\left( x1_{\tau} \right) &= x\xi.
  \end{align*}
  Here, $1_{\tau}$ refers to the cyclic vector for the representation $\tau$.

  If $\left( x_i \right)_i\in M$ and $\eta\in L_2\left( M,\tau \right)$ are such that $\norm{x_i}_{\tau}\rightarrow 0$ and $L_{\xi}^0\left( x_i1_{\tau} \right)\rightarrow\eta$, then for all $z\in M$, we have
  \begin{align*}
    \left\vert \iprod{\eta}{z} \right\vert &= \lim_{i} \left\vert \iprod{\left( Jx_i^{\ast}J \right)\xi}{z} \right\vert\\
                                           &= \lim_{i} \left\vert \iprod{\xi}{zx_i^{\ast}} \right\vert\\
                                           &\leq \lim_{i} \norm{\xi}_{\tau}\norm{z}_{\op}\norm{x_i}_{\tau}\\
                                           &= 0,
  \end{align*}
  meaning that $L_{\xi}^0$ is closable. Completing the graph gives rise to a closed operator $L_{\xi}$, and similarly $R_{\xi}^0$ is closable with corresponding operator $R_{\xi}$. We have that $L_{x1_{\tau}} = x$ for all $x\in M$.

  Now, if $x,y\in M$ and $\xi\in L_2\left( M,\tau \right)$, then
  \begin{align*}
    \iprod{R_{J\xi}\left( x1_{\tau} \right)}{y1_{\tau}} &= \iprod{xJ\xi}{y1_{\tau}}\\
                                                        &= \iprod{y^{\ast}x1_{\tau}}{\xi}\\
                                                        &= \iprod{x1_{\tau}}{R_{\xi}\left( y1_{\tau} \right)},
  \end{align*}
  meaning that $R_{J\xi} = R_{\xi}^{\ast}$. Additionally,
  \begin{align*}
    \left( JL_{\xi}J \right)\left( x1_{\tau} \right) &= xJ\xi\\
                                                     &= R_{J\xi} \left( x1_{\tau} \right)\\
                                                     &= R_{\xi}^{\ast}\left( x1_{\tau} \right),
  \end{align*}
  so that $JL_{\xi}J = R_{\xi}^{\ast}$.

  If we set $ \widetilde{M} = \set{L_{\xi} | L_{\xi}\in B\left( L_2\left( M,\tau \right) \right)} $ and $ \widetilde{N} = \set{R_{\xi} | R_{\xi}\in B\left( L_2\left( M,\tau \right) \right)} $, then we have $J \widetilde{M}J = \widetilde{N}$.

  Therefore, it suffices to show that $M = \widetilde{M}$ and $M' = \widetilde{N}$. For this, we observe that $M\subseteq \widetilde{M}$ and $ \widetilde{N}\subseteq M' $. For any $x\in M'$, let $\xi = x1_{\tau}$. For any $y\in M$, we then have
  \begin{align*}
    R_{\xi}\left( y1_{\tau} \right) &= y\xi\\
                                    &= yx1_{\tau}\\
                                    &= xy1_{\tau},
  \end{align*}
  meaning that $R_{\xi} = x$, so $M'\subseteq \widetilde{N}$. For any $L_{\xi}\in \widetilde{M}$, we then have $L_{\xi}\in \widetilde{N}' = M'' = M$, so $\widetilde{M}\subseteq M$.
\end{proof}
We observe that by considering the maps $R_{\xi}$ and $L_{\xi}$ as left and right multiplication by $M$, the standard representation can be considered as a $M$-$M$-bimodule. For more information on this, see the \href{https://ai-bearing.github.io/Classes_and_Homework/After\%20College/Other\%20Notes/completely_positive_maps.pdf}{notes} on completely positive maps.
\subsection{Affiliated Operators}%
If $\left( M,\tau \right)$ is a tracial von Neumann algebra acting on a Hilbert space $\mathcal{H}$, we will show that the closed densely defined operators on $\mathcal{H}$ affiliated with $M$ behave ``nicely'' in a certain sense.

To start, we discuss some spectral theory of unbounded self-adjoint operators. If $x$ is a self-adjoint operator on a Hilbert space $\mathcal{H}$ --- that is, a densely defined possibly unbounded operator with $x = x^{\ast}$. The spectrum $\sigma\left( x \right)$ is a closed subset of $\R$. There is a $\ast$-homomorphism $f\mapsto f(x)$ taking $B_{\infty}\left( \sigma(x) \right)$ to $B\left( \mathcal{H} \right)$.

The functional calculus gives a spectral measure $E\colon \Omega\rightarrow E(\Omega)$, defined by $E(\Omega) = \chi_{\omega}(x)$, on the Borel subsets of $\sigma(x)$. Setting $E_t = E\left( \left( -\infty,t \right] \right)$, we have
\begin{align*}
  f(x) &= \int_{\sigma(x)}^{} f(t)\:dE_t.
\end{align*}
The functional calculus is extended to the algebra of all Borel functions $B\left(\sigma(x)\right)$ as follows. If $f\in B\left( \sigma(x) \right)$, then we have $\operatorname{dom}\left( f(x) \right)$ is given by
\begin{align*}
  \operatorname{dom}\left( f(x) \right) &= \set{\eta\in \mathcal{H} | \int_{\sigma(x)}^{} \left\vert f(t) \right\vert^2\:d \iprod{\eta}{E_t\eta} < \infty},
\end{align*}
and we define
\begin{align*}
  \iprod{\xi}{f(x)\eta} &= \int_{\sigma(x)}^{} f(t)\:d \iprod{\xi}{E_t\eta}.
\end{align*}
Whenever $f$ is real-valued, this is a closed densely defined operator, and we may write
\begin{align*}
  f(x) &= \int_{\sigma(x)}^{} f(t)\:dE_t\\
  x &= \int_{\sigma(x)}^{} t\:dE_t.
\end{align*}
Furthermore, we have
\begin{align*}
  \norm{f(x)\xi}^2 &= \int_{\sigma(x)}^{} \left\vert f(t) \right\vert^2\:d \iprod{\eta}{E_t\eta}.
\end{align*}
Now, if $y\in B\left( \mathcal{H} \right)$, then we say $y$ commutes with an unbounded operator $z$ if $\operatorname{dom}\left( yz \right)\subseteq \operatorname{dom}\left( zy \right)$ and $zy = yz$ on $\operatorname{dom}\left( yz \right)$. A bounded operator $y$ commutes with a self-adjoint operator $x$ if and only if it commutes with all its spectral projections. In that case, it commutes with $f(x)$ for every $f\in B\left( \sigma(x) \right)$.

There is an analogue of the polar decomposition for closed densely defined operators.
\begin{proposition}
  Let $x$ be a closed densely defined operator on $\mathcal{H}$. Then, the following hold:
  \begin{enumerate}[(i)]
    \item $x^{\ast}x$ is a positive self-adjoint operator;
    \item there exists a unique partial isometry $u$ such that $x = u\left\vert x \right\vert$ and $\ker\left( x \right) =\ ker\left( u \right)$, where $\left\vert x \right\vert = \left( x^{\ast}x \right)^{1/2}$.
  \end{enumerate}
\end{proposition}
We have $u^{\ast}u$ is the source projection onto $\ker\left( x \right)^{\perp}$ and $uu^{\ast}$ is the range projection onto $\img\left( x \right)$, denoted $s_r(x)$ and $s_{l}(x)$ respectively. We call them the right support and left support of $x$ respectively (note that they are Murray--von Neumann equivalent projections). An operator $y\in B(H)$ commutes with $x$ if and only if it commutes with both $u$ and $\left\vert x \right\vert$.
\begin{definition}
  Let $M$ be a von Neumann algebra on a Hilbert space $\mathcal{H}$. We say an (unbounded) operator $x$ is \textit{affiliated} with $M$ if for every unitary $u\in U\left( M' \right)$, we have $ux = xu$. We will write $x \widetilde{\in} M$
\end{definition}
That is, the operators $ux = xu$ have the same domain and coincide on this common domain. Denote by $ \widetilde{M} $ the set of all closed densely defined operators affiliated with $M$. The following is a consequence of the double commutant theorem and what we discussed earlier.
\begin{proposition}
  Let $M$ be a von Neumann algebra, $x$ a closed densely defined operator on $\mathcal{H}$, and $x = u\left\vert x \right\vert$ its polar decomposition. Then, $x\in \widetilde{M}$ if and only if $u$ and the spectral projections of $\left\vert x \right\vert$ are in $M$.
\end{proposition}
In particular, the left and right supports $s_r(x)$ and $s_l(x)$ are contained in $M$ whenever $x\in \widetilde{M}$.

In the case when $\tau$ is a trace, then $ \widetilde{M} $ has a particularly nice property.
\begin{proposition}
  Let $\left( M,\tau \right)$ be a tracial von Neumann algebra on $\mathcal{H}$, and let $x,y\in \mathcal{M}$ be such that $x\subseteq y$. Then, $x = y$.
\end{proposition}
Here, $x\subseteq y$ means that $\operatorname{dom}\left( x \right)\subseteq \operatorname{dom}\left( y \right)$ and $x = y|_{\operatorname{dom}\left( x \right)}$.
\begin{proof}
  Let
  \begin{align*}
    G(x) &= \set{\left( \xi,x\xi \right) | \xi\in \operatorname{dom}\left( x \right)}.
  \end{align*}
  Almost by definition, $x$ is a closed operator when $G(x)$ is a closed subspace of $\mathcal{H}\oplus \mathcal{H} = \mathcal{H}^{(2)}$. Similarly, we introduce $G(y)$ which is the graph of $y$. Let $\left[ G(x) \right]$ and $\left[ G(y) \right]$ be the projections of $\mathcal{H}^{(2)}$ onto $G(x)$ and $G(y)$ respectively.

  The algebra $\M_2\left( M \right)$ of $2\times 2$ matrices with entries in $M$ is a von Neumann subalgebra of $B\left( \mathcal{H}^{(2)} \right)$ with commutant given by
  \begin{align*}
    \M_2\left( M \right)' &= \set{ \operatorname{diag}\left( a,a \right) | a\in M' }.
  \end{align*}
  We claim that the projections $\left[ G(x) \right]$ and $\left[ G(y) \right]$ are in $\M_2\left( M \right)$. Since $ x \widetilde{\in} M $, we have for every $u'\in U\left( M' \right)$ that $u'x = xu'$, so that
  \begin{align*}
    \operatorname{diag}\left( u',u' \right) \left[ G(x) \right] &= \left[ G(x) \right] \operatorname{diag}\left( u',u' \right),
  \end{align*}
  so that $\left[ G(x) \right]\in \M_2\left( M \right)'' = \M_2\left( M \right)$, and similarly for $\left[ G(y) \right]$.

  Consider the projection
  \begin{align*}
    p_1 &= \begin{pmatrix}1 & 0 \\ 0 & 0\end{pmatrix}
  \end{align*}
  in $\M_2\left( M \right)$. Then, $p_1$ is the left support of the element $p_1\left[ G(x) \right]$ and $\left[ G(x) \right]$ is its right support. Similar reasoning holds for $p_1\left[ G(y) \right]$, meaning that $\left[ G(x) \right]\sim p_1\sim \left[ G(y) \right]$.

  Since $x\subseteq y$, we have $G(x)\subseteq G(y)$, so $\left[ G(x) \right]\leq \left[ G(y) \right]$. Since $\M_2(M)$ has a trace, it is finite, so $\left[ G(x) \right] = \left[ G(y) \right]$, meaning $G(x) = G(y)$, whence $x = y$.
\end{proof}
In general, if we consider two closed densely defined operators $x$ and $y$ on $\mathcal{H}$, the domains interact as
\begin{align*}
  \operatorname{dom}\left( x+y \right) &= \operatorname{dom}\left( x \right)\cap \operatorname{dom}\left( y \right),
\end{align*}
which in general is not dense in $\mathcal{H}$ (and can even be equal to $0$). Yet, if $x,y\in \widetilde{M}$, we will show that $x+y$ is densely defined and closable, which allows us to define an addition (and product) in $\widetilde{M}$.
\begin{lemma}
  Let $x\in \widetilde{M}$. Then, for every $\ve > 0$, there is a projection $p\in M$ such that $ p\mathcal{H}\subseteq \operatorname{dom}\left( x \right) $ and $\tau\left( 1-p \right)\leq \ve$.
\end{lemma}
\begin{proof}
  Let $x = u\left\vert x \right\vert$ be the polar decomposition of $x$, and set
  \begin{align*}
    p_n &= \chi_{[0,n]} \left( \left\vert x \right\vert \right).
  \end{align*}
  Then, $p_n\mathcal{H}\subseteq \operatorname{dom}\left( x \right)$, with $\lim_{n\rightarrow\infty}\tau\left( p_n \right) = 1$, so we may find some $n$ large enough with $1-\tau\left( p_n \right)\leq \ve$.
\end{proof}
\begin{lemma}
  Let $V$ be a subspace of $\mathcal{H}$ such that for every $\ve > 0$, there is a projection $p\in M$ with $p\mathcal{H}\subseteq V$ and $\tau\left( 1-p \right) \leq \ve$. Then, $V$ is dense in $\mathcal{H}$.
\end{lemma}
\begin{proof}
  It suffices to construct an increasing sequence $\left( q_n \right)_n$ of projections with $\sup_{n}q_n = 1$ and $q_n\mathcal{H}\subseteq V$.

  For each $k\geq 1$, find $p_k\in M$ with $p_k\mathcal{H}\subseteq V$ and $\tau\left( 1-p_k \right)\leq 2^{-k}$. Defining
  \begin{align*}
    q_n &= \inf_{k > n} p_k,
  \end{align*}
  we have $q_n\mathcal{H}\subseteq V$, and
  \begin{align*}
    \tau\left( 1-q_n \right) &= \tau\left( \sup_{k > n}\left( 1-p_k \right) \right)\\
                             &\leq \sum_{k > n} \tau\left( 1-p_k \right)\\
                             &\leq \sum_{k > n} 2^{-k}\\
                             &= 2^{-n},
  \end{align*}
  so since $\tau$ is normal, we have $\tau\left( 1-\sup_n q_n\right) = 0$, whence $\sup_{n}q_n = 1$.
\end{proof}
\begin{lemma}
  Let $\left( p_i \right)_{i\in I}$ be a family of projections in $\left( M,\tau \right)$. Then,
  \begin{align*}
    \tau\left( \sup_{i}p_i \right) &\leq \sum_{i}\tau\left( p_i \right).
  \end{align*}
\end{lemma}
\begin{proof}
  Using the Kaplansky formula, given two projections $p,q\in M$, we have
  \begin{align*}
    \tau\left( p\vee q-p \right) &= \tau\left( q-p\wedge q \right),
  \end{align*}
  whence $\tau\left( p\vee q \right)\leq \tau(p) + \tau(q)$. By induction, we have the inequality when $I$ is finite, and the case for infinite $I$ follows from the normality of $\tau$.
\end{proof}
\begin{theorem}
  Let $\left( M,\tau \right)$ be a tracial von Neumann algebra on $\mathcal{H}$.
  \begin{enumerate}[(i)]
    \item If $x\in \widetilde{M}$, then $x^{\ast}\in \widetilde{M}$.
    \item If $x,y\in \widetilde{M}$, then $x + y$ and $xy$ are densely defined and closable, with closures in $\widetilde{M}$.
    \item We have $\widetilde{M}$ equipped with these operations is a $\ast$-algebra.
  \end{enumerate}
\end{theorem}
\begin{proof}
  \begin{enumerate}[(i)]
    \item Since $x^{\ast}$ and $x$ have the same domain, this follows from taking adjoints in the definition of $x$.
    \item We start by showing that if $x,y\in \widetilde{M}$, then $x + y$ is densely defined. Given $\ve > 0$, let $p,q\in P(M)$ be such that $p\mathcal{H}\subseteq \operatorname{dom}\left( x \right)$, $q\mathcal{H}\subseteq \operatorname{dom}\left( y \right)$, and $\tau\left( 1-p \right)\leq \ve/2$, $\tau\left( 1-q \right)\leq \ve/2$. Then, we have
      \begin{align*}
        \left( p\wedge q \right)\mathcal{H} &= p\mathcal{H}\cap q\mathcal{H}\\
                                            &\subseteq \operatorname{dom}\left( x \right)\cap \operatorname{dom}\left( y \right)\\
                                            &= \operatorname{dom}\left( x + y \right),
      \end{align*}
      with
      \begin{align*}
        \tau\left( 1-p\wedge q \right) &= \tau\left( \left( 1-p \right)\vee \left( 1-q \right) \right)\\
                                       &\leq \tau\left( 1-p \right) + \tau\left( 1-q \right)\\
                                       &\leq \ve,
      \end{align*}
      so by one of the above lemmas, we have $x + y$ is densely defined, and is affiliated with $M$. Similarly, since $x^{\ast}$ and $y^{\ast} $ are affiliated with $M$, so too is $x^{\ast} + y^{\ast}$. Since $x+y\subseteq \left( x^{\ast} + y^{\ast} \right)^{\ast}$, it follows that $x + y$ is closable, and the closure, which we also denote as $x + y$, is also affiliated with $M$.

      To prove that $xy$ is closable, consider the projections $p$ and $q$ as above. Then, $yq$ is closed and everywhere defined, so it is bounded. Let $r$ be the projection onto the kernel of the operator $\left( 1-p \right)yq$. Then, $r \mathcal{H}\subseteq \operatorname{dom}\left( xyq \right)$, and so $\left( q\wedge r \right) \mathcal{H}\subseteq \operatorname{dom}\left( xy \right)$. Since $1-r \preceq 1-p$, it follows that
      \begin{align*}
        \tau\left( 1-\left( q\wedge r \right) \right) &= \tau\left( \left( 1-q \right)\vee \left( 1-r \right) \right)\\
                                                      &\leq \tau\left( 1-q \right) + \tau\left( 1-p \right)\\
                                                      &\leq \ve,
      \end{align*}
      so that $xy$ is densely defined, meaning that $xy$ is closable. The closure, also denoted $xy$, is affiliated with $M$.
  \end{enumerate}
\end{proof}
\begin{example}
  Suppose $\left( X,\mu \right)$ is a probability measure space, and $M = L_{\infty}\left( X,\mu \right)$ with trace given by integration against $\mu$. Consider the standard representation of $M$ on $L_2\left( X,\mu \right)$, and let $f\colon X\rightarrow \C$ be a measurable function with multiplication operator $M_f$. Then, 
  \begin{align*}
    \operatorname{dom}\left( M_f \right) = \set{\xi\in L_2\left( X,\mu \right) | f\xi\in L_2\left( X,\mu \right)}
  \end{align*}
  is a closed, densely defined affiliated operator. It can also be shown that every closed densely defined operator affiliated with $M$ is of this form, meaning that $ \widetilde{M} $ can be identified with the complex-valued measurable functions modulo null sets.
\end{example}
\subsection{Noncommutative $L_2$ Spaces}%
We let $\left( M,\tau \right)$ be a tracial von Neumann algebra represented in standard form on $L_2\left( M,\tau \right)$ with unit cyclic vector $\hat{1}$. The representation of $M$ on $L_2\left( M,\tau \right)$ will be denoted $ \widehat{M} $.
\begin{definition}
  A closed densely-defined operator $x$ on $L_2\left( M,\tau \right)$ is called \textit{square-integrable} if $\hat{1}\in \operatorname{dom}\left( x \right)$.
\end{definition}
Given $\xi\in L_2\left( M,\tau \right)$, the densely defined operator $L_{\xi}$, given by extending the map $x\hat{1}\mapsto \left( JxJ^{\ast} \right)\xi$ is bounded if and only if $\xi\in\widehat{M}$.
\begin{theorem}
  For every $\xi\in L_2\left( M,\tau \right)$, the operator $L_{\xi}$ is square-integrable. Moreover, the map $\xi\mapsto L_{\xi}$ is a bijection from $L_2\left( M,\tau \right)$ onto the space of square-integrable operators.
\end{theorem}
\begin{proof}
  Let $\xi\in L_2\left( M,\tau \right)$. Then, every unitary operator in $M' = R\left( M \right)$ is of the form $R_u$, where $u\in U(M)$. For all $x\in M$, we have
  \begin{align*}
    R_uL_{\xi}\left( R_u^{\ast} \right)\hat{x} &= R_uL_{\xi} \widehat{xu^{\ast}}\\
                                               &= R_uR_{xu^{\ast}}\xi\\
                                               &= R_{xu^{\ast}u}\xi\\
                                               &= R_x\xi\\
                                               &= L_{\xi}\hat{x}.
  \end{align*}
  Since $\widehat{M}$ is an essential domain of both $R_uL_{\xi}\left( R_u \right)^{\ast}$ and $L_{\xi}$, it follows that they coincide, so $L_{\xi}$ is affiliated with $M$, with $\hat{1}$ in the domain of $L_{\xi}$.

  We show that $\xi\mapsto L_{\xi}$ is linear. For any $\xi,\eta\in L_2\left( M,\tau \right)$, we have
  \begin{align*}
    L_{\xi + \eta}^{0} &= L_{\xi}^{0} + L_{\eta}^{0}\\
                       &\subseteq L_{\xi} + L_{\eta},
  \end{align*}
  so that the closure of $L_{\xi + \eta}^{0}$ is such that $L_{\xi + \eta}\subseteq L_{\xi} + L_{\eta}$, whence they are equal as if $x,y\in \widetilde{M}$ with $x\subseteq y$, then $x = y$.

  The map $\xi\mapsto L_{\xi}$ is injective since $\xi = L_{\xi}\hat{1}$. Therefore, it remains to show surjectivity. Let $T$ be a square-integrable operator, and set $\xi = T\hat{1}$. Then, for any $x\in M$, we have
  \begin{align*}
    L_{\xi}\hat{x} &= R_xT\hat{1}\\
                   &= TR_x\hat{1}\\
                   &= T\hat{x},
  \end{align*}
  whence $L_{\xi}\subseteq T$ so $L_{\xi} = T$.
\end{proof}
Therefore, we may consider elements of $L_2\left( M,\tau \right)$ as operators under this identification. For any $x\in M$ and $\xi\in L_2\left( M,\tau \right)$, we have $\xi x$ and $x\xi$ are products of two operators in $ \widetilde{M} $, with adjoint corresponding to the canonical conjugation operator. For the latter, observe that if $x,y\in M$, then
\begin{align*}
  \iprod{\hat{x}}{L_{\xi}\hat{y}} &= \iprod{\hat{x}}{R_y\xi}\\
                                  &= \iprod{R_{y^{\ast}}\hat{x}}{\xi}\\
                                  &= \iprod{\widehat{xy^{\ast}}}{\xi}\\
                                  &= \iprod{J\xi}{\widehat{yx^{\ast}}}\\
                                  &= \iprod{J\xi}{R_{x^{\ast}}\hat{y}}\\
                                  &= \iprod{R_xJ\xi}{\hat{y}}\\
                                  &= \iprod{L_{J\xi}\hat{x}}{\hat{y}}.
\end{align*}
Therefore, we have $L_{J\xi}\subseteq \left( L_{\xi} \right)^{\ast}$, whence $L_{J\xi} = \left( L_{\xi} \right)^{\ast}$. We will often write $J\xi = \xi^{\ast}$, and say $\xi$ is self-adjoint whenever $\xi = J\xi$.

We can now provide characterizations of self-adjoint and positive square-integrable operators.
\begin{proposition}
  Let $\xi\in L_2\left( M,\tau \right)$. The following are equivalent:
  \begin{enumerate}[(i)]
    \item $\xi$ is self-adjoint;
    \item $ \iprod{\xi}{\hat{x}}\in \R $ for all $x\in M_{\sa}$;
    \item there exists a sequence $\left( x_n \right)_n\subseteq M_{\sa}$ such that $\lim_{n}\norm{\hat{x}_n-\xi}_{\tau} = 0$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  For any $\xi\in L_2\left( M,\tau \right)$ and $x\in M_{\sa}$, we have
  \begin{align*}
    \iprod{\xi}{\hat{x}} &= \overline{ \iprod{J\xi}{\hat{x}} },
  \end{align*}
  which gives the equivalence between (i) and (ii).

  Now, we show that (i) implies (iii). Let $\xi = J\xi$. Then, there exists a sequence $\left( x_n \right)_n\in M$ such that $\lim_{n}\norm{\hat{x}_n-\xi}_{\tau} = 0$. Setting
  \begin{align*}
    y_n &= \frac{x_n + x_n^{\ast}}{2}\\
        &\in M_{\sa},
  \end{align*}
  then since conjugation is an isometry, we have
  \begin{align*}
    \norm{\widehat{x^{\ast}}_n-J\xi}_{\tau} &= \norm{\hat{x}_n-\xi}_{\tau},
  \end{align*}
  so that $\lim_{n}\norm{\hat{y}_n-\xi}_{\tau} = 0$. The converse follows from the fact that $J$ is an isometry.
\end{proof}
We will say that $\xi\in L_2\left( M,\tau \right)$ is positive if $L_{\xi}$ is positive and self-adjoint. That is, $ \iprod{\eta}{L_{\xi}\eta} \geq 0 $ for all $\eta\in \operatorname{dom}\left( L_{\xi} \right)$.
\begin{proposition}
  Let $\xi\in L_2\left( M,\tau \right)$. The following are equivalent:
  \begin{enumerate}[(i)]
    \item $\xi$ is positive;
    \item $ \iprod{\hat{x}}{\xi} \geq 0 $ for all $x\in M_{+}$;
    \item there exists a sequence $\left( x_n \right)_n$ in $M_{+}$ such that $\lim_{n}\norm{\hat{x}_n-\xi}_{\tau} = 0$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  Note that $\xi$ is positive if and only if $ \iprod{\hat{x}}{L_{\xi}\hat{x}} \geq 0 $ for all $x\in M$. Write $x\in M_{+}$ as $x = yy^{\ast}$, so that
  \begin{align*}
    \iprod{\hat{x}}{\xi} &= \iprod{\hat{y}}{L_{\xi}\hat{y}},
  \end{align*}
  which gives the equivalence between (i) and (ii).

  Now, we prove (iii) implies (i). Suppose there is a sequence $\left( x_n \right)_n$ in $M_{+}$ such that $\lim_{n}\norm{\hat{x}_n-\xi}_{\tau} = 0$. Then, we have $\xi = J\xi$, so if we write $x_n = y_n^{\ast}y_n$, then
  \begin{align*}
    \iprod{\hat{x}}{x_n\hat{x}} &= \iprod{\widehat{y_n x}}{\widehat{y_n x}}\\
                                &\geq 0,
  \end{align*}
  with
  \begin{align*}
    \iprod{\hat{x}}{L_{\xi}\hat{x}} &= \iprod{\hat{x}}{R_x\xi}\\
                                    &= \lim_{n} \iprod{\hat{x}}{R_x\hat{x}_n}\\
                                    &= \lim_{n} \iprod{\hat{x}}{x_n\hat{x}}\\
                                    &\geq 0.
  \end{align*}
  Finally, for (iii) implies (i), let $e_n$ be the spectral projection of $\xi$ relative to $\left[ 0,n \right]$.

  Then, $e_n\xi\in M_{+}$, with $\lim_{n}\norm{e_n\xi - \xi}_{\tau} = 0$.
\end{proof}
If $a\in M$ is any element, then denote by $\tau_a$ the linear functional $x\mapsto \tau\left(ax \right)$ (where $\tau$ is the trace on $M$). Observe that if $a\in M_{+}$, then $\tau_a$ is a positive linear functional, and $\tau_a\leq \norm{a}\tau$.
\begin{proposition}[Noncommutative Radon--Nikodym Theorem, Version 1]
  Let $\varphi$ be a positive linear functional on $\left( M,\tau \right)$, and suppose there is $\lambda\in \R_{+}$ such that $\varphi\leq \lambda\tau$. Then, there is a unique $a\in M$ such that $\varphi = \tau_a$ with $0\leq a \leq \lambda 1$.
\end{proposition}
\begin{proof}
  Scaling allows us to assume that $\lambda = 1$. Define a linear functional $\psi$ on $ \widehat{M} $ by $\psi\left( \hat{x} \right) = \varphi(x)$. By Cauchy--Schwarz, we have
  \begin{align*}
    \left\vert \psi\left( \hat{x} \right) \right\vert &\leq \varphi\left( 1 \right)\varphi\left( x^{\ast}x \right)\\
                                                      &\leq \tau\left( x^{\ast}x \right)\\
                                                      &= \norm{\hat{x}}_{\tau}^2.
  \end{align*}
  Therefore, $\psi$ extends to a continuous linear functional on $L_2\left( M,\tau \right)$ that we still denote $\psi$. By the Riesz Representation theorem, there is $\xi\in L_2\left( M,\tau \right)$ such that $\psi\left( \eta \right) = \iprod{\xi}{\eta}$, where we use linearity in the second variable. In particular, we have $\varphi(x) = \iprod{\xi}{\hat{x}}$. We have that $\xi$ is positive.

  Similarly, we have
  \begin{align*}
    \iprod{\hat{1}-\xi}{\hat{x}} &= \tau\left( x \right)-\varphi\left( x \right)\\
                                 &\geq 0,
  \end{align*}
  meaning $\hat{1}-\xi \geq 0$, so that $0\leq \xi\leq \hat{1}$. This means that $\xi$ is norm-bounded by $\hat{1}$, meaning that $\xi = \hat{a}$ with $0\leq a \leq 1$, whence $\varphi = \tau_a$.

  If $\varphi = \tau_b$ for another $b\in M$, then $\tau\left( \left( a-b \right)\left( a-b \right)^{\ast} \right) = 0$, so $a = b$.
\end{proof}
The element $a$ is called the Radon--Nikodym derivative for $\varphi$ with respect to $\tau$. We will discuss a more powerful version of the Radon--Nikodym theorem after discussing integrable operators.

First, we recall the Powers--Størmer inequality, which is proven in \href{https://ai-bearing.github.io/Classes_and_Homework/After\%20College/Other\%20Notes/injective_implies_hyperfinite.pdf}{these notes}. For $\xi\in L_2\left( M,\tau \right)$, define $\omega_{\xi}$ to be the positive linear functional given by $ x\mapsto \iprod{\xi}{x\xi} $ on $M$. Then, the following holds.
\begin{theorem}[Powers--Størmer]
  We have
  \begin{align*}
    \norm{\xi-\eta}_{\tau}^2 \leq \norm{\omega_{\xi}-\omega_{\eta}}\leq \norm{\xi-\eta}_{\tau}\norm{\xi + \eta}_{\tau}
  \end{align*}
  for every $\xi,\eta\in L_{2}\left( M,\tau \right)_{+}$.
\end{theorem}
In particular, we can use this theorem to see that any \textit{normal} positive linear functional on $M$ is a vector state.
\begin{theorem}
  The map $\xi\mapsto \omega_{\xi}$ is a homeomorphism from $L_2\left( M,\tau \right)_{+}$ onto the cone of normal positive linear functionals on $M$.
\end{theorem}
\begin{proof}
  Injectivity follows from the inequality
  \begin{align*}
    \norm{\xi-\eta}_{\tau}^2 &\leq \norm{\omega_{\xi}-\omega_{\eta}}.
  \end{align*}
  Now, let $\varphi$ be a positive normal linear functional on $M$. We claim that for every $\ve > 0$, there is $a\in M_{+}$ such that $\norm{\varphi-\omega_{\hat{a}}}\leq \ve$. From the structure of normal linear functionals, we have some $\xi_1,\dots,\xi_m$ in $L_2\left( M,\tau \right)$ such that
  \begin{align*}
    \norm{\varphi-\sum_{k=1}^{m}\omega_{\xi_k}} &\leq \ve/2,
  \end{align*}
  so that there are $a_1,\dots,a_m\in M$ with
  \begin{align*}
    \norm{\varphi-\sum_{k=1}^{m}\omega_{\hat{a}_k}} &\leq \ve.
  \end{align*}
  Since we have
  \begin{align*}
    \sum_{k=1}^{m}\omega_{\hat{a}_k} &= \tau_b,
  \end{align*}
  where $b = \sum_{k=1}^{m}a_ka_k^{\ast}\geq 0$, the claim follows from setting $a = b^{1/2}$.

  Now, take a sequence $\left( b_n \right)_n$ in $M$ such that $\lim_{n}\omega_{\hat{b}_n} = \varphi$. From the Powers--Størmer inequality, the sequence $\left( \hat{b}_n \right)_n$ is Cauchy, so it converges to an element $\xi\in L_2\left( M,\tau \right)_{+}$, whence $\varphi = \omega_{\xi}$.

  The proof that it is a homeomorphism follows applying Powers--Størmer to establish continuity in both directions.
\end{proof}
\subsection{Noncommutative $L_1$ and the Predual}%
\begin{definition}
  Let $x\in \widetilde{M}$, and let $E$ be the spectral measure for $\left\vert x \right\vert$. We say $x$ is integrable if
  \begin{align*}
    \int_{\R}^{} \left\vert t \right\vert\:d\tau\left( E_t \right) &= \int_{\R}^{} \left\vert \tau \right\vert\:d \iprod{\hat{1}}{E_t\hat{1}},
  \end{align*}
  where $E_t = E\left( \left( -\infty,t \right) \right)$.
\end{definition}
The set of integrable operators is denoted $L_1\left( M,\tau \right)$. Generally, we may define $L_p\left( M,\tau \right)$ to be the space of all $x\in \widetilde{M}$ such that $ \int_{\R}^{} \left\vert t \right\vert^{p}\:d\tau\left( E_t \right) < \infty $. We have $L_{\infty}\left( M,\tau \right) = M$.

We have that $M\subseteq L_2\left( M,\tau \right)\subseteq L_1\left( M,\tau \right)$. Soon, we will see that $L_1\left( M,\tau \right)$ is a Banach space whose dual is $M$.

We extend $\tau|_{M_{+}}$ to a map $\tau\colon \widetilde{M}_{+}\rightarrow [0,\infty]$ by the formula
\begin{align*}
  \tau\left( x \right) &= \int_{\R_{+}}^{} t\:d\tau\left( E_t \right)\\
                       &= \int_{\R_{+}}^{} t\:d \iprod{\hat{1}}{E_t\hat{1}}.
\end{align*}
\begin{lemma}
  For every $x,y\in \widetilde{M}_{+}$ and $\lambda\geq 0$, we have
  \begin{align*}
    \tau\left( x + y \right) &= \tau\left( x \right) + \tau\left( y \right)\\
    \tau\left( \lambda x \right) &= \lambda \tau\left( x \right).
  \end{align*}
  Moreover, for every $x\in \widetilde{M}$,
  \begin{align*}
    \tau\left( x^{\ast}x \right) &= \tau\left( xx^{\ast} \right).
  \end{align*}
\end{lemma}
\begin{proof}
  Let $e_n$ be the spectral projection of $x + y$ corresponding to $\left[ 0,n \right]$. Then,
  \begin{align*}
    \tau\left( x + y \right) &= \lim_{n} \tau\left( \left( x+y \right)e_n \right)\\
                             &= \lim_{n} \tau\left( e_nxe_n + e_nye_n \right).
  \end{align*}
  Since both $\left( e_nxe_n \right)_n$ and $\left( e_nye_n \right)_n$ are bounded sequences with both $\left( \tau\left( e_nxe_n \right) \right)_n$ and $\left( \tau\left( e_nye_n \right) \right)_{n}$ increasing, we have
  \begin{align*}
    \tau\left( x + y \right) &= \lim_{n}\tau\left( e_nxe_n \right) + \lim_{n}\tau\left( e_nye_n \right)\\
                             &= \lim_{n} \norm{x^{1/2}\hat{e}_n}_{\tau}^2 + \lim_{n}\norm{y^{1/2}\hat{e}_n}_{\tau}^{2}.
  \end{align*}
  If we have $\tau(x) < \infty$, then $x^{1/2}\in L_2\left( M,\tau \right)$, whence
  \begin{align*}
    \lim_{n} \norm{x^{1/2}\hat{e}_n}_{\tau}^2 &= \norm{x^{1/2}}_{\tau}^2\\
                                              &= \tau(x),y
  \end{align*}
  so that $\tau\left( x + y \right) = \tau\left( x \right) + \tau\left( y \right)$ whenever $\tau\left( x \right) < \infty$ and $\tau\left( y \right) < \infty$.

  Now, we claim that if $\tau\left( x \right) = \infty$, then $\lim_{n}\tau\left( e_nxe_n \right) = \infty$. Else, it follows that $\left( x^{1/2}\hat{e}_n \right)_{n}$ is a Cauchy sequence in $L_2\left( M,\tau \right)$ that converges to some $\xi$. Since $x^{1/2}$ is a closed operator, it follows that $\hat{1}$ is in its domain with $x^{1/2}\hat{1} = \xi$, which is a contradiction. Then, since
  \begin{align*}
    \tau\left( e_n\left( x + y \right)e_n \right) &\geq \tau\left( e_nxe_n \right),
  \end{align*}
  it follows $\tau\left( x + y \right) = \infty = \tau\left( x \right) + \tau\left( y \right)$.

  The proof that $\tau\left( \lambda x \right) = \lambda \tau(x)$ follows from the linearity of $\tau$ when $\tau(x) < \infty$ and a comparable argument when $\tau(x) = \infty$.

  Finally, given a polar decomposition $x = u\left\vert x \right\vert\in \widetilde{M}$, where $u\in U\left( M \right)$, we see that 
  \begin{align*}
    E_t\left( u\left\vert x \right\vert^2u^{\ast} \right) &= uE_t\left( \left\vert x \right\vert^2 \right)u^{\ast},
  \end{align*}
  where $E_t(y)$ is the spectral projection of $y\in \widetilde{M}_+$ relative to $\left( -\infty,t \right]$. It follows that
  \begin{align*}
    \tau\left( E_t\left( u\left\vert x \right\vert^2u^{\ast} \right) \right) &= \tau\left( uE_t\left( \left\vert x \right\vert^2 \right)u^{\ast} \right)\\
                                                                             &= \tau\left( E_t\left( \left\vert x \right\vert^2 \right) \right).
  \end{align*}
\end{proof}
Comparable to the general discussion of \href{https://ai-bearing.github.io/Classes_and_Homework/After\%20College/Other\%20Notes/vna_projections.pdf}{weights and semifinite traces} on von Neumann algebras, we set
\begin{align*}
  \mathfrak{n} &= \set{x\in \widetilde{M} | \tau\left( x^{\ast}x \right) < \infty}\\
  \mathfrak{m} &= \set{\sum_{i=1}^{n}x_iy_i | x_i,y_i\in \mathfrak{n}}.
\end{align*}
\begin{lemma}
  Let $\tau\colon \widetilde{M}_{+}\rightarrow [0,\infty]$ be defined as above. Then,
  \begin{enumerate}[(a)]
    \item $ \mathfrak{n} $ and $ \mathfrak{m} $ are self-adjoint $M$-$M$-bimodules contained in $ \widetilde{M} $;
    \item we have
      \begin{align*}
        \mathfrak{m} \cap \widetilde{M}_{+} &= \set{ x\in \widetilde{M}_{+} | \tau(x) < \infty },
      \end{align*}
      and $ \mathfrak{m} $ is generated as a subspace $ \mathfrak{m}\cap \widetilde{M}_{+} $;
    \item the restriction of $\tau$ to $ \mathfrak{m}\cap \widetilde{M}_{+} $ extends to a unique linear functional $\tau$ on $ \mathfrak{m} $ with $\tau\left( x^{\ast} \right) = \overline{\tau(x)}$ for every $x\in \mathfrak{m}$;
    \item we have $\tau\left( xy \right) = \tau\left( yx \right)$ for any $x,y\in \mathfrak{n}$ or $x\in M$ and $y\in \mathfrak{m}$.
  \end{enumerate}
\end{lemma}
