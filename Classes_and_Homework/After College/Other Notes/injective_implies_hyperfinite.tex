\section{Overviewing Amenability in von Neumann Algebras}%
This material is taken from the \href{https://ai-bearing.github.io/Classes_and_Homework/After\%20College/Other\%20Notes/completely_positive_maps.pdf}{completely positive maps} notes.

Recall that a discrete group $G$ is called amenable if it admits $m\in \ell_{\infty}\left( G \right)^{\ast}$ such that $m\left( \lambda_s f \right) = m(f)$, where $\lambda_sf(t) = f\left( s^{-1} t\right)$. There are a variety of other equivalent notions, such as the existence of a net $\left( \xi_i \right)_{i}\in \ell_1\left( G \right)$ satisfying
\begin{align*}
  \norm{\lambda_s\xi_i - \xi_i}_{\ell_1} &\rightarrow 0
\end{align*}
for all $s\in G$ (approximate means), or if $\lambda\colon G\rightarrow U\left( \ell_2\left( G \right) \right)$ be the left regular representation, we have some net of unit vectors $\left( \xi_i \right)_{i}\subseteq S_{\ell_2(G)}$ satisfying
\begin{align*}
  \norm{\lambda_(s)\xi_i-\xi_i}_{\ell_2} &\rightarrow 0,
\end{align*}
known as almost invariant vectors. The existence of almost invariant vectors is furthermore equivalent to the existence of a net $\left( F_i \right)_{i\in I}$ of finite subsets of $G$ whose union is equal to $G$, with the net satisfying
\begin{align*}
  \frac{\left\vert sF_i\triangle F_i \right\vert}{\left\vert F_i \right\vert} &\rightarrow 0,
\end{align*}
known as a Følner net (or sequence).

Most all of these have analogous ideas in von Neumann algebras. We will start by discussing an analogue of an invariant mean on a von Neumann algebra, known as a hypertrace.

Let $\left( M,\tau \right)$ be a von Neumann algebra, and let $L_2\left( M \right)$ denote the standard representation of $M$, which is the GNS representation of $M$ with respect to the faithful state $\tau$. We observe that $L_2\left( M \right)$ is equipped with an $M$-$M$-bimodule structure.

We start by introducing a way to characterize amenability of a group via its group von Neumann algebra.
\begin{proposition}
  Let $G$ be a group, and let $M = L(G)$ be its group von Neumann algebra. Then, $G$ is amenable if and only if there exists a conditional expectation $E$ from $B\left( L_2\left( M \right) \right)$ onto $M$.
\end{proposition}
\begin{proof}
  Suppose such an $E$ exists. If $f\in \ell_{\infty}\left( G \right)$ is some function, let $M_f$ be the multiplication operator on $\ell_2\left( G \right)$ corresponding to $f$. Set $m(f) = \tau\left( E\left( M_f \right) \right)$, where $\tau$ is the canonical trace on $M$. Since $u_sM_fu_s^{\ast} = M_{\lambda_sf}$ for every $s\in G$, we have that $m$ is left invariant.

  Now, suppose $G$ is amenable, and let $m$ be a mean on $\ell_{\infty}\left( G \right)$. Given $\xi,\eta\in \ell_2\left( G \right)$, and $T\in B\left( L_2\left( M \right) \right)$, let
  \begin{align*}
    f_{\xi,\eta}^{T}(s) &= \iprod{\xi}{\rho(s)T\rho\left( s^{-1} \right)\eta},
  \end{align*}
  where $\rho$ is the right regular representation of $G$. Then, $f_{\xi,\eta}^{T}$ is a bounded function on $G$ satisfying
  \begin{align*}
    \left\vert f_{\xi,\eta}^{T}\left( s \right) \right\vert &\leq \norm{T}\,\norm{\xi}\,\norm{\eta},
  \end{align*}
  which allows us to define a continuous sesquilinear form
  \begin{align*}
    \left( \xi,\eta \right) &= m\left( f_{\xi,\eta}^{T} \right).
  \end{align*}
  By the structure of sesquilinear form,s we have a unique operator $E(T)$ satisfying
  \begin{align*}
    \iprod{\xi}{E(T)\eta} &= m\left( f_{\xi,\eta}^{T} \right).
  \end{align*}
  Since $m$ is invariant under the left regular action, it is invariant under the right regular action, so that 
  \begin{align*}
    \rho(s)E(T)\rho\left( s^{-1} \right) = E\left( T \right),
  \end{align*}
  meaning that $E(T)$ commutes with $\rho(G)$, whence $E(T)\in L(G)$. Furthermore, $E(T)$ is a conditional expectation as it is equal to the identity on $L(G)$ and has norm less than or equal to $1$.
\end{proof}
This property motivates the following definition.
\begin{definition}
  We say a von Neumann algebra is \textit{amenable} (or \textit{injective}) if $M$ can be faithfully represented as a von Neumann subalgebra of some $B\left( H \right)$ such that there exists a conditional expectation $E\colon B\left( H \right)\rightarrow M$.
\end{definition}
\begin{proposition}
  Let $M$ be a von Neumann algebra. The following are equivalent:
  \begin{enumerate}[(i)]
    \item $M$ is injective;
    \item for every inclusion $A\subseteq B$ of unital $C^{\ast}$-subalgebras, any unital completely positive map $\phi\colon A\rightarrow M$ extends to a completely positive map from $B$ to $M$;
    \item for any $B(H)$ that contains $M$ as a von Neumann subalgebra, there is a conditional expectation from $B(H)$ onto $M$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  To show (i) implies (ii), assume $M$ is a von Neumann subalgebra of $B(H)$, and let $E\colon B(H)\rightarrow M$ be a conditional expectation. Extend $\phi$ to a completely positive map $ \widetilde{\phi}\colon B\rightarrow B(H) $ by using Arveson's extension theorem. Then, $E\circ \widetilde{\phi}\colon B\rightarrow M$ is a completely positive extension of $\phi$.

  For (ii) implies (iii), if $M\subseteq B(H)$, then apply (ii) with $A = M$, $B = B(H)$, and the identity map $\id_M$. Then, there exists a completely positive map $\phi\colon B(H)\rightarrow M$ whose restriction to $M$ is $\id_M$, which is necessarily a conditional expectation.
\end{proof}
\begin{example}
  The hyperfinite $\text{II}_1$ factor, $R$, is amenable. Recall that the hyperfinite $\text{II}_1$ factor is defined to be the SOT-closure of the inductive limit of $Q_n = \M_{2^n}\left( \C \right)$, where we equip $Q_n\rightarrow Q_{n+1}$ with the block-diagonal embedding,
  \begin{align*}
    A &\mapsto \begin{pmatrix}A  & 0 \\ 0 & A\end{pmatrix}.
  \end{align*}
  Let $J$ be the conjugation operator on $L_2\left( R \right)$. For any $n\geq 1$ and $T\in B\left( L_2\left( R \right) \right)$, set
  \begin{align*}
    E_n(T) &= J \left( \int_{U\left( 2^{n} \right)}^{} uJTJu^{\ast}\:d\mu(u) \right)J,
  \end{align*}
  where $\mu$ denotes the Haar probability measure on the compact group $U\left( 2^{n} \right)$, the group of unitary $2^{n}\times 2^{n}$ matrices. Then, $\left( E_n\left( T \right) \right)_n$ is a norm-bounded sequence of operators in $B\left( L_2\left( R \right) \right)$.

  Take the limit of this sequence along a free ultrafilter, which allows for the existence of a unique bounded operator denoted $E(T)$ satisfying
  \begin{align*}
    \iprod{\xi}{E(T)\eta} &= \lim_{\omega} \iprod{\xi}{E_n\left( T \right)\eta}
  \end{align*}
  for all $\xi,\eta\in L_2\left( R \right)$. Observe that for any $n\geq n_0$, we have that $E_n(T)\in J\left( Q_{n_0} \right)'J$, so that
  \begin{align*}
    E(T) &\in \bigcap_{n_0\geq 1} J\left( Q_{n_0} \right)'J\\
         &= R,
  \end{align*}
  and $E(T) = T$ if $T\in R$. This gives that $E$ is our desired conditional expectation.
\end{example}
\begin{definition}
  Let $\left( M,\tau \right)$ be a tracial von Neumann algebra. A state $\psi$ on $B\left( L_2\left( M,\tau \right) \right)$ is called a \textit{hypertrace} if $\psi\left( xT \right) = \psi\left( Tx \right)$ for every $x\in M$ and every $T\in B\left( L_2\left( M,\tau \right) \right)$, with $\psi|_{M} = \tau$.
\end{definition}
\begin{proposition}
  Let $\left( M,\tau \right)$ be a tracial von Neumann algebra. Then, $M$ is amenable if and only if it has a hypertrace.
\end{proposition}
\begin{proof}
  If $M$ is amenable, then since $M\subseteq B\left( L_2\left( M,\tau \right) \right)$, we have a conditional expectation $E\colon B\left( L_2\left( M,\tau \right) \right)\rightarrow M$, so that $\tau\circ E$ is a hypertrace.

  Conversely, if $\psi$ is a hypertrace, then given $T\in B\left( L_2\left( M,\tau \right) \right)_{+}$, we may define a positive linear functional $\varphi$ on $M$ by taking $\varphi(x) = \psi\left( Tx \right)$. For any $x\in M_{+}$, we then have
  \begin{align*}
    \left\vert \varphi(x) \right\vert^2 &= \left\vert \psi\left( x^{1/2}Tx^{1/2} \right) \right\vert^2\\
                                        &\leq \psi\left( x^{1/2}T^2x^{1/2} \right)\psi\left( x \right)\\
                                        &\leq \norm{T}^2\psi\left( x \right)^2.
  \end{align*}
  Since $\psi|_{M} = \tau$, we have $\varphi(x)\leq \norm{T}\tau(x)$ for any $x\in M_{+}$, and in particular, $\varphi$ is normal. By the noncommutative Radon--Nikodym theorem, there is some $E(T)\in M_{+}$ such that for all $x\in M$,
  \begin{align*}
    \psi\left( Tx \right) &= \tau\left( E(T)x \right).
  \end{align*}
  We may extend $E$ to yield a conditional expectation from $B\left( L_2\left( M,\tau \right) \right)\rightarrow M$.
\end{proof}
\section{Approximation-Type Characterizations}%
Next, we will discuss some approximation-type properties (akin to the almost-invariant vectors characterization of amenability of $G$). For this, we must provide an overview of the trace-class and Hilbert--Schmidt operators on the space $L_2\left( M \right)$. This will allow us to use the Powers--Størmer inequality to prove that amenability of a von Neumann algebra is equivalent to the existence of almost central vectors in the coarse $M$-$M$-bimodule $L_2\left( M \right)\otimes L_2\left( M \right)$.
\begin{definition}
  If $H$ is a Hilbert space, then we set
  \begin{align*}
    S^1\left( H \right) &= \set{T\in K\left( H \right) | \Tr(T) = \sum_{i\in I} \iprod{\left\vert T \right\vert e_i}{e_i} < \infty}\\
    S^2\left( H \right) &= \set{T\in K\left( H \right) | \Tr\left( T^{\ast}T \right) < \infty}.
  \end{align*}
  We call $S^1\left( H \right)$ the trace-class operators, and $S^2\left( H \right)$ the Hilbert--Schmidt operators. The Hilbert--Schmidt operators admit an inner product and norm defined by
  \begin{align*}
    \iprod{A}{B} &= \Tr\left( B^{\ast}A \right)\\
    \norm{A}_{\operatorname{HS}} &= \Tr\left( A^{\ast}A \right)^{1/2}.
  \end{align*}
\end{definition}
Some of the following properties can be shown:
\begin{itemize}
  \item the value of $\Tr\left( A \right)$ is independent of orthonormal basis;
  \item the space $S^2\left( H \right)$ is a $B\left( H \right)$-$B\left( H \right)$-bimodule, as $\norm{AB}_{\operatorname{HS}}\leq \norm{A}\norm{B}_{\operatorname{HS}}$ and $\norm{BA}_{\operatorname{HS}}\leq \norm{A}\norm{B}_{\operatorname{HS}}$, where $A\in B(H)$ and $B\in S^2\left( H \right)$;
  \item the trace is tracial, in that $\Tr\left( AB \right) = \Tr\left( BA \right)$;
  \item the value $\norm{A}_{\Tr} = \Tr\left( \left\vert A \right\vert \right)$ defines a norm on $S^1\left( H \right)$;
  \item the space $S^1\left( H \right)$ is a $B\left( H \right)$-$B\left( H \right)$-bimodule;
  \item if $A,B\in S^2\left( H \right)$, then $\norm{AB}_{\Tr} \leq \norm{A}_{\operatorname{HS}}\norm{B}_{\operatorname{HS}}$.
\end{itemize}
We will focus on showing some of the properties for $\norm{\cdot}_{\Tr}$. This follows from the fact that if $A,B\in B\left( H \right)$, then
\begin{align*}
  0 &\leq \left( A-B \right)^{\ast}\left( A-B \right) \\
  &= A^{\ast}A - A^{\ast}B - B^{\ast}A + B^{\ast}B\\
                                              &= -2\re\left( B^{\ast}A \right) + \left\vert A \right\vert^2 + \left\vert B \right\vert^2,
\end{align*}
whence
\begin{align*}
  \re\left( B^{\ast}A \right) &\leq \frac{1}{2}\left( \left\vert A \right\vert^2 + \left\vert B \right\vert^2 \right).
\end{align*}
If $A\in S^1\left( H \right)$ is any trace-class operator and $B\in B(H)$ any operator where we have $1 = \norm{A}_{\Tr} = \norm{B}$, and we have partial isometries $AB = V\left\vert AB \right\vert$ and $A = W\left\vert A \right\vert$, then we observe that
\begin{align*}
  \norm{AB}_{\Tr} &= \Tr\left( \left\vert AB \right\vert \right)\\
                  &= \Tr\left( V^{\ast}AB \right)\\
                  &= \Tr\left( V^{\ast}W^{\ast}\left\vert A \right\vert^{1/2}\left\vert A \right\vert^{1/2}B \right)\\
                  &= \Tr\left( \left( B^{\ast}\left\vert A \right\vert^{1/2} \right)^{\ast}V^{\ast}W^{\ast}\left\vert A \right\vert^{1/2} \right)\\
                  &\leq \frac{1}{2}\left( \Tr\left( \left\vert A \right\vert^{1/2}BB^{\ast}\left\vert A \right\vert^{1/2} \right) + \Tr\left( \left\vert A \right\vert^{1/2}VV^{\ast}W^{\ast}\left\vert A \right\vert^{1/2} \right) \right)\\
                  &\leq \frac{1}{2}\left( \norm{B}\norm{A}_{\Tr} + \norm{A}_{\Tr} \right)\\
                  &= 1,
\end{align*}
whence $\norm{AB}_{\Tr}\leq \norm{A}_{\Tr}\norm{B}$, and similarly for $BA$.

Additionally, if $A + B = V\left\vert A + B \right\vert$, then
\begin{align*}
  \norm{A + B}_{\Tr} &\leq \norm{A}_{\Tr} + \norm{B}_{\Tr}.
\end{align*}
We know from some von Neumann algebra theory that the trace class operators are the predual of $B(H)$, and we may identify $S^2\left( H \right)$ with $H\otimes H^{\ast}$ by taking $\xi\otimes\eta^{\ast}\mapsto \theta_{\xi,\eta}$, where
\begin{align*}
  \theta_{\xi,\eta}\left( \zeta \right) &= \iprod{\zeta}{\eta}\xi.
\end{align*}
\begin{proposition}[Powers--Størmer Inequality]
  If $A,B\in S^{2}\left( H \right)$ are positive, then
  \begin{align*}
    \norm{A-B}_{\operatorname{HS}}^2 &\leq \norm{A^2 - B^2}_{\Tr}\\
                                     &\leq \norm{A + B}_{\operatorname{HS}}\norm{A - B}_{\operatorname{HS}}.
  \end{align*}
\end{proposition}
\begin{proof}
  We have
  \begin{align*}
    A^2 - B^2 &= \frac{1}{2}\left( \left( A + B \right)\left( A-B \right) + \left( A - B \right)\left( A + B \right) \right),
  \end{align*}
  whence
  \begin{align*}
    \norm{A^2 - B^2}_{\Tr} &\leq \frac{1}{2} \left( \norm{A+B}_{\operatorname{HS}}\norm{A-B}_{\operatorname{HS}} + \norm{A-B}_{\operatorname{HS}}\norm{A+B}_{\operatorname{HS}} \right)\\
                           &= \norm{A+B}_{\operatorname{HS}}\norm{A-B}_{\operatorname{HS}}.
  \end{align*}
  For the other inequality, we observe that if $x\in S^1\left( H \right)_{+}$ and $y\in B(H)_{+}$, then $\Tr\left( xy \right) = \Tr\left( y^{1/2}xy^{1/2} \right)\geq 0$.

  Let $e = \chi_{[0,\infty)}\left( A-B \right)$, and let $e^{\perp}\coloneq 1-e$. Then, we have
  \begin{align*}
    \Tr\left( \left( A-B \right)^2 \right) &= \Tr\left( \left( A-B \right)\left( A-B \right)e \right) + \Tr\left( \left( B-A \right)\left( B-A \right)e^{\perp} \right)\\
                                           &\leq \Tr\left( \left( A + B \right)\left( A-B \right)e \right) + \Tr\left( \left( B+A \right)\left( B-A \right)e^{\perp} \right).
  \end{align*}
  Since $e$ is a spectral projection of the operator $A - B$, it follows that $e$ and $A - B$ commute. Therefore, we have
  \begin{align*}
    \Tr\left( \left( A + B \right)\left( A - B \right)e \right) &= \Tr\left( \left( A + B \right)e \left( A-B \right) \right)\\
                                                                &= \Tr\left( \left( A-B \right)\left( A + B \right)e \right),
  \end{align*}
  whence
  \begin{align*}
    \Tr\left( \left( A - B \right)^2 \right) &\leq \Tr\left( \left( A^2 - B^2 \right)e \right) + \Tr\left( \left( B^2 - A^2 \right)e^{\perp} \right)\\
                                             &\leq \Tr\left( \left\vert A^2 - B^2 \right\vert e \right) + \Tr\left( \left\vert B^2 - A^2 \right\vert e^{\perp} \right)\\
                                             &= \Tr\left( \left\vert A^2 - B^2 \right\vert \right).
  \end{align*}
\end{proof}
The Powers--Størmer will allow us to establish the following equivalence, which provides an analogue to the almost invariant vectors/approximate mean conditions for amenability in groups but for injectivity of von Neumann algebras.
\begin{theorem}
  Let $M$ be a $\text{II}_1$ factor. The following are equivalent:
  \begin{enumerate}[(i)]
    \item $M$ has a hypertrace;
    \item there exists $\left( A_i \right)_i\in S^1\left( H \right)$ with $A_i\geq 0$ and $\Tr\left( A_i \right) = 1$ such that $\norm{\left[ x,A_i \right]}_{\Tr}\rightarrow 0$ for all $x\in M$;
    \item there exists $\left( \xi_i \right)_i\in L_2\left( M \right)\otimes L_2\left( M \right)$ such that $\norm{\xi_i}_{\operatorname{HS}} = 1$ and $\norm{\left[ x,\xi_i \right]}_{\operatorname{HS}}\rightarrow 0$ for all $x\in M$.
  \end{enumerate}
\end{theorem}
\begin{proof}
  If (ii) holds, then we realize $S^2\left( L_2\left( M \right) \right)$ as $L_2\left( M \right)\otimes L_2\left( M \right)$, where we map $\xi\otimes\eta\in L_2\left( M \right)\otimes L_2\left( M \right)$ to $\theta_{\xi, J\eta}\in S^2\left( L_2\left( M \right) \right)$, where $J$ is the canonical conjugation isometry. Setting $\xi_i = A_i^2$ and applying the Powers--Størmer inequality gives, for all unitaries $u\in U(M)$
  \begin{align*}
    \norm{u\xi_i u^{\ast}-\xi_i}_{\operatorname{HS}} &\leq \norm{uA_i^2u^{\ast} - A_i^2}_{\Tr}\\
                                                     &\rightarrow 0.
  \end{align*}
  Therefore, (iii) holds.

  Now, if (iii) holds, we let $\varphi\in M^{\ast}$ be the linear functional such that $ \iprod{\cdot \xi_i}{\xi_i}\rightarrow \varphi $ weak*. Then, for all $x\in M$ and $y\in B\left( L_2\left( M \right) \right)$, we have
  \begin{align*}
    \left\vert \varphi\left( xy \right) = \varphi\left( yx \right) \right\vert &= \lim_{i\in I} \left\vert \iprod{xy\xi_i}{\xi_i} - \iprod{yx\xi_i}{\xi_i} \right\vert.
    \intertext{Since $\norm{x\xi_i - \xi_i x}_{\operatorname{HS}}\rightarrow 0$, we have}
                                                                               &= \lim_{i\in I} \left\vert \iprod{xy\xi_i}{\xi_i} - \iprod{y\xi_i x}{\xi_i} \right\vert\\
                                                                               &= \lim_{i\in I} \left\vert \iprod{xy\xi_i}{\xi_i} - \iprod{y\xi_i}{\xi_ix^{\ast}} \right\vert\\
                                                                               &= \lim_{i\in I} \left\vert \iprod{xy\xi_i}{\xi_i} - \iprod{y\xi_i}{x^{\ast}\xi_i} \right\vert\\
                                                                               &= 0.
  \end{align*}
  Thus, $\varphi$ is our desired hypertrace.

  Now, suppose $\varphi$ is a hypertrace.

  We start by observing a general lemma. If $M\subseteq B(H)$ is a von Neumann algebra, then
  \begin{align*}
    S(M) &= \underbrace{\overline{\set{\Tr\left( \cdot A \right) | A\in S^1\left( H \right)_{+},\Tr\left( A \right) = 1}}^{w^{\ast}}}_{\eqcolon K},
  \end{align*}
  where we view $B(H) = \left( S^1\left( H \right) \right)^{\ast}$. Observe that both sides are $w^{\ast}$-compact and convex, meaning that $S(M)$ containing $K$ follows immediately from the fact that $\Tr\left( A \right) = 1$. For inclusion, suppose $\varphi\in S(M)\setminus K$. It follows that there is some $w^{\ast}$-continuous linear functional $L\colon M^{\ast}\rightarrow \C$ and $\alpha < \beta \in \R$ such that $\re\left( L\left( \varphi \right) \right) \geq \alpha > \beta \geq \re\left( L\left( \psi \right) \right)$ for all $\psi\in K$. Since $\left( M^{\ast},w^{\ast} \right)^{^{\ast}} = M$, there is some $a\in M$ such that $L\left( \rho \right) = \rho\left( a \right)$ for all $\rho\in M^{\ast}$. In particular, this means we may write
  \begin{align*}
    \re\left( \varphi\left( a \right) \right) &\geq \alpha\\
                                              &> \beta\\
                                              &\geq \re\left( \Tr\left( Aa \right) \right)
  \end{align*}
  for all $A\in S^1\left( H \right)_{+}$. In particular, this holds for the vector state, whence we may assume
  \begin{align*}
    \varphi\left( \re\left( a \right) \right) &\geq \alpha\\
                                              &> \beta\\
                                              &\geq \iprod{\re\left( a \right)\xi}{\xi}
  \end{align*}
  for all $\xi\in H$ with $\norm{\xi}=  1$. Thus, without loss of generality, $a = a^{\ast}$. Taking the supremum over all satisfactory $\xi$ gives
  \begin{align*}
    \sup_{\norm{\xi} = 1} \iprod{a\xi}{\xi} &= \norm{a_{+}}\\
                                            &\leq \beta\\
                                            &< \alpha\\
                                            &\leq \varphi\left( a \right)\\
                                            &= \varphi\left( a_{+} \right) - \varphi\left( a_{-} \right)\\
                                            &\leq \varphi\left( a_{+} \right)\\
                                            &\leq \norm{a_{+}},
  \end{align*}
  which gives our desired contradiction.

  In order to show (ii), it suffices to show that for all $x_1,\dots,x_n\in M$, we have
  \begin{align*}
    0 &\in \overline{\set{ \left( \left[ x_j,A \right] \right)_{j=1}^{n} | A\in S^1\left( H \right)_{+}, \Tr\left( A \right) = 1 }}^{\norm{\cdot}_{\Tr}}\\
      &\subseteq S^1\left( H \right)^{\left( n \right)}.
  \end{align*}
  Since this is convex, the weak and norm closures coincide, where the weak topology is defined via the trace. By the lemma, if $\varphi\colon B(H)\rightarrow \C$ is a hypertrace for $M$, then there exists $\left( A_i \right)_{i\in I}$ in $S^1\left( H \right)_{+}$ with $\Tr\left( A_i \right) = 1$ for each $i$ such that $\lim_{i}\Tr\left( \cdot A_i \right) = \varphi$.

  It thus suffices to show that for all $B\in B(H)$, we have $\Tr\left( \left[ x_j,A_i \right]B \right) \rightarrow 0$ for each $x_j$. This follows from a computation, where we use traciality and the fact that the trace-class operators are a two-sided ideal, giving
  \begin{align*}
    \Tr\left( \left[ x_j,A_i \right]B \right) &= \Tr\left( x_jA_iB \right)-\Tr\left( A_ix_jB \right)\\
                                              &= \Tr\left( Bx_jA_i \right)-\Tr\left( x_jBA_i \right)\\
                                              &\rightarrow \varphi\left( Bx_j-x_jB \right)\\
                                              &= 0,
  \end{align*}
  since $\varphi$ is a hypertrace. Therefore, for all $j = 1,\dots,n$, we have $\left[ x_j,A_i \right]\rightarrow 0$ weakly, giving our desired result.
\end{proof}
\section{Connes's Trick and Følner Projections}%
Thus far, we have shown an equivalence between amenability/injectivity, the existence of a hypertrace, and certain almost-invariant or almost-central vectors. The next step is to construct an analogue to the Følner sequences, and for this, we must use a noncommutative version of the classic Namioka trick.

It can be observed (and is a classic exercise of real analysis) that for any $s_1,s_2\in \R$, we have
\begin{align*}
  \left\vert s_1-s_2 \right\vert &= \int_{-\infty}^{\infty} \left\vert \chi_{\left( t,\infty \right)}\left( s_1 \right) - \chi_{\left( t,\infty \right)}\left( s_2 \right) \right\vert\:dt.
\end{align*}
Applying various bootstrapping measures, we have that if $\left( X,\mu \right)$ is a $\sigma$-finite measure space, then
\begin{align*}
  \norm{f-g}_{L_1} &= \int_{-\infty}^{\infty} \norm{ \chi_{f^{-1}\left( t,\infty \right)} - \chi_{g^{-1}\left( t,\infty \right)} }_{L_1}\:dt\\
  \norm{f}_{L_1} &= \int_{-\infty}^{\infty} \norm{\chi_{f^{-1}\left( t,\infty \right)}}_{L_1}\:dt
\end{align*}
In particular, if we have $\norm{f-g}_{L_1} < \ve \norm{f}_{L_1}$ for some $\ve > 0$, then there is some $t_0\in \R$ such that
\begin{align*}
  \norm{\chi_{f^{-1}\left( t_0,\infty \right)} - \chi_{g^{-1}\left( t_0,\infty \right)}}_{L_1} < \ve \norm{\chi_{f^{-1}\left( t_0,\infty \right)}}_{L_1}.
\end{align*}
We will seek to obtain a noncommutative version of the Namioka trick. First, we must reduce computations of Hilbert--Schmidt norms to computations of $L_2$ norms of functions.
\begin{proposition}
  Let $ H $ be a Hilbert space, and let $x,y$ be two positive finite rank operators on $H$. There exists a positive Radon measure $\nu$ on $\R^2_{+}$ such that for every pair of Borel complex-valued functions on $\R_{+}$ with $f(0) = g(0) = 0$, we have
  \begin{align*}
    \norm{f(x)-g(y)}_{\operatorname{HS}}^2 &= \int_{\R^2_{+}}^{} \left\vert f\left( \alpha \right)-g\left( \beta \right) \right\vert^{2}\:d\nu\left( \alpha,\beta \right).
  \end{align*}
\end{proposition}
\begin{proof}
  Write
  \begin{align*}
    x &= \sum_{i=1}^{m} \lambda_i e_i\\
    y &= \sum_{j=1}^{n}\mu_j f_j,
  \end{align*}
  where the $\lambda_i$ and $\mu_j$ are the distinct strictly positive eigenvalues of $x$ and $y$ respectively, and the $e_i$ and $f_j$ are the corresponding spectral projections. Set
  \begin{align*}
    e_0 &= 1-\sum_{i=1}^{m}e_i\\
    f_0 &= 1-\sum_{j=1}^{n} f_j,
  \end{align*}
  and let $\Tr$ denote the usual trace on $B\left( H \right)_{+}$.

  Define
  \begin{align*}
    X &= \left( \sigma(x)\times \sigma(y) \right)\setminus \set{\left( 0,0 \right)},
  \end{align*}
  where $\sigma(x)$ denotes the spectrum, and define a measure $\mu$ on $X$ by setting
  \begin{align*}
    \nu\left( \set{\left( \lambda_i,\mu_j \right)} \right) &= \Tr\left( e_if_j \right)\\
    \nu\left( \set{\left( \lambda_i,0 \right)} \right) &= \Tr\left( e_if_0 \right)\\
    \nu\left( \set{\left( 0,\mu_j \right)} \right) &= \Tr\left( e_0f_j \right).
  \end{align*}
  If we have two real-valued Borel functions on $\R_{+}$ with $f(0) = g(0) = 0$, then we have
  \begin{align*}
    f(x) &= \sum_{i=1}^{m}f\left( \lambda_i \right)e_i\\
    g\left( y \right) &= \sum_{j=1}^{n}g\left( \mu_j \right)f_j
  \end{align*}
  are still finite-rank operators, and
  \begin{align*}
    \Tr\left( f\left( x \right)g\left( y \right) \right) &= \int_{\R_{+}^{2}}^{} f\left( \alpha \right)g\left( \beta \right)\:d\nu\left( \alpha,\beta \right)\\
    \Tr\left( f\left( x \right)^2 \right) &= \int_{\R_{+}^{2}}^{} f\left( \alpha \right)^2\:d\nu\left( \alpha,\beta \right)\\
    \Tr\left( g\left( y \right)^2 \right) &= \int_{\R^{2}_{+}}^{} g\left( \beta \right)^2\:d\nu\left( \alpha,\beta \right),
  \end{align*}
  so that
  \begin{align*}
    \norm{f(x)-g(y)}_{\operatorname{HS}}^2 &= \int_{\R^2_{+}}^{} \left\vert f\left( \alpha \right)-g\left( \beta \right) \right\vert^2\:d\nu\left( \alpha,\beta \right).
  \end{align*}
\end{proof}
\begin{theorem}[Connes's Trick]
  Let $x_1,\dots,x_n$ be positive finite-rank elements of $B(H)$, and $0 < \ve < 1$ such that $\norm{x_i-x_1}_{\operatorname{HS}} < \ve \norm{x_1}_{\operatorname{HS}}$. Then, there is $t_0 > 0$ such that
  \begin{align*}
    \norm{\chi_{\left( t_0,\infty \right)}\left( x_i \right) - \chi_{\left( t_0,\infty \right)}\left( x_1 \right)}_{\operatorname{HS}} < \left( 3n\ve \right)^{1/2} \norm{\chi_{\left( t_0,\infty \right)}\left( x_1 \right)}_{\operatorname{HS}}.
  \end{align*}
\end{theorem}
\begin{proof}
  Applying the earlier lemma with $f = g = \chi_{\left( t^{1/2},\infty \right)}$ gives
  \begin{align*}
    \norm{\chi_{\left( t^{1/2},\infty \right)}\left( x \right) - \chi_{\left( t^{1/2},\infty \right)}\left( y \right)}_{\operatorname{HS}}^2 &= \int_{}^{} \left\vert \chi_{\left( t^{1/2},\infty \right)}\left( \alpha \right) - \chi_{\left( t^{1/2},\infty \right)}\left( \beta \right) \right\vert\:d\nu\left( \alpha,\beta \right).
  \end{align*}
  Integrating with respect to $t$, using Fubini's Theorem, and Hölder's Inequality respectively gives
  \begin{align*}
    \int_{0}^{\infty} \norm{\chi_{\left( t^{1/2},\infty \right)}\left( x \right) - \chi_{\left( t^{1/2},\infty \right)}\left( y \right)}_{\operatorname{HS}}^2\:dt &= 
    \int_{\R^2_{+}}^{} \left( \int_{0}^{\infty} \left\vert \chi_{\left( t^{1/2},\infty \right)}\left( \alpha \right) - \chi_{\left( t^{1/2},\infty \right)}\left( \beta \right) \right\vert\:dt \right)\:d\nu\left( \alpha,\beta \right)\\
                                                                 &= \int_{\R_{+}^2}^{} \left\vert \alpha^2-\beta^2 \right\vert\:d\nu\left( \alpha,\beta \right)\\
                                                                 &\leq \left( \int_{\R^2_{+}}^{} \left\vert \alpha-\beta \right\vert^2\:d\nu\left( \alpha,\beta \right) \right)^{1/2} \left( \int_{\R^2_{+}}^{} \left\vert \alpha + \beta \right\vert^2\:d\nu\left( \alpha,\beta \right) \right)\\
                                                                      &= \norm{x-y}_{\operatorname{HS}}\norm{x+y}_{\operatorname{HS}}
  \end{align*}
  For any $\ve < 1$, the triangle inequality gives that
  \begin{align*}
    \norm{x_j + x_1}_{\operatorname{HS}} &\leq 3\norm{x_1}_{\operatorname{HS}},
  \end{align*}
  so we have
  \begin{align*}
    \int_{0}^{\infty} \sum_{i=1}^{n}\norm{\chi_{\left( t^{1/2},\infty \right)}\left(x_j\right)-\chi_{\left( t^{1/2},\infty \right)}\left( x_1 \right)}_{\operatorname{HS}}\:dt &< 3n\ve \norm{x_1}_{\operatorname{HS}}^2\\
                                                                                                                                                                                                &= 3n\ve \int_{0}^{\infty} \norm{\chi_{\left( t^{1/2},\infty \right)}\left( x_1 \right)}_{\operatorname{HS}}^2\:dt.
  \end{align*}
  Hence, the Namioka trick gives some $t_0 > 0$ with
  \begin{align*}
    \sum_{i=1}^{n} \norm{\chi_{\left( t_0,\infty \right)}\left( x_i \right) - \chi_{\left( t_0,\infty \right)}\left( x_1 \right)}_{\operatorname{HS}}^2 &< 3n\ve\norm{\chi_{\left( t_0,\infty \right)}\left( x_1 \right)}_{\operatorname{HS}}^2.
  \end{align*}
\end{proof}
The theorem applies in any semifinite von Neumann algebra $\left( M,\Tr \right)$ instead of just $\left( B\left( H \right),\Tr \right)$ upon using one's preferred faithful representation.
\begin{proposition}
  Let $\left( M,\tau \right)$ be a tracial von Neumann algebra, and let $\xi,\eta\in L_2\left( M \right)_{+}$. There exists a positive Radon measure $\nu$ on $\R^2_{+}$ such that for all Borel functions $f,g$ on $\R_{+}$ with $f(0) = g(0) = 0$ and $f\left( \xi \right),g\left( \eta \right)\in L_2\left( M \right)$, the maps $\left( \alpha,\beta \right)\mapsto f\left( \alpha \right)$ and $\left( \alpha,\beta \right)\mapsto g\left( \beta \right)$ are square-integrable, with
  \begin{align*}
    \norm{f\left( \xi \right)-g\left( \eta \right)}_{L_2(M)}^2 &= \int_{\R^2_{+}}^{} \left\vert f\left( \alpha \right)-g\left( \beta \right) \right\vert^2\:d\nu\left( \alpha,\beta \right).
  \end{align*}
\end{proposition}
\begin{theorem}
  Let $\left( M,\tau \right)$ be a tracial von Neumann algebra, and let $\xi_1,\dots,\xi_n\in L_2\left( M \right)_{+}$. Let $0 < \ve < 1$ be such that
  \begin{align*}
    \norm{\xi_i - \xi_1}_{L_2(M)} &< \ve\norm{\xi_1}_{L_2(M)}.
  \end{align*}
  Then, there is $t_0 > 0$ such that
  \begin{align*}
    \norm{\chi_{\left( t_0,\infty \right)}\left( \xi_i \right) - \chi_{\left( t_0,\infty \right)}\left( \xi_1 \right)}_{L_2(M)} &< \left( 3n\ve \right)^{1/2} \norm{\chi_{\left( t_0,\infty \right)}\left( \xi_1 \right)}_{L_2(M)}.
  \end{align*}
\end{theorem}
The main benefit we will find from the Connes trick is the existence of Følner projections.
\begin{corollary}
  If $M$ is amenable, then there exist projections $p_i\in B\left( L_2\left( M \right) \right)$ with $0 < \Tr\left( p_i \right) < \infty$ and
  \begin{align*}
    \frac{\norm{\left[ x,p_i \right]}_{\Tr}}{\Tr\left( p_i \right)} &\rightarrow 0
      \intertext{and}
    \frac{\Tr\left( xp_i \right)}{\Tr\left( p_i \right)} &\rightarrow \tau\left( x \right)
  \end{align*}
  for all $x\in M$, where $\Tr$ denotes the trace on $B\left( L_2\left( M \right) \right)$.
\end{corollary}
\begin{proof}
  Given $\ve > 0$ and some subset $F\subseteq U(M)$ finite, there exists $A\in S^2\left( H \right)$ with $\norm{A}_{\operatorname{HS}} = 1$ and $\sum_{u\in F} \norm{uAu^{\ast}-A}_{\operatorname{HS}}^2 < \ve$.

  This gives
  \begin{align*}
    \int_{0}^{\infty} \sum_{u\in F}\norm{u\chi_{\left( t^{1/2},\infty \right)}(A)u^{\ast}-\chi_{\left( t^{1/2},\infty \right)}\left( A \right)}_{\operatorname{HS}}^2\:dt &\leq 2\sum_{u\in F} \norm{uAu^{\ast}-A}_{\operatorname{HS}}^2\\
                                                                                                                                                                       &< 2\ve^2\\
                                                                                                                                                                                                                                                                                                                                                                                                         &= 2\ve^2 \int_{0}^{\infty} \norm{\chi_{\left( t^{1/2},\infty \right)}\left( A \right)}\:dt,
  \end{align*}
  so there exists $t_0$ such that $p = \chi_{\left( t_0^{1/2},\infty \right)}\left( A \right)$ has
  \begin{align*}
    \sum_{u\in F} \norm{upu^{\ast} - p}_{\operatorname{HS}}^2 &< 2\ve^2 \Tr\left( p \right).
  \end{align*}
  Therefore, we have
  \begin{align*}
    \norm{\left[ u,p \right]}_{\Tr} &\leq 2\ve^2 \Tr(p),
  \end{align*}
  which gives our desired net.

  Moreover, if we have
  \begin{align*}
    \frac{\Tr\left( \cdot p_i \right)}{\Tr\left( p_i \right)} &\rightarrow \psi
  \end{align*}
  for some $\psi\in M^{\ast}$, we have that $\psi$ is a trace, meaning that $\psi$ is necessarily the trace.
\end{proof}
\section{The Local Approximation Property}%
\begin{definition}
  Let $\left( M,\tau \right)$ be a finite von Neumann algebra. We say $\left( M,\tau \right)$ has the \textit{local approximation property} if, for all $\ve > 0$, there exists $e\in P(M)$ such that for all finite $F\subseteq B_M$, there exists $q\in P(M)$ with $q\leq e$ and a finite-dimensional matrix algebra $Q$ in $qMq$ such that
  \begin{enumerate}[(i)]
    \item $\norm{\left[ q,eve \right]}_{\tau} < \ve\norm{q}_{\tau}$ for all $v\in F$;
    \item $\norm{qvq - E\left( qvq \right)}_{\tau} < \ve\norm{q}_{\tau}$, where $E\colon qMq\rightarrow Q$ is the trace-preserving conditional expectation on $qMq$ onto $Q$.
  \end{enumerate}
\end{definition}
\begin{theorem}
  If $\left( M,\tau \right)$ has the local approximation property, then $M$ is hyperfinite.
\end{theorem}
\begin{proof}
  Let $\ve > 0$ and $F\subseteq B_M$ finite. Suppose that $q$ satisfies (i) and (ii) in the case of $e = 1$.

  We observe that for all $v\in H$, the vectors
  \begin{align*}
    \left( 1-q \right)vq,qv\left( 1-q \right),qvq-E(v),\left( 1-q \right)v\left( 1-q \right)
  \end{align*}
  are mutually orthogonal. Therefore, for all $v\in F$, we have
  \begin{align*}
    \norm{v - \left( 1-q \right)v\left( 1-q \right) - E\left( qvq \right)}_{\tau}^2 &= \norm{qvq - E\left( v \right)}_{\tau}^2 + \norm{\left( 1-q \right)vq}_{\tau}^2 + \norm{qv\left( 1-q \right)}_{\tau}^2\\
                                                                                    &= \norm{qvq - E\left( v \right)}_{\tau}^2 + \norm{\left[ v,q \right]q}_{\tau}^2 + \norm{q\left[ v,q \right]}_{\tau}^2\\
                                                                                    &< 3\ve^2\norm{q}_{\tau}^2.
  \end{align*}
  Consider the partially ordered set $ \mathcal{F}\subseteq \set{\left( q,Q \right) | 0\neq q\in P(M), Q\leq qMq} $ that has $q_1q_2 = 0$ for all $\left( q_1,Q_1 \right),\left( q_2,Q_2 \right)\in \mathcal{F}$ and $Q$ finite-dimensional for all $\left( q,Q \right)\in \mathcal{F}$; using the local approximation property, we have
  \begin{align*}
    \norm{x-\left( 1-\widetilde{q} \right)x\left( 1-\widetilde{q} \right) - E_{ \widetilde{Q} }\left( qxq \right)}_{\tau} &< \ve\norm{\widetilde{q}}_{\tau},
  \end{align*}
  where we set
  \begin{align*}
    \widetilde{q} &= \sum_{\left( q,Q \right)\in \mathcal{F}}q\\
    \widetilde{Q} &= \bigoplus_{\left( q,Q \right)\in \mathcal{F}} Q.
  \end{align*}
  The family $\mathcal{F}$ is nonempty, partially ordered by inclusion, and every increasing sequence in $\mathcal{F}$ has an upper bound in $ \mathcal{F} $ by the normality of the conditional expectation.

  Therefore, we have a maximal family $ \set{\left( q_j,Q_j \right)}_{j\in J}\subseteq \mathcal{F} $, and set
  \begin{align*}
    q' &= \sum_{j\in J}q_j\\
    Q' &= \bigoplus_{j\in J}Q_j.
  \end{align*}
  We will show that $q' = 1$. Suppose not. Then, $1-q'\neq 0$, so there is some nonzero projection $p\leq 1-q'$ and a finite-dimensional subalgebra $Q_p\subseteq pMp$ satisfying
  \begin{align*}
    \norm{\left[ p,\left( 1-q \right)x\left( 1-q \right) \right]}_{\tau} &< \ve\norm{p}_{\tau}\\
    \norm{pxp - E_{Q_p}\left( pxp \right)}_{\tau} &< \ve \norm{p}_{\tau}.
  \end{align*}
  Then, from the earlier discussion, we have
  \begin{align*}
    \norm{\left( 1-q \right)x\left( 1-q \right) - \left( 1-\left( q + p \right) \right)x\left( 1-\left( q+p \right) \right) - E_{Q_p}\left( pxp \right)}_{\tau} &< \ve\sqrt{3}\norm{p}_{\tau}.
  \end{align*}
  Set
  \begin{align*}
    Q_0 &= Q' \oplus Q_p\\
    \widetilde{F}' &= \set{q_j}_{j\in J}\cup\set{p}.
  \end{align*}
  This gives
  {\small\begin{align*}
    \norm{x-\left( 1-\left( q' + p \right) \right)x\left( 1-\left( q'+p \right) \right) - E_{Q_0}\left( \left( q'+p \right)x\left( q'+p \right) \right)}_{\tau}^2 &= \norm{x- \left( 1-\left( q'+p \right) \right)x\left( 1-\left( q'+p \right) \right) - E_{Q_0}\left( q' xq' \right) + E_{Q_p}\left( pxp \right)}_{\tau}^2\\
                                                                                                                                                                          &= \norm{\left( 1-q' \right)x\left( 1-q' \right) - \left( 1-\left( q'+p \right) \right)x\left( 1-\left( q'+p \right) \right) - E_{Q_p}\left( pxp \right)}_{\tau}^2\\
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                          &+ \norm{x-\left( 1-q' \right)x\left( 1-q' \right) - E_{Q_0}\left( q'xq' \right)}_{\tau}^2\\

                     &< C \ve\norm{q' + p}_{\tau}^2
  \end{align*}}
  for some constant $C$. This contradicts the maximality of our family $\set{\left( q_j,Q_j \right)}_{j\in J}$. Since $ Q' $ is hyperfinite, this gives our desired result.
\end{proof}
