\section{Definition}%
\begin{frame}[allowframebreaks]
  \frametitle{Noncommutative Probability}
  \begin{itemize}
    \item A \textbf{$C^{\ast}$-probability space} is a pair $\left( \mathcal{A},\varphi \right)$, where $\mathcal{A}$ is a (unital) $C^{\ast}$-algebra --- a complete normed $\ast$-algebra satisfying $\norm{a^{\ast}a} = \norm{a}^2$ --- and a linear functional $\varphi\colon \mathcal{A}\rightarrow \C$ satisfying $\varphi(1) = 1$ and $\varphi\left( a^{\ast}a \right)\geq 0$ for all $a\in A$, with equality only if $a = 0$. We call elements $a\in \mathcal{A}$ \textit{noncommutative random variables}.
    \item If $a\in \mathcal{A}$ is an element of a $C^{\ast}$-probability space, the \textit{moments} of $a$ are the values $\set{\varphi\left( a^n \right) | n=0,1,\dots}$.
    \item If $\set{\mathcal{A}_i}_{i\in I}\subseteq \mathcal{A}$ is a family of unital $C^{\ast}$-subalgebras, the family is said to be classically independent if the algebras commute and mixed moments factorize. That is, if $a\in \mathcal{A}_i$ and $b\in \mathcal{A}_j$ with $i\neq j$, then
      \begin{align*}
        \varphi\left( a^{n}b^{m} \right) &= \varphi\left( a^{n} \right)\varphi\left( b^{m} \right).
      \end{align*}
    \item If $\set{\mathcal{A}_i}_{i\in I}\subseteq \mathcal{A}$ is a family of unital $C^{\ast}$-subalgebras, the family is said to be \textit{freely independent} if $\varphi\left( a_1\cdots a_k \right) = 0$ whenever
      \begin{itemize}
        \item $k > 0$;
        \item $a_j\in \mathcal{A}_{i_j}$ for each $j$;
        \item $\varphi\left( a_j \right) = 0$ for each $j$;
        \item $i_1\neq i_2$, $i_2\neq i_3$, etc.
      \end{itemize}
    \item If the $\mathcal{A}_i = C^{\ast}\left( a_i \right)$ are generated by single elements, then the elements $\set{a_i}_{i\in I}$ are called (freely) independent if the $\mathcal{A}_i$ are (freely) independent.
    \item If $\left( \mathcal{A}_N,\varphi_N \right)$ is a sequence of $C^{\ast}$-probability spaces, and $\left( \mathcal{A},\varphi \right)$ is another $C^{\ast}$-probability space, then a sequence $\left( a_N \right)_N$ with each $a_N\in \mathcal{A}_N$ is said to \textit{converge in distribution} to $a\in \mathcal{A}$ if, for each $k\in \N$,
      \begin{align*}
        \lim_{N\rightarrow\infty} \varphi_N\left( a_N^{k} \right) &= \varphi\left( a^k \right).
      \end{align*}
  \end{itemize}
\end{frame}
\begin{frame}[allowframebreaks]
  \frametitle{Combinatorics}
  \begin{itemize}
    \item If $[r] = \set{1,\dots,r}$ is some set, a \textit{partition} $\pi = \set{V_1,\dots,V_s}$ of the set is a collection of ordered and disjoint subsets (known as blocks) of $[r]$ whose union is equal to $[r]$. The partitions of $[r]$ are denoted $P(r)$.
    \item The partition is called \textit{noncrossing} if, for any blocks $V_i = \left( v_1,\dots,v_n \right)$ and $V_j = \left( w_1,\dots,w_m \right)$, we have $w_k < v_1 < w_{k+1}$ if and only if $w_k < v_n < w_{k+1}$ for each $k=1,\dots,m-1$. The non-crossing partitions of $[r]$ are denoted $NC(r)$.
    \item If the size of each block in $\pi$ is $2$, then we call the partition a pair partition. The set of pair partitions of $[2k]$ is denoted $P_2\left( 2k \right)$. The noncrossing pair partitions are denoted $NC_2\left( 2k \right)$.
  \end{itemize}
\end{frame}
\section{Classical Central Limit Theorem}%
\begin{frame}
  \frametitle{Measures and Moments}
  A probability measure $\mu$ on $\R$ is said to be determined by its moments if the sequence of moments
  \begin{align*}
    m_n &= \int_{\R}^{} t^n\:d\mu(t)
  \end{align*}
  is unique --- i.e., if $\nu$ is a measure that has
  \begin{align*}
    m_n &= \int_{\R}^{} t^{n}\:d\nu(t)
  \end{align*}
  for each $n$, then $\nu = \mu$.\newline

  For instance, compactly supported measures are determined by their moments.
\end{frame}
\begin{frame}
  \frametitle{The Normal Distribution}
  \textbf{Fact:} The standard normal distribution,
  \begin{align*}
    d\mu &= \frac{1}{\sqrt{2\pi}} e^{-t^2/2}\:dt
  \end{align*}
  is determined by its moments \cite[Example 30.1]{billingsley_probability_and_measure}. The moments are given by
  \begin{align*}
    \mathbb{E}\left[ X^{2k} \right] &= \left( 2k-1 \right)!!\\
    \mathbb{E}\left[ X^{2k+1} \right] &= 0.
  \end{align*}
  Note: $\left( 2k-1 \right)!! = \left\vert P_2\left( 2k \right) \right\vert$.
\end{frame}
\begin{frame}[allowframebreaks]
  \frametitle{Classical Independence and Pair Partitions}
  If $\left( a_N \right)_{N\in\N}$ is a family of (self-adjoint) independent classical random variables with $\varphi\left( a_k \right) = 0$ and $\varphi\left( a_k^2 \right) = 1$ for each $k$, then we may express
  \begin{align*}
    \varphi\left( \left( a_1 + \cdots + a_N \right)^{n} \right) &= \sum_{ \mathbf{r}\in [N]^{n} } \varphi\left( a_{ r_1 }\cdots a_{r_n} \right),
  \end{align*}
  where $ \left( r_1,\dots,r_n \right) $ denotes all $n$-tuples (repeats allowed) contained in $\set{1,\dots,N}$.\newline

  Since $a_1,\dots,a_N$ are classically independent, the value $\varphi\left( a_{r_1}\cdots a_{r_n} \right)$ depends exclusively on the level sets of the tuple $\left( r_1,\dots,r_n \right)$ (i.e., the values at different indices with identical values in $[N]$).

  These indices determine a partition $\pi = \set{V_1,\dots,V_s}$ of $[n]$. Denote this equivalence by $\left( r_1,\dots,r_n \right)\hat{=}\pi$. This allows us to write
  \begin{align*}
    \varphi\left( \left( a_1+\cdots+a_N \right)^{n} \right) &= \sum_{\pi\in P(n)} K_{\pi}A_{\pi}^{N},
  \end{align*}
  where $K_{\pi}$ is the value of $\varphi\left( a_{r_1}\cdots a_{r_n} \right)$, satisfying $\left( r_1,\dots,r_n \right) \hat{=}\pi$, and
  \begin{align*}
    A_{\pi}^{N} &= \left\vert \set{\left( r_1,\dots,r_n \right)\hat{=}\pi | \left( r_1,\dots,r_n \right)\in [N]^{n}} \right\vert.
  \end{align*}
  By independence, if there is a block $V_k = \set{\ell}$, then we have
  \begin{align*}
    \varphi\left( a_{r_1}\cdots a_{\ell}\cdots a_{r_n} \right) &= \varphi\left( a_{\ell} \right)\varphi\left( a_{r_1}\cdots\widehat{a_{\ell}}\cdots a_{r_n} \right)\\
                                                               &= 0.
  \end{align*}
  In particular, this means that we only need to consider partitions $\pi = \set{V_1,\dots,V_s}$ with $s\leq n/2$ (each block must have at least two elements).\newline

  The value of $A_{\pi}^{N}$ is given by counting available common indices between $1,\dots,N$, giving
  \begin{align*}
    A_{\pi}^{N} &= \prod_{i=0}^{\left\vert \pi \right\vert-1} \left( N-i \right).
  \end{align*}
  Asymptotically, this grows like $N^{\left\vert \pi \right\vert}$.\newline

  This gives
  \begin{align*}
    \lim_{N\rightarrow\infty} \varphi\left( \left( \frac{a_1 + \cdots + a_N}{\sqrt{N}} \right)^{n} \right) &= \lim_{N\rightarrow\infty}\sum_{\pi\in P(n)} K_{\pi}N^{\left\vert \pi \right\vert-(n/2)}.
  \end{align*}
  The only non-vanishing values are ones where $\left\vert \pi \right\vert = n/2$ --- i.e., pair partitions. Independence then gives that $K_{\pi} = 1$ for any pair partition, so we get
  \begin{align*}
    \lim_{N\rightarrow\infty} \varphi\left( \left( \frac{a_1 + \cdots + a_N}{\sqrt{N}} \right)^{2k} \right) &= \left\vert P_2\left( 2k \right) \right\vert\\
    \lim_{N\rightarrow\infty} \varphi\left( \left( \frac{a_1 + \cdots + a_N}{\sqrt{N}} \right)^{2k+1} \right) &= 0,
  \end{align*}
  which are exactly the moments of the standard normal distribution.
\end{frame}
\section{Free Independence}%
\begin{frame}
  \frametitle{The Semicircular Distribution}
  The \textit{semicircular distribution} is the probability distribution $d\mu = f(t)\:dt$, where
  \begin{align*}
    f(t) &= \begin{cases}
      \frac{1}{2\pi}\sqrt{4-t^2}, & \text{whenever }-2 \leq t \leq 2\\
      0, & \text{else}.
    \end{cases}
  \end{align*}
  The moments of the semicircular distribution are given by
  \begin{align*}
    \mathbb{E}\left[ X^{2k} \right] &= \frac{1}{k+1} {2k\choose k},\text{ the Catalan number $C_k$}\\
    \mathbb{E}\left[ X^{2k + 1} \right] &= 0.
  \end{align*}
  Note: $C_k = \left\vert NC_2\left( 2k \right) \right\vert = \left\vert NC(k) \right\vert$.
\end{frame}
\begin{frame}[allowframebreaks]
  \frametitle{Free Independence and Noncrossing Pair Partitions}
  We want to develop an analogous approach to the classical central limit theorem for when our collection $\left( a_N \right)_{N}$ is freely independent, still satisfying $\varphi\left( a_k \right) = 0$ and $\varphi\left( a_k^2 \right) = 1$ for each $k$.\newline

  Similar to the case of classical independence, the moments of the sum can be expressed as a sum over partitions,
  \begin{align*}
    \varphi\left( \left( a_1 + \cdots + a_N \right)^{n} \right) &= \sum_{ \mathbf{r}\in \left[ N \right]^{n} } \varphi\left( a_{r_1}a_{r_2}\cdots a_{r_n} \right)\\
                                                                &= \sum_{\pi\in P(n)} L_{\pi}A_{\pi}^{N},
  \end{align*}
  but the values $L_{\pi}$ (the mixed moment determined by the partition $\pi$) will not be the same as the values $K_{\pi}$, since we are dealing with free independence instead.
  \begin{enumerate}[(i)]
    \item If $\left( r_1,\dots,r_n \right)\hat{=}\pi$ has $r_1\neq r_2$, $r_2\neq r_3$, etc., then the definition of free independence gives $\varphi\left( a_{r_1}a_{r_2}\cdots a_{r_n} \right) = 0$.
    \item If there is some index $m$ such that $r = r_m = r_{m+1}$, then applying a series of factorization rules for freely independent variables gives
      \begin{align*}
        L_{\pi} &= \varphi\left( a_{r_1}\cdots a_{r}a_{r}\cdots a_{r_n} \right)\\
                &= \varphi\left( a_{r_1}\cdots a_{r_{m-1}}a_{r_{m+2}}\cdots a_{r_n} \right)\varphi\left( a_r^2 \right)\\
                &= \varphi\left( a_{r_1}\cdots a_{r_{m-1}}a_{r_{m+2}}\cdots a_{r_n} \right).
      \end{align*}
  \end{enumerate}
  That is, $L_{\pi}$ equals either $0$ or $1$ upon completion of this reduction process. All we need to do now is figure out when $L_{\pi} = 0$ (or $1$).\newline

  If $L_{\pi} = 0$, then we eventually find a pair partition $\tau$ of $m > 0$ elements such that $\varphi\left( a_{s_1}\cdots a_{s_m} \right)$ has level sets corresponding to $\tau$, and there is no index $r = s_{k} = s_{k+1}$ (else we would be able to extract this block and iterate the process again).\newline

  If $\set{t_1,t_2}$, with $t_1 < t_2$, is a block of this partition with corresponding elements $s_i$ and $s_j$, then we cannot have $s_i = s_j$ (or else we would again perform the reduction process).\newline

  Therefore, we must have some element $t_1 < t_1' < t_2$ with corresponding pair $t_2'$.\newline

  If $t_2' < t_1$ or $t_2 < t_2'$, then we observe that $\tau$ is a crossing pair partition. Else, if $t_1' < t_2' < t_2$, then we use the same reasoning process on the block $\set{t_1',t_2'}$ to eventually find a crossing pair.\newline

  Therefore, we have
  \begin{align*}
    \lim_{N\rightarrow\infty} \varphi\left( \left( \frac{a_1 + \cdots + a_N}{\sqrt{N}} \right)^{n} \right) &= \lim_{N\rightarrow\infty} \sum_{\pi\in P(n)} L_{\pi}N^{\left\vert \pi \right\vert-(n/2)},
  \end{align*}
  where $L_{\pi}$ is non-vanishing exclusively on noncrossing pair partitions. Therefore, we have
  \begin{align*}
    \lim_{N\rightarrow\infty} \varphi\left( \left( \frac{a_1 + \cdots + a_N}{\sqrt{N}} \right)^{2k} \right) &= \left\vert NC_2\left( 2k \right) \right\vert\\
    \lim_{N\rightarrow\infty} \varphi\left( \left( \frac{a_1 + \cdots + a_N}{\sqrt{N}} \right)^{2k+1} \right) &= 0,
  \end{align*}
  which are the moments of the semicircular distribution.
\end{frame}
